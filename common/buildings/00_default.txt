	}
	
	chance = {
		modifier = {
			add = 3
			NOT = { has_building = port_building }
			OR = {
				has_city_status = yes
				owner = { num_of_ports < 4 }
			}
		}
		modifier = {
			add = 5
			has_building = port_building
		}
		modifier = {
			add = 15
			owner = { num_of_ports < 1 }
		}
		modifier = {
			add = 10
			total_population >= 35
			has_building = port_building
			has_city_status = yes
			owner = { 
				num_of_ports >= 6 
				rank > regional_power
			}
		}
	}
	
}
