befriend = {
	on_other_nation = no 
	on_own_nation = no
	close_ui = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_befriend"
	
	potential_trigger = {
		hidden:scope:actor = { exists = current_ruler }
		hidden:scope:target = {
			employer = { exists = current_ruler }
		}
		befriend_potential_trigger = yes
	}
	
	allowed_trigger = {
		befriend_allowed_trigger = yes
	}
	
	effect = {
		befriend_effect = yes
	}
	
	ai_will_do = {
		modifier = {
			add = 1
			trigger = {				
				power_base >= 5
			}
		}
	}
}

defriend = {
	on_other_nation = no 
	on_own_nation = no 
	close_ui = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_banish"
	
	potential_trigger = {
		hidden:scope:actor = { exists = current_ruler }
		hidden:scope:target = {
			employer = { exists = current_ruler }
			is_friend = scope:actor.current_ruler
		}
	}
	
	allowed_trigger = {
		trigger_if = {
			limit = {
				exists = scope:target.employer
				scope:target.employer = scope:actor
			}
			scope:actor = {
				political_influence >= 30
			}
		}
	}
	
	effect = {
		if = {
			limit = {
				exists = scope:target.employer
				scope:target.employer = scope:actor
			}
			scope:actor = {
				add_political_influence = -30
			}
		}
		scope:target = {
			remove_friend = scope:actor.current_ruler
		}
	}
}
