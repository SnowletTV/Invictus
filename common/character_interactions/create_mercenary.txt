create_mercenary = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_economy"
	
	potential_trigger = {
		hidden:scope:target = {
			is_alive = yes
			employer = scope:actor
			is_mercenary = no
			is_ruler = no
			trigger_if = {
				limit = {
					scope:actor = {
						gender_equality = no
					}
				}
				is_female = no
			}
		}
		hidden:scope:actor = {
			OR = {
				is_monarchy = yes
				has_variable = celtic_mercenary_allowance
			}
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			is_adult = yes
			from_ruler_family = yes
			loyalty >= 60
			is_general = yes
			any_character_unit = {
				unit_size >= 10
			}
			NAND = {
				employer = {
					is_monarchy = yes
				}
				is_co_ruler = yes
			}
			NOT = {
				has_character_modifier = disloyal_successor_cmod
			}
			custom_tooltip = {
				text = "no_other_mercs_here"
				location = {
					any_unit_in_province = {
						count = 1
						is_army = yes
					}
					NOT = {
						is_sea = yes
					}
					has_owner = yes
					NOT = {
						#owned_or_subject_owned = scope:actor
					}
				}
			}
		}
		scope:actor = {
			war = no
			has_civil_war = no
		}
	}
	

	
	effect = {
		scope:target = {
			if = {
				limit = {
					is_primary_heir = yes
				}
				set_variable = {
					name = next_in_succession
					value = scope:actor.current_ruler
				}
			}
			set_variable = {
				name = adventuring_from
				value = scope:actor
			}
			set_variable = {
				name = wont_go_home
				days = 1825
			}
			hidden_effect = {
				add_martial = -5
				move_country = c:MER
			}
		}
		if = {
			limit = {
				scope:actor = {
					has_variable = celtic_mercenary_allowance
				}
			}
			set_variable = {
				name = celtic_mercenary_created
				value = 1
			}
		}
		custom_tooltip = create_mercenary_may_return_some_day_tt
		
	}
	
}
