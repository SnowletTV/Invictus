get_divorced = { 

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_arrange_marriage"
	
	message = no
	
	
	potential_trigger = {
		scope:actor = {
			OR = {
				is_monarchy = yes 
				is_tribal = yes
			}
			tyranny <= 99
			is_ai = no
		}
		hidden:scope:target = {
			is_alive = yes
			employer = scope:actor
			is_married = yes
			spouse = {
				is_alive = yes
				is_ruler = yes 
			}
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			is_married = yes
			is_adult = yes
			prisoner = no			
		}
	}
	
	
	effect = {
		scope:target = {
			divorce_character = spouse
			add_loyalty = divorced_by_ruler
			hidden_effect = {
				family = {
					every_family_member = {
						limit = {
							employer = scope:actor
							is_ruler = no
						}
						add_loyalty = family_forced_to_divorce 
					}
				} 
			}
		}

		scope:actor = { 
			trigger_event = character_interactions_inv.2
			if = {
				limit = { 
					scope:target = { is_minor_character = no }
				}
				custom_tooltip = character_interactions_1_tooltip
				show_as_tooltip = {
					random_character = {
						limit = {
							has_same_family = scope:target
						}
						add_loyalty = family_forced_to_divorce
					}
				}
			}
			add_tyranny = {
				value = scope:target.power_base
				divide = 2
				min = 5
				max = 20
			}
			add_political_influence = -20
		}
		scope:actor.current_ruler = { 
			add_popularity = -20
			#add_prominence = -30
		}
	}	
}
