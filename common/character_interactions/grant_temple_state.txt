	on_own_nation = yes

	potential_trigger = {
		scope:actor = {
			is_monarchy = yes
		}		
		hidden:scope:target = {#Zeal
			employer = scope:actor
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
			is_alive = yes
			is_mercenary = no
			prisoner = no
			is_female = no
			OR = {
			   has_office = office_high_priest_monarchy
			   has_office = office_pontifex
			   has_office = office_high_priest
			   zeal >= 8
			}
		}
	}

	allowed_trigger = {
		trigger_if = {
			limit = {
				NOR = {
					can_make_temple_state = yes
					invention = temple_estates_inv
				}
			}
			invention = temple_estates_inv
		}
		
		trigger_if = {
			limit = {
				has_variable = make_temple_state_cooldown
			}
			custom_tooltip = {
				text = temple_state_cooldown_tooltip
				always = no
			}
		}

		trigger_if = {
			limit = {
		        any_owned_province = {
					NOR = {
						own_religious_center_province_trigger = yes
						has_province_modifier = temple_state_modifier 
					}
		        }
			}
			custom_tooltip = {
				text = own_valid_temple
		        any_owned_province = {
					OR = {
						own_religious_center_province_trigger = yes
						has_province_modifier = temple_state_modifier 
						is_holy_site = yes
					}
				}
			}
		}
		scope:actor = {
			can_pay_price = grant_temple_estate_price
			war = no
			has_civil_war = no
			is_subject = no
		}
	}
	
	effect = {#Make event instead
		scope:actor = {
			pay_price = grant_temple_estate_price
			hidden_effect = {
				scope:actor = {
					every_character = {
						limit = {
							has_same_family = scope:target
							is_ruler = no
							NOT = {
								this = scope:target
							}
						}
						add_loyalty = family_made_temple_state
					}
				}	
			}
			hidden_effect = {
				if = {
					limit = {
				        any_owned_province = {
							has_province_modifier = temple_state_modifier 
				        }
					}
			        random_owned_province = {
			            limit = {
							has_province_modifier = temple_state_modifier 
			            }
						save_scope_as = temple_state_province
			        }
				}
				else = {
			        random_owned_province = {
			            limit = {
							OR = {
								own_religious_center_province_trigger = yes
								has_province_modifier = temple_state_modifier 
							}       
			            }
						save_scope_as = temple_state_province
			        }
				}
				else = {
			        random_owned_province = {
			            limit = {
							is_holy_site = yes    
			            }
						save_scope_as = temple_state_province
			        }
				}
				scope:temple_state_province = {
					create_country = { save_scope_as = new_temple_state }
				}
				scope:target = {
					random_list = {
						100 = {
							add_trait = obsessive
						}
						100 = {
							add_trait = pious
						}
						100 = {
							add_trait = self_controlled
						}
						100 = {
							add_trait = zealous
						}
						100 = {
							add_trait = xenophobic
						}
					}
					move_country = scope:new_temple_state
					remove_character_modifier = foreign_citizen
					set_home_country = scope:new_temple_state
				}
				scope:new_temple_state = {
					create_family = scope:target
					set_as_ruler = scope:target				
				}
				scope:target = {
					if = {
						limit = {
							is_married = yes
							spouse = {
								employer = root
								prisoner = no
							}
						}
						spouse = {
							remove_all_positions = yes
							move_country = scope:new_temple_state
							remove_character_modifier = foreign_citizen
							set_home_country = scope:new_temple_state
						}
					}
					if = {
						limit = {
							num_of_children > 0
							any_child = {
								employer = root
								prisoner = no
							}
						}
						every_child = {
							limit = {
								employer = root
								prisoner = no
							}
							remove_all_positions = yes
							move_country = scope:new_temple_state
							remove_character_modifier = foreign_citizen
							set_home_country = scope:new_temple_state
							set_family = scope:target.family
						}
					}
				}
			}

			scope:actor = {
				show_as_tooltip = {
					scope:new_temple_state = {
						create_country = {}
					}
				}
				custom_tooltip = made_temple_state
				hidden_effect = {

						limit = {
							root = {

							}

						limit = {
						}

						capital_scope = {

								limit = {
									NOT = { 
										has_province_modifier = temple_state_modifier

								}
								add_permanent_province_modifier = {
									name = temple_state_modifier
									duration = -1
								}
							}
						}
					}
				}
				hidden_effect = {
					scope:actor = {
						make_subject = {
							target = scope:new_temple_state
							type = temple_state
						}
		            	trigger_event = { id = temple_state_event.2 }
					}					
				}
			}
			custom_tooltip = family_made_temple_state_tt
			set_variable = {#can make variable longer to prevent AI spam if that is an issue
				name = make_temple_state_cooldown
				days = 9125
			}
			custom_tooltip = temple_state_cooldown_effect_tooltip	
		}
	}
	
	ai_will_do = {

		}
		modifier = {
			add = 2.5 #%
			trigger = {
				employer = {
					towards_civil_war = yes
				}
			}
		}
