impose_corruption_sanctions = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Unique/sfx_ui_character_bribe"
	
	potential_trigger = {
		scope:target = {
			is_alive = yes
			is_ruler = no
			employer = scope:actor
		}
		hidden:scope:target = {
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }
		}
	}
	
	
	allowed_trigger = {
		scope:target = {
			is_adult = yes
			corruption >= 15
			NOR = {
				has_trait = crafty
				has_trait = corrupt
				has_trait = pillager
				has_character_modifier = populist_demagogue_oh_no
			}
			NOT = { has_loyalty = corruption_sanctions_l }
			NAND = {
				employer = {
					is_monarchy = yes
				}
				is_co_ruler = yes
			}
		}
	}
	

	
	effect = {
		scope:actor = {
			add_tyranny = {
				value = scope:target.corruption
				divide = 10
				min = 1
			}
		}
		scope:target = {
			if = {
				limit = {
					scope:actor = {
						NOT = {
							invention = bribe_character_cost_modifier_inv
						}
					}
				}
				hidden_effect = { add_loyalty = corruption_sanctions_l }
				custom_tooltip = impose_corruption_sanctions_tt_1
			}
			else = { custom_tooltip = impose_corruption_sanctions_tt_2 }
			hidden_effect = {
				add_corruption = {
					value = 0
					subtract = corruption
					max = -5
					min = -50
				}
			}
			if = {
				limit = {
					is_minor_character = no
					family = {
						prestige > 0
					}
				}
				family = {
					add_prestige = {
						value = 0
						subtract = prestige
						divide = 100
						multiply = {
							value = scope:target.corruption
							divide = 4
						}
						max = -10
					}
				}
			}
		}
	}
	
}
