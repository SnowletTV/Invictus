switch_allowed_region = {

	on_other_nation = yes
	on_own_nation = no
	
	message = no
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_corruption"
	
	potential_trigger = {
		scope:actor = {
			tag = PTH
			has_variable = parthia_decentralized_wg
		}
		hidden:scope:target = {
			is_ruler = yes
			employer = { is_subject_of = scope:actor }
			employer = { is_subject_type = march }
			employer = { NOT = { has_variable = pth_excepted_var } }
			employer = {
				any_owned_province = {
					any_neighbor_province = {
						has_variable = allowed_province
					}				
				}
				NOT = {
					any_owned_province = {
						has_variable = allowed_province				
					}				
				}
			}
			is_alive = yes
			is_mercenary = no
		}
	}
	
	allowed_trigger = {
		hidden:scope:actor = {
			NOT = { has_variable = pth_region_cooldown }
		}
	} 
	
	effect = {
		scope:target.employer = {
			save_scope_as = march_scope
		}
		scope:actor = {
			trigger_event = me_parthia.9903
		}
	}
}

	on_other_nation = yes
	on_own_nation = no
	
	message = no
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_corruption"
	
	potential_trigger = {
		scope:actor = {
			tag = PTH
			has_variable = parthia_decentralized_wg

		}
		hidden:scope:target = {
			is_ruler = yes
			employer = { is_subject_of = scope:actor }
			employer = { is_subject_type = march }
			employer = { NOT = { has_variable = pth_excepted_var } }
			is_alive = yes
			is_mercenary = no

	}
	
	allowed_trigger = {
		hidden:scope:actor = {
			NOT = { has_variable = pth_region_cooldown }

		}
	} 
	
	effect = {
		scope:target.employer = {
			save_scope_as = march_scope
		}
		scope:actor = {
			trigger_event = me_parthia.9920
		}
	}
}
