ritual_behead = {

	on_other_nation = no
	on_own_nation = yes
	
	sound = "event:/SFX/UI/Character/Generic/sfx_ui_character_war"
	
	potential_trigger = {
		hidden:scope:actor = {
			is_ai = no
			tag = SLV
			has_variable = salluvian_beheading_allowed
			has_this_omen = deity:omen_severed_head
		}
		hidden:scope:target = {
			employer = scope:actor
			is_alive = yes
			prisoner_valid_for_country_events_trigger = yes
		}
	}
	
	allowed_trigger = {
		hidden:scope:actor = {
			can_pay_price = ritual_behead_price
			stability > 25
		}
		hidden:scope:target = {
			prisoner = yes
			is_adult = yes
			NAND = {
				employer = {
					is_monarchy = yes
				}
				is_co_ruler = yes
			}
		}
	}

	effect = {
		scope:actor = {
			pay_price = ritual_behead_price
			if = {
				limit = {
					scope:target = {
						has_variable = came_from
						NOT = {
							var:came_from = scope:actor
						}
					}
				}
				current_ruler = {
					add_popularity = {
						value = scope:target.prominence
						min = 1
						divide = 10
						min = 2
					}
				}
			}
			hidden_effect = {
				trigger_event = {
					on_action = custom_on_action_imprison
					days = { 15 90 }
				}
			}
		}
		scope:target = {
			death = { 
				death_reason = death_decapitated
				killer = scope:actor.current_ruler
			}
		}
	}
}
