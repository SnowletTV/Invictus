deity_bandus = { # WAR
	trigger = {
		polytheistic_check_religion_trigger = yes
		OR = {
			common_deity_culture_group_region_trigger = {
				RELIGION = animism
				CULTURE_GROUP = indo_iberian
				REGION = lusitania_region
			}
			common_deity_culture_trigger = {
				RELIGION = druidism
				CULTURE = callaecian
			}
			deity:omen_bandus = {
				holy_site_deity_check_trigger = yes
			}
		}
	}
	icon = deity_iberian_war
	passive_modifier = { enslavement_efficiency = 0.05 }
	omen = { discipline = omen_discipline_svalue } #Bonus from Omen
	on_activate = {
		fertility_apotheosis_capital_slaves_effect = yes
		if = {
			limit = {
				exists = scope:deified_ruler
				is_ai = no
			}
			custom_tooltip = deity_bandus_apotheosis_effect
			hidden_effect = {
				random = {
					chance = 100
					random_character = {
						limit = {
							prisoner = yes
						}
						death = { death_reason = death_claimed_by_the_underworld }
					}
				}
			}
		}
	}
	religion = animism
	deity_category = war #This Deity uses a War slot.

	deification_trigger = {
		can_deify_trigger = yes
	}
}
