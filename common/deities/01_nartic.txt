### General Deity Checks ###

deity_sawaserqua = { # ECONOMY
	trigger = { 
		polytheistic_check_religion_trigger = yes
		OR = {
			rare_deity_trigger = {
				RELIGION = circassian_pantheon
			}
			deity:omen_sawaserqua = {
				holy_site_deity_check_trigger = yes
			}
		}
	}
	icon = deity_tribal_wealth
	passive_modifier = { global_goods_from_slaves = -2 }
	omen = { civic_tech_investment = omen_civic_tech_investment_svalue }  #Bonus from Omen
	religion = circassian_pantheon
	deity_category = economy #This Deity uses an Economy Slot
	deification_trigger = {
		can_deify_trigger = yes
	}
	
	on_activate = {
		economy_apotheosis_province_improvement_effect = yes
	}
}

deity_batraz = { # WAR
	trigger = { 
		polytheistic_check_religion_trigger = yes
		OR = {
			rare_deity_trigger = {
				RELIGION = circassian_pantheon
			}
			deity:omen_batraz = {
				holy_site_deity_check_trigger = yes
			}
		}
	}
	icon = deity_tribal_wealth
	passive_modifier = { military_tech_investment = deity_military_tech_investment_svalue }
	omen = { land_morale_modifier = omen_land_morale_modifier_svalue } #Bonus from Omen
	religion = circassian_pantheon
	deity_category = war #This Deity uses a War Slot
	deification_trigger = {
		can_deify_trigger = yes
	}
	
	on_activate = {
		military_apotheosis_military_experience_effect = yes
		if = {
			limit = {
				exists = scope:deified_ruler
				is_ai = no
			}
			hidden_effect = {
				random = {
					chance = 5
					random_character = {
						limit = {
							is_ruler = no
							is_head_of_family = no
							is_general = no
							is_admiral = no
							is_primary_heir = no
							is_governor = no
							has_any_office = no
							from_ruler_family = no
							is_mercenary = no
							is_party_leader = no
							is_pretender = no
						}
						death = { death_reason = death_struck_by_batraz_lightening }
					}
				}
			}
		}
	}
}
