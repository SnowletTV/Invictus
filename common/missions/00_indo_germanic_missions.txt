	igk_mission_hybridization = { 
		icon = "task_political"

		requires = { igk_mission_integrate_local_nobility } 
		duration = 365

		allow = { 
			political_influence >= 50
			stability > 33
			war = no
		}

		on_start = {
			add_political_influence = -50
			add_stability = -25
			show_as_tooltip = {
				add_country_modifier = {
					name = igk_mission_hybridization_bad_modifier
					duration = 365 # 1 years
				}
			}
		}

		on_completion = {

			#Up here to ensure that there will be some Indo-Germanic pops
			#before switching culture, otherwise primary culture will not change
			capital_scope = {
				capital_formable_medium_effect = yes
			}

			custom_tooltip = igk_mission_hybridization_pops_tt
			#Change up to 1,000 old culture pops to new culture
			hidden_effect = {
				every_owned_province = {
					while = {
						count = 1000
						every_pops_in_province = {
							limit = {
								culture = root.culture
							}
							set_pop_culture = indo_germanic
						}
					}
				}
			}

			custom_tooltip = igk_mission_hybridization_characters_tt
			hidden_effect = {
				#Change all old culture characters to new culture			
				every_character = {
					limit = {
						AND = {
							OR = {
								from_ruler_family = yes
								is_minor_character = no
							}
							has_culture = root.culture
						}
					}
					set_culture = indo_germanic
				}
				set_primary_culture_cleanup_effect = {
					NEW_PRIMARY_CULTURE = indo_germanic
					MAKE_OLD_CULTURE_INTEGRATED = yes
				}
			}

			show_as_tooltip = {
				add_country_modifier = {
					name = igk_mission_hybridization_modifier
					duration = 6570 #18 years
				}
			}
			add_stability = 5


		} 
	}
