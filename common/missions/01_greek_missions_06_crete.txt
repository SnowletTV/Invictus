	icon = generic_greek_1
	header = mission_image_generic_greek

		factor = 10
	}

    	primary_culture = cretan
		is_subject = no

			ai_mission_back_out_trigger = yes

			}
		}
	}

    	start_mission_ai_effect = yes

            name = crete_mission_cooldown

		duration = 90

			political_influence >= 20
		}

		on_completion = {
			trigger_event = me_greek_6_crete.1
			add_tyranny = 10

				name = "grumbling_soldiers"
				duration = 1850
			}

	}

		icon = task_expansion
		duration = 90

		requires = {crete_mission_inv_3_task}

		allow = {
			capital_scope = {

				is_importing_trade_good = leather is_importing_trade_good = iron
