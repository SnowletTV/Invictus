	roman_12_mission_task_9 = { #Approach Illyrian Colonies
		icon = task_diplomatic
		duration = 180
		
		highlight = {
			scope:province = {
				OR = {
					is_in_area = histria_area
					is_in_area = dalmatia_occidentalis_area

				}

				}
			}
		}
		
		bypass = {
        	OR = {
                region:illyria_region = { 
                    NOT = {
                		any_region_province = {
                            has_owner = yes
                			owner = {
                                is_subject = no
                				not = { war_with = root }
                				num_of_cities >= 1
                				country_culture_group = hellenic
                			}
                		}
                    }
            	}
            }
        }
		
		allow = {
        	custom_tooltip = {
        		text = roman_12_mission_task_9_trigger_tt
        		region:illyria_region = {
	        		any_region_province = {
	        			owner = {
                            is_subject = no
	        				not = { war_with = root }
	        				opinion = {
	        					target = root
	        					value >= 15
	        				}
	        				num_of_cities >= 1
	        				country_culture_group = hellenic
	        			}
	        		}
	        	}
        	}
        }

        on_completion = {
        	if = {
        		limit = {
        			region:illyria_region = {
		        		any_region_province = {
		        			owner = {
                                is_subject = no
		        				not = { war_with = root }
		        				opinion = {
		        					target = root
		        					value >= 15
		        				}
		        				num_of_cities >= 1
		        				country_culture_group = hellenic
		        			}
		        		}
		        	}
        		}
        		ordered_country = {
        			order_by = rank
        			limit = {
        				is_subject = no
        				NOT = { war_with = root }
        				opinion = {
        					target = root
        					value >= 15
        				}
        				any_owned_province = {
        					is_in_region = illyria_region
        				}
        				num_of_cities >= 1
        				country_culture_group = hellenic
        			}

        			custom_tooltip = roman_12_mission_task_9_effect_effect_2_tt

        		}
        	}
        	else = {
        		custom_tooltip = roman_12_mission_task_9_effect_effect_2_tt
        	}
			scope:negotiate_hellenic_illyria_country = {
				trigger_event = { id = me_roman_missions_inv.12 }
			}
        }
