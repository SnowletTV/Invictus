            amount >= 2

            limit = {
                NOT = { owns = 426 }

            limit = {
                c:MAC = {
                    num_of_cities >= 1

                    owns = 379 
                    owns = 373

                    tag = MAC

            name = spartan_mission_04_rivals
            days = 7300
