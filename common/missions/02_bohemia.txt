	barbaricum_bohemia_mission_task_11 = { # Lugian League
        requires = { barbaricum_bohemia_mission_task_8  }
		duration = 180
        icon = "task_political"

        allow = {
			political_influence >= 100
			war = no
		}

		on_start = {
			if = {
                limit = {
                    any_country = {
                        primary_culture = vandal
                        OR = {
                        	is_subject = no
                        	is_subject_of = ROOT
                        }
                        is_ai = no
                    }
                }
                ordered_country = {
                    limit = {
                        primary_culture = vandal
                        OR = {
                        	is_subject = no
                        	is_subject_of = ROOT
                        }
                        is_ai = no
                    }
                    order_by = country_population
                    add_to_list = lugian_league_list
                }
            }
            every_country = {
                limit = {
                    primary_culture = vandal
                    OR = {
                    	is_subject = no
                    	is_subject_of = ROOT
                    }
                    is_ai = yes
                }
                add_to_list = lugian_league_list
            }
			show_as_tooltip = { add_political_influence = -100 }
			custom_tooltip = bohemia_lugian_league_reward_tt
			trigger_event = flavor_gaul.6
		}

		on_completion = { trigger_event = flavor_gaul.8 }
    }
