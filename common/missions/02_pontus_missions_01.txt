                current_ruler = {

                }
            }

                    limit = {

			scope:province = {

							is_in_area = paphlagonia_inferior_area

                        name = pontus_euxine_ports_mod

                }
            }
        }

        requires = { pontus_kingdom_missions_task_14 }

            political_influence >= 25
        }
        on_completion = {
            trigger_event = me_pontus_kingdom.15

                limit = {
                    this.culture = culture:median
