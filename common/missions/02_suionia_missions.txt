suionia_missions_01 = {
    header = "mission_image_general"
    icon = cde_gallic_chiefs_council

    repeatable = no
    chance = {
		factor = 100
	}

    potential = {
        NOT = { has_variable = mission_cooldown_var }
        is_subject = no
        has_civil_war = no
        tag = SWE
    }

    abort = {
        ai_mission_back_out_trigger = yes
    }
    on_start = {
        start_mission_ai_effect = yes
    }
    on_abort = {
        custom_tooltip = general_mission_cooldown_tt
        set_variable = {
            name = mission_cooldown_var
            days = 7300
        }
    }

        complete_mission_effect = yes
        add_2_free_province_investments = yes

        icon = "task_battle"

        allow = {
			custom_tooltip = {
				text = suionia_tt_1

					any_area_province = {
						count >= 2
						owner = root
						has_building = port_building
					}
				}
			}
            war = no
        }

        highlight = {
			scope:province = {
				is_in_area = east_gothia_area
			}
        }

        on_completion = {
			add_country_modifier = {
				name = suionia_mod_1
				duration = 7300 #20 years
			}
        }
    }
	
	suionia_missions_task_2 = { #build a navy
        icon = "task_battle"

        allow = {
			num_of_ships >= 10
            political_influence >= 10
            war = no
        }
		
		requires = {
			suionia_missions_task_1
		}

        on_completion = {
			add_political_influence = -10

				name = suionia_mod_2
				duration = 7300 #20 years
			}
        }
    }

				name = suionia_mod_3
