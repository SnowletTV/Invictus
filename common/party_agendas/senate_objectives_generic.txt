					limit = { has_variable = party_law_change_cooldown }
					has_variable = sen_obj_cha_law_oli_target
					NOT = { var:party_law_change_cooldown = var:sen_obj_cha_law_oli_target }
				}
			}
		}
	}

	on_finish = { #Party scope
		senate_objective_complete_effect = yes
		party_country = {
			set_variable = {
				name = party_law_change_cooldown
				days = 7300 # 20 years
				value = var:sen_obj_cha_law_oli_target
			}
			remove_variable = sen_obj_cha_law_oli_target
		}
	}

	abort_when = {
		OR = {
			party_country = { has_variable = oligarchic_cooldown }

				party_country = { has_variable = sen_obj_cha_law_oli_target }
			}
			NOT = { exists = scope:sen_obj_cha_law_oli_party }
			trigger_if = {
				limit = {
					party_country = {
						has_variable = party_law_change_cooldown
						has_variable = sen_obj_cha_law_oli_target
					}
				}
				party_country = { var:party_law_change_cooldown = var:sen_obj_cha_law_oli_target }
			}
		}
	}

	on_abort = {
		party_country = {
			if = {
				limit = {
					has_variable = sen_obj_cha_law_oli_target
					has_variable = party_law_change_cooldown

						value = -5
					}

			}
			remove_variable = sen_obj_cha_law_oli_target

	}

					limit = {
						NOT = { this.culture = PREV.culture }

					order_by = {
						value = 0
						subtract = country_culture_pop_count
					}
