army_size_merc_estimation = {
	value = 0
	if = {
		limit = {
			exists = scope:target
		}
		scope:target = {
			value = army_size_merc_estimation_calculation
		}
	}
}

army_size_merc_estimation_calculation = {
	value = 0
	every_character_unit = {
		limit = {
			unit_size > 0
		}
		add = subunits_recruitment_cost
	}
    multiply = 2
	min = 200
}

army_size_merc_estimation_sg = { 
	value = 0
	ROOT = {
		value = army_size_merc_estimation_calculation
	}
}
