weighted_power_base = {
    value = power_base
    if = {
        limit = {
            character_is_disloyal = yes
        }
        multiply = 2
    }
}

# Requires country to be saved as scope:country_at_war
num_of_territories_of_enemy_side = {
    value = 0
    if = {
        limit = {
            any_war_attacker = {
                this = scope:country_at_war
            }
        }
        every_war_defender = {
            add = num_of_territories_svalue
        }
    }
    else = {
        every_war_attacker = {
            add = num_of_territories_svalue
        }
    }
}
