territory_levy_population = {
    value = 0
    if = {
        limit = {
            exist = owner
        }
        every_pops_in_province = {
            limit = {
                NOT = {
                    pop_type = slaves
                }
                save_temporary_scope_as = pop
                owner = {
                    any_country_culture = {
                        is_culture = scope:pop.culture
                        is_integrated = yes
                    }
                }
            }
            add = 1
        }
    }
}

country_levy_population = {
    value = 0
    every_owned_province = {
        add = territory_levy_population
    }
}

capital_region_levy_population = {
    value = 0
    capital_scope = {
        region = {
            every_region_province = {
                limit = {
                    exists = owner
                    owner = prev.prev.owner
                }
                add = territory_levy_population
            }
        }
    }
}

non_capital_regions_levy_population = {
    value = 0
    every_owned_province = {
        limit = {
            NOT = {
                region = capital_scope.region
            }
        }
        add = territory_levy_population
    }
}
