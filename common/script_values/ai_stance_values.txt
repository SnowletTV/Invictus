# Requires root country to be saved as scope:war_leader_country
num_of_territories_of_enemies_where_is_war_leader = {
    value = 0
    if = {
        limit = {
            war = yes
        }
        every_current_war = {
            limit = {
                is_war_leader = scope:war_leader_country
            }
            if = {
                limit = {
                    any_war_attacker = {
                        this = scope:war_leader_country
                    }
                }
                every_war_defender = {
                    add = num_of_territories_svalue
                }
            }
            else = {
                every_war_attacker = {
                    add = num_of_territories_svalue
                }
            }
        }
    }
}

num_of_territories_of_subjects = {
    value = 0
    every_subject = {
        add = num_of_territories_svalue
    }
}
