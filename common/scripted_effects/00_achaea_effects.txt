release_AIG_effect = {
	if = {
		limit = {
			#NOT = { exists = AIG }
			owns = 444 #Aigion
		}
		custom_tooltip = custom_tooltip = dhe_achaean_league.7.AIG.TT
		hidden_effect = {
	        p:444 = { #Aigion
	            create_country = {
	                save_scope_as = aigion_country
					name = {
						name = AIG
						adjective = AIG_ADJ
					}
					change_country_tag = AIG
					set_country_religion = roman_pantheon
					change_government = oligarchic_republic
					change_country_color = aigion_color
					change_country_flag = AIG
					set_country_heritage = aigion_heritage
	            }
	        }
		}
	}
}
