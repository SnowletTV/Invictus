release_MTE_effect = {
	if = {
		limit = {
			#NOT = { exists = MTE }
			owns = 438 #Mantineia
		}
		custom_tooltip = custom_tooltip = dhe_arkadian_league.1.MTE.TT
		hidden_effect = {
	        p:438 = { #Mantineia
	            create_country = {
	                save_scope_as = mantineia_country
					name = {
						name = MTE
						adjective = MTE_ADJ
					}
					change_country_tag = MTE
					set_country_religion = roman_pantheon
					change_government = oligarchic_republic
					change_country_color = mantineia_color
					change_country_flag = MTE
					set_country_heritage = mantinea_heritage
	            }
	        }
		}
	}
}
