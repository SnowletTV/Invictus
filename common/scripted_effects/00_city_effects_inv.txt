add_neighbor_city_modifier_effect = { #Province scope
	if = {
		limit = {
			is_uninhabitable = no
		}
		if = {
			limit = { has_province_modifier = neighbor_city }
			remove_province_modifier = neighbor_city
		}
		while = {
			count = neighbor_cities_svalue

			add_permanent_province_modifier = {
				name = neighbor_city
				mode = add
			}
		}
	}
}
