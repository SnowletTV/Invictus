##Bloodline legitimacy scheme effect.
add_bloodline_effect = {
	add_trait = $BLOODLINE$
	if = {
		limit = {
			num_of_children > 0
		}
		every_child = {
			limit = { is_alive = yes }
			add_trait = $BLOODLINE$
			if = {
				limit = { num_of_children > 0 }
				every_child = {
					limit = { is_alive = yes }
					add_trait = $BLOODLINE$
				}
			}
		}
	}
}
