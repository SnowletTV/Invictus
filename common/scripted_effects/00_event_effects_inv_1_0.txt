
# Claim a region, autogenerates a nice tooltip
# Also see the "mission_claim_area_effect" for area claims
mission_claim_region_effect = {
	if = {
		limit = {
			region:$REGION$ = {
				any_region_province = {
					NOT = { owned_or_subject_owned = root }
				}
			}
		}
		random_province = {
			limit = { is_in_region = $REGION$ }
			custom_tooltip = mission_claim_region_tooltip
		}
		region:$REGION$ = {
			hidden_effect = {
				every_region_province = {
					limit = {
						NOT = { owned_or_subject_owned = root }
					}
					add_claim = root
				}
			}
		}
	}
}

add_bloodline_effect = {
	add_trait = $BLOODLINE$
	if = {
		limit = {
			num_of_children > 0
		}
		every_child = {
			limit = { is_alive = yes }
			add_trait = $BLOODLINE$
			if = {
				limit = { num_of_children > 0 }
				every_child = {
					limit = { is_alive = yes }
					add_trait = $BLOODLINE$
				}
			}
		}
	}
}

# Hunting
hunt_calc_effect = { # hunted_animal_var has to be set before; randomized calucalation of hunt events
	set_variable = { # variable has to be remove manually after the effect is run and used (outcome)
		name = hunting_variable
		value = current_ruler.martial
	}
	change_variable = {
		name = hunting_variable
		add = var:hunted_animal_var
	}
	random_list = {
		1 = { # really bad hunt
			change_variable = {
				name = hunting_variable
				add = -5
			}
		}
		1 = { # bad hunt
			change_variable = {
				name = hunting_variable
				add = -3
			}
		}
		1 = { #unlucky hunt
			change_variable = {
				name = hunting_variable
				add = -1
			}
		}
		1 = { #lucky hunt
			change_variable = {
				name = hunting_variable
				add = 1
			}
		}
		1 = { #good hunt
			change_variable = {
				name = hunting_variable
				add = 3
			}
		}
		1 = { #really successful hunt
			change_variable = {
				name = hunting_variable
				add = 5
			}
		}
	}
}
