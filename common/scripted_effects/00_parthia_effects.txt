add_peaceful_dahae_weight_effect = {
	custom_tooltip = "dahae_peaceful_weight_up_tt"
	if = {
		limit = {
			NOT = {
				has_variable = dahae_stance_weight 
			}
		}
		set_variable = {
			name = dahae_stance_weight
			value = 1
		}
	}
	else = {
		change_variable = {
			name = dahae_stance_weight
			add = 1
		}
	}
}
