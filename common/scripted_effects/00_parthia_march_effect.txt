add_allowed_region_effect = {
	custom_tooltip = "dahae_peaceful_weight_up_tt"
	if = {
		limit = {
			NOT = {
				has_variable = parthia_allowed_region_var
			}
		}
		set_variable = {
			name = parthia_allowed_region_var
			value = 1
		}
	}
	else = {
		change_variable = {
			name = parthia_allowed_region_var
			add = 1
		}
	}
}

subtract_allowed_region_effect = {
	custom_tooltip = "dahae_peaceful_weight_up_tt"
	if = {
		limit = {
			NOT = {
				has_variable = parthia_allowed_region_var 
			}
		}
		set_variable = {
			name = parthia_allowed_region_var
			value = -1
		}
	}
	else = {
		change_variable = {
			name = parthia_allowed_region_var
			add = -1
		}
	}
}
