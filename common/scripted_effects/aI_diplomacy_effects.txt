ai_admire_overlord = {
    if = {
        limit = {
            is_ai = yes
            exists = overlord
            rank <= local_power
            overlord = {
                is_ai = yes
                rank >= major_power
            }
            save_temporary_scope_as = subject_country
            any_owned_province = {
                any_neighbor_province = {
                    owner = scope:subject_country.overlord
                }
            }
        }
        add_opinion = {
            modifier = ai_overlord_admiration
            target = overlord
        }
    }
}
