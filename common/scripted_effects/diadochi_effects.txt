republic_stasis_effect = {
	if = {
		limit = {
			is_ai = yes
				OR = {
					government = democratic_republic
					government = oligarchic_republic
				}
		}
		if = {
			limit = {
				government = democratic_republic
			}
			change_government = oligarchic_republic
			hidden_effect = { add_political_influence = 75 }
		}
		else_if = {
			limit = {
				government = oligarchic_republic
			}
			change_government = democratic_republic
			hidden_effect = { add_political_influence = 75 }
		}
	}
}
