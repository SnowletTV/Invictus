#Ensuring that Roman women have proper cognomina
on_giving_birth_effect = {
	if = {
		limit = { #Italic men inherit their father's cognomen
			scope:newborn = {
				exists = father
				father = { has_nickname = yes }
				is_male = yes
				OR = {
					has_culture = roman
					has_culture = samnite
					has_culture = umbrian
					has_culture = siculian
					has_culture = bruttian
					has_culture = lucanian
					has_culture = catari
					has_culture = etruscan
				}
			}
		}
		scope:newborn = {
			add_nickname = father
		}
	}
	else_if = {
		limit = { #Etruscan women inherit their mother's cognomen
			scope:newborn = {
				exists = mother
				mother = { has_nickname = yes }
				is_female = yes
				OR = {
					has_culture = etruscan
				}
			}
		}
		scope:newborn = {
			add_nickname = mother
		}
	}
	else_if = {
		limit = { #Other Italic cultures have cognomina based on birth order
			scope:newborn = {
				exists = father
				is_female = yes
				OR = {
					has_culture = roman
					has_culture = samnite
					has_culture = umbrian
					has_culture = siculian
					has_culture = bruttian
					has_culture = lucanian
					has_culture = catari
					has_culture = etruscan
				}
			}
		}
		scope:newborn = {
			if = {
				limit = {
					has_nickname = no
					any_sibling = {
						OR = {
							has_same_family = scope:newborn
							NOR = {
								has_variable = first_daughter
								has_variable = second_daughter
								has_variable = third_daughter
								has_variable = fourth_daughter
								has_variable = fifth_daughter
								has_variable = sixth_daughter
								has_variable = seventh_daughter
								has_variable = eighth_daughter
								has_variable = ninth_daughter
								has_variable = tenth_daughter
								has_variable = eleventh_daughter
								has_variable = twelfth_daughter
							}
						}
					}
				}
				add_nickname = NICKNAME_PRIMA
				set_variable = {
					name = first_daughter
					value = 1
				}
			}
			else_if = {
				limit = {
					has_nickname = no
					any_sibling = {
						has_same_family = scope:newborn
						has_variable = first_daughter
						NOR = {
							has_variable = second_daughter
							has_variable = third_daughter
							has_variable = fourth_daughter
							has_variable = fifth_daughter
							has_variable = sixth_daughter
							has_variable = seventh_daughter
							has_variable = eighth_daughter
							has_variable = ninth_daughter
							has_variable = tenth_daughter
							has_variable = eleventh_daughter
							has_variable = twelfth_daughter
						}
					}
				}
				add_nickname = NICKNAME_SECUNDA
				set_variable = {
					name = second_daughter
					value = 1
				}
			}
			else_if = {
				limit = {
					has_nickname = no
					any_sibling = {
						has_same_family = scope:newborn
						OR = {
							has_variable = first_daughter
							has_variable = second_daughter
						}
						NOR = {
							has_variable = third_daughter
							has_variable = fourth_daughter
							has_variable = fifth_daughter
							has_variable = sixth_daughter
							has_variable = seventh_daughter
							has_variable = eighth_daughter
							has_variable = ninth_daughter
							has_variable = tenth_daughter
							has_variable = eleventh_daughter
							has_variable = twelfth_daughter
						}
					}
				}
				add_nickname = NICKNAME_TERTIA
				set_variable = {
					name = third_daughter
					value = 1
				}
			}
			else_if = {
				limit = {
					has_nickname = no
					any_sibling = {
						has_same_family = scope:newborn
						OR = {
							has_variable = first_daughter
							has_variable = second_daughter
							has_variable = third_daughter
						}
						NOR = {
							has_variable = fourth_daughter
							has_variable = fifth_daughter
							has_variable = sixth_daughter
							has_variable = seventh_daughter
							has_variable = eighth_daughter
							has_variable = ninth_daughter
							has_variable = tenth_daughter
							has_variable = eleventh_daughter
							has_variable = twelfth_daughter
						}
					}
				}
				add_nickname = NICKNAME_QUARTA
				set_variable = {
					name = fourth_daughter
					value = 1
				}
			}
			else_if = {
				limit = {

						has_same_family = scope:newborn
						OR = {
							has_variable = first_daughter
							has_variable = second_daughter
							has_variable = third_daughter
							has_variable = fourth_daughter
						}

							has_variable = ninth_daughter
							has_variable = tenth_daughter
							has_variable = eleventh_daughter

						}
					}
				}
				add_nickname = NICKNAME_QUINTA
				set_variable = {
					name = fifth_daughter
					value = 1
				}
			}
			else_if = {
				limit = {
					has_nickname = no
					any_sibling = {
						has_same_family = scope:newborn
						OR = {
							has_variable = first_daughter

on_becoming_adult_effect = {
	if = {
		limit = {
			has_nickname = no
			is_male = yes
			OR = {
				has_culture = roman
				has_culture = samnite
				has_culture = umbrian
				has_culture = siculian
				has_culture = bruttian
				has_culture = lucanian
				has_culture = catari
				has_culture = etruscan
			}
		}
		random_list = {
			75 = { }
			25 = { add_nickname = random }
		}
	}
	else_if = { #Etruscan women have normal cognomina
		limit = {
			has_nickname = no
			is_female = yes
			OR = {
				has_culture = etruscan
			}
		}
		random_list = {
			70 = { }
			2 = { add_nickname = NICKNAME_PALPENIA }
			2 = { add_nickname = NICKNAME_PAPANIA }
			2 = { add_nickname = NICKNAME_PREZUNIA }
			2 = { add_nickname = NICKNAME_PURCENIA }
			2 = { add_nickname = NICKNAME_SENATINA }
			2 = { add_nickname = NICKNAME_SEPUNIA }
			2 = { add_nickname = NICKNAME_SINUNIA }
			2 = { add_nickname = NICKNAME_TREPUNIA }
			2 = { add_nickname = NICKNAME_TREPIA }
