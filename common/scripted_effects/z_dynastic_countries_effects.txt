#-----------------------------------------------------------------------------------------
#
#	After the right family has been identified, actually change the country name.
#
#-----------------------------------------------------------------------------------------

z_dynastic_countries_rename_this_country = {
	set_variable = {
		name = z_dynastic_countries_flag
		value = current_ruler.family
	}
	if = {
		limit = {
			has_variable = z_dynastic_countries_flag_e
		}
		remove_variable = z_dynastic_countries_flag_e
	}
	if = {
		limit = {
			has_variable = z_dynastic_countries_flag_k
		}
		remove_variable = z_dynastic_countries_flag_k
	}
	if = {
		limit = {
			z_dynastic_countries_is_empire = yes
		}
		change_country_name = $DYNASTY_NAME$_DCN_E
		hidden_effect = { change_country_adjective = $DYNASTY_NAME$_DCN_A }
		set_variable = {
			name = z_dynastic_countries_flag_e
		}
	}
	else = {
		change_country_name = $DYNASTY_NAME$_DCN_K
		hidden_effect = { change_country_adjective = $DYNASTY_NAME$_DCN_A }
		set_variable = {
			name = z_dynastic_countries_flag_k
		}
	}
}
