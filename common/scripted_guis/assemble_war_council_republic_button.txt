assemble_war_council_republic_button = {
    scope = country
	saved_scopes = {
		
	}
    is_shown = { # Can be omitted as always true
       is_republic = yes
    }
	ai_is_valid = {
		always = no #we want the AI to use common sense.
		war = no
	}
    is_valid = {
		trigger_if = {
			limit = {
				tag = ROM
				NOT = {
					has_law = lex_calpurnia
				}
			}
			has_law = lex_calpurnia
		} 
		trigger_if = {
			limit = {
				tag = CAR
				NOT = {
					has_law = republic_corruption_car_4
				}
			}
			has_law = republic_corruption_car_4
		} 
		trigger_else_if = {
			limit = {
				NOT = {
					tag = ROM
					tag = CAR
				}
				is_republic = yes
				NOT = {
					has_law = republic_corruption_4
				}
			}
			has_law = republic_corruption_4
		}
		trigger_if = {
			limit = {
				NOT = {
					any_character = {
						count >= 3
						is_ruler = no
						is_adult = yes
					}
				}
			}
			any_character = {
				count >= 3
				is_ruler = no
				is_adult = yes
			}
		}
		trigger_if = {
			limit = {
				has_variable = summoned_war_council
			}
			custom_tooltip = {
				text = "not_summoned_war_council_recently_tt"
				NOT = {
					has_variable = summoned_war_council
				}
			}
		}
		trigger_if = {
			limit = {
				NOT = {
					any_neighbour_country = {
						NOR = {
							alliance_with = root
							is_subject_of = root
						}
					}
				}
			}
			any_neighbour_country = {
				NOR = {
					alliance_with = root
					is_subject_of = root
				}
			}
		}
    }
    effect = {
       set_variable = {
		name = summoned_war_council
		days = 3650
	   }
	   trigger_event = {
		id = war_council.2
	   }
	   custom_tooltip = "assemble_war_council_republic_button_desc_tt"
    }
}
