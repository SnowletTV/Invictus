bribe_mercenary_button = {
    scope = character
	saved_scopes = {
		player
	}
    is_shown = {
		is_mercenary = yes
		OR = {
			is_general = yes
			is_admiral = yes
		}
		NOT = {
			employer = {
				OR = {
					tag = MER
					this = scope:player
				}
			}
		}
    }
	ai_is_valid = {
		always = no
	}
    is_valid = {
		is_mercenary = yes
		in_command = yes
		custom_tooltip = {
			text = eligible_merc_buyoff_tt
			any_character_unit = {
				unit_size > 0
				in_combat = no
			}
		}
		trigger_if = {
			limit = {
				exists = root.employer
			}
			root.employer = {
				NOT = {
					invention = recruit_mercenary_inv
				}
			}
		}
		scope:player = {
			war_with = root.employer
			treasury >= army_size_merc_estimation_sg 
		}
		custom_tooltip = {
			text = merc_1_year_buyoff_tt
			NOT = {
				root = {
					has_variable = merc_buyoff_cooldown
				}
			}
		}
    }
    effect = {
		custom_tooltip = merc_buyoff_effect_tooltip_sg 
		scope:player = {
			add_treasury = {
				value = 0
				subtract = army_size_merc_estimation_sg
			}
		}
		set_variable = {
			name = merc_buyoff_cooldown
			days = 365
		}
		hidden_effect = {
			save_scope_as = target
			scope:player = {
				save_scope_as = actor
			}
			employer = {
				set_local_variable = {
					name = merc_payoff_value
					value = army_size_merc_estimation_sg
				}
				trigger_event = {
					id = warfare.3
				}
			}
		}
    }
}
