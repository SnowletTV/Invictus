build_in_subject = {
	scope = country

	saved_scopes = {
		target
	}

	ai_is_valid = {
		always = no
	}
	
	is_shown = {
		scope:target = {
			is_subject_of = ROOT
			is_subject_type = march	
		}
		ROOT = {
			tag = PTH
			has_variable = parthia_decentralized_wg
			NOT = { THIS = scope:target }
		}	
	}

	is_valid = {
		hidden:scope:target = {
			is_subject_of = ROOT
			is_subject_type = march	
		}
		hidden:ROOT = {
			tag = PTH
			has_variable = parthia_decentralized_wg
		}
	}

	effect = {
		scope:target = {
			save_scope_as = march_scope
		}
		ROOT = {
			trigger_event = me_parthia.9910
		}
	}
}

switch_allowed_region = {
	scope = country

	saved_scopes = {
		target
	}

	ai_is_valid = {
		always = no
	}
	
	is_shown = {
		hidden:scope:target = {
			is_subject_of = ROOT
			is_subject_type = march
			NOT = { has_variable = pth_excepted_var }
			any_owned_province = {
				any_neighbor_province = {
					has_variable = allowed_province
				}				
			}		
		}
		hidden:ROOT = {
			tag = PTH
			has_variable = parthia_decentralized_wg
		}
	}

	is_valid = {
		hidden:scope:target = {
			is_subject_of = ROOT
			is_subject_type = march
			NOT = { has_variable = pth_excepted_var }
			any_owned_province = {
				any_neighbor_province = {
					has_variable = allowed_province
				}				
			}		
		}
		hidden:ROOT = {
			tag = PTH
			has_variable = parthia_decentralized_wg
			NOT = { has_variable = pth_region_cooldown }
		}
	}

	effect = {
		scope:target = {
			save_scope_as = march_scope
		}
		ROOT = {
			trigger_event = me_parthia.9903
		}
	}
}
