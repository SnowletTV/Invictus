summon_curiate_assembly = {
    scope = country
	saved_scopes = {
		
	}
    is_shown = { # Can be omitted as always true
       is_republic = yes
    } 
	ai_is_valid = {
		war = no
	}
    is_valid = {
		trigger_if = {
			limit = {
				tag = ROM
				NOT = {
					has_law = no_laws_on_corruption
				}
			}
			has_law = no_laws_on_corruption
		}
		trigger_if = {
			limit = {
				tag = CAR
				NOT = {
					has_law = republic_corruption_car_1
				}
			}
			has_law = republic_corruption_car_1
		}  
		trigger_else_if = {
			limit = {
				NOT = {
					tag = ROM
					tag = CAR
				}
				is_republic = yes
				NOT = {
					has_law = republic_corruption_1
				}
			}
			has_law = republic_corruption_1
		}
		trigger_if = {
			limit = {
				NOT = {
					any_character = {
						count >= 3
						is_ruler = no
						is_adult = yes
					}
				}
			}
			any_character = {
				count >= 3
				is_ruler = no
				is_adult = yes
			}
		}
		trigger_if = {
			limit = {
				NOT = {
					any_character = {
						is_governor = yes
						is_ruler = no
					}
				}
			}
			custom_tooltip = {
				text = "any_non_ruler_governor_character"
				any_character = {
					is_governor = yes
					is_ruler = no
				}
			}
		}
		trigger_if = {
			limit = {
				has_variable = summoned_curiate_assembly
			}
			custom_tooltip = {
				text = "not_summoned_curiate_assembly_recently_tt"
				NOT = {
					has_variable = summoned_curiate_assembly
				}
			}
		}
    }
    effect = {
       set_variable = {
		name = summoned_curiate_assembly
		days = 3650
	   }
	   trigger_event = {
			id = assembly.1
	   }
	   custom_tooltip = "assemble_curiate_assembly_desc_tt"
    }
}
