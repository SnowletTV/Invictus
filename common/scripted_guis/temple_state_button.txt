temple_state_button = {
    scope = province
	saved_scopes = {
		player
	}
	confirm_title = "temple_state_button_title"
	confirm_text = "temple_state_button_desc"
    is_shown = { # Can be omitted as always true
		scope:player = {
			OR = {
				country_culture_group = anatolian
				has_variable = temple_state_variable
			}
		}
		has_owner = yes
		controller = scope:player
		owner = scope:player
		OR = {
			has_holy_site = yes
			own_religious_center_province_trigger = yes
			has_province_modifier = embelished_great_shrine_modifier 
			has_province_modifier = temple_state_modifier 
		}
    }
	ai_is_valid = {
		always = no
	}
    is_valid = {
		trigger_if = {
			limit = {
				num_of_slaves <= 5
			}
			num_of_slaves >= 5
		}
		trigger_if = {
			limit = {
				scope:player = {
					current_ruler = {
						zeal <= 5
					}
				}
			}	
			scope:player = {
				current_ruler = {
					zeal >= 5
				}
			}
		}
		trigger_if = {
			limit = {
				is_capital = yes
			}
			is_capital = no
		}
		trigger_if = {
			limit = {
				scope:player = {
					war = yes
				}
			}
			scope:player = {
				war = no 
			}
		}
		hidden:scope:player = {
			trigger_if = {
				limit = {
					NOT = {
						can_pay_price = create_merc_state_price
					}
				}
				can_pay_price = create_merc_state_price
			}
		}
    }
	
    effect = {
		custom_tooltip = "MAKE_TEMPLE_STATE_EFFECT_TT"
		scope:player = {
			pay_price = create_merc_state_price
		} 
		hidden_effect = {
			owner = {
				save_scope_as = pParent
			}
			create_country = {
				create_character = {
					culture = root.dominant_province_culture
					religion = root.dominant_province_religion
					age = 30
					save_scope_as = temple_state_high_priest
				}
				set_as_ruler = scope:temple_state_high_priest
				save_scope_as = pChild
				scope:temple_state_high_priest = {
					add_zeal = 3
				}
			}

			scope:pParent = {
				if = {
					limit = {
						is_republic = yes
					}
					scope:pChild = {
						change_government = theocratic_republic
					}
				}
				else = {
					scope:pChild = {
						change_government = theocratic_monarchy
					}
				}
			}
			scope:pParent = {
				make_subject = {
					target = scope:pChild
					type = temple_state
				}
			}
			scope:pChild = {
				set_country_heritage = religious_enclave_heritage
				capital_scope = {
					if = {
						limit = {
							NOR = { 
								has_province_modifier = embelished_great_shrine_modifier 
								has_province_modifier = temple_state_modifier 

							}
						}
						add_province_modifier = {
							name = temple_state_modifier
							duration = -1
						}
					}
				}
			}
		}
	}
}
