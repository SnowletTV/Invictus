trade_good_decision_potential = {
	#INPUT: Late = X (1-3)
	is_ai = no
	trigger_if = {
		limit = {
			$TIME$ = 1
		}
		current_date > 450.11.2
	}
	trigger_else_if = {
		limit = {
			$TIME$ = 2
		}
		current_date > 510.1.25
	}
	trigger_else_if = {
		limit = {
			$TIME$ = 3
		}
		current_date > 600.1.25
	}
	trigger_else = {
		current_date > 450.11.2
	}
}

trade_good_decision_allow = {
	#INPUT: DIFFICULTY = X (1-4)
	trigger_if = {
		limit = {
			$DIFFICULTY$ = 1
		}
		has_city_status = yes
		num_goods_produced >= 2
		total_population >= 20
	}
	trigger_else_if = {
		limit = {
			$DIFFICULTY$ = 2
		}
		has_city_status = yes
		total_population >= 30
		num_goods_produced >= 2
	}
	trigger_else_if = {
		limit = {
			$DIFFICULTY$ = 3
		}
		has_city_status = yes
		total_population >= 40
		num_goods_produced >= 3
		free_building_slots = 0
		has_no_ongoing_construction = yes
	}
	trigger_else_if = {
		limit = {
			$DIFFICULTY$ = 4
		}
		has_city_status = yes
		total_population >= 50
		num_goods_produced >= 4
		free_building_slots = 0
		has_no_ongoing_construction = yes
	}
	trigger_else = {
		has_city_status = yes
		num_goods_produced >= 2
		total_population >= 20
	}
}

# This trigger should be used in all mission tasks that require a province to build a building.
has_no_ongoing_construction = {
	trigger_if = {
	    limit = { has_construction = yes }
	    custom_tooltip = {
	        text = province_has_no_construction_tooltip
	        has_construction  = no
	    }
	}
}
