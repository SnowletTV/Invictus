	exists = fam:$DYNASTY_NAME$
	exists = current_ruler.family
	current_ruler.family = fam:$DYNASTY_NAME$

	OR = {
		government = imperium
		government = imperial_cult
		#num_of_cities >= 500
	}

		# is_tier_2_formable_trigger = yes

	}
