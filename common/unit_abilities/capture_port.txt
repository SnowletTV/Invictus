capture_port = {

	enable = yes
	toggle = no
	soundeffect = "event:/SFX/UI/Unit/sfx_ui_unit_capture_port"

	navy_only = yes


	allow = {
		is_navy = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		custom_tooltip = {
			text = capture_port_explain_tt
			any_sub_unit = {
				count >= 5
				ship_category = heavy
				subunit_strength_percentage >= 0.50
			}
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						province_id = 84 #Syracuse
					}
				}
			}
			custom_tooltip = {
				text = "capture_port_no_valid_port_syracuse"
				NOT = {
					unit_location = {
						any_neighbor_province = {
							province_id = 84 #Syracuse
						}
					}
				}		
			}			
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						num_of_port_building >= 1
						NOT = {
							province_id = 84 #Syracuse
						}
						owner = {
							OR = {
								war_with = root.unit_owner
								this = root.unit_owner
							}
						}
						controller = { war_with = root.unit_owner }
						fort_level < 1
						NOT = { controller = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			unit_location = {
				any_neighbor_province = {
					is_sea = no
					has_owner = yes
					num_of_port_building >= 1
					NOT = {
						province_id = 84 #Syracuse
					}
					owner = {
						OR = {
							war_with = root.unit_owner
							this = root.unit_owner
						}
					}
					controller = { war_with = root.unit_owner }
					fort_level < 1
					NOT = { controller = root.unit_owner }
					NOT = { has_enemy_army = root.unit_owner }
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = "capture_port_no_valid_port"
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						num_of_port_building >= 1
						NOT = {
							province_id = 84 #Syracuse
						}
						owner = {
							OR = {
								war_with = root.unit_owner
								this = root.unit_owner
							}
						}
						controller = { war_with = root.unit_owner }
						fort_level < 1
						NOT = { controller = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
		}
	}
	
	start_effect = {
		random_sub_unit = {
			limit = {
				ship_category = heavy
				subunit_strength_percentage >= 0.50
			}
			if = {
				limit = {
					has_variable = improved_naval_siege
				}
				add_subunit_strength = -0.40
			}
			else = {
				add_subunit_strength = -0.50
			}
		}
		random_sub_unit = {
			limit = {
				ship_category = heavy
				subunit_strength_percentage >= 0.50
			}
			if = {
				limit = {
					has_variable = improved_naval_siege
				}
				add_subunit_strength = -0.40
			}
			else = {
				add_subunit_strength = -0.50
			}
		}
		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						province_id = 84 #Syracuse
					}
				}
			}
			custom_tooltip = capture_port_no_valid_port_syracuse		
		}
		else_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						num_of_port_building >= 1
						NOT = {
							province_id = 84 #Syracuse
						}
						owner = {
							OR = {
								war_with = root.unit_owner
								this = root.unit_owner
							}
						}
						controller = { war_with = root.unit_owner }
						fort_level < 1
						NOT = { controller = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			unit_location = {
				random_neighbor_province = {
					limit = { 
						is_sea = no
						has_owner = yes
						num_of_port_building >= 1
						NOT = {
							province_id = 84 #Syracuse
						}
						owner = {
							OR = {
								war_with = root.unit_owner
								this = root.unit_owner
							}
						}
						controller = { war_with = root.unit_owner }
						fort_level < 1
						NOT = { controller = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
					set_controller = root.unit_owner
				}
			}
		}
		else = {
			custom_tooltip = capture_port_explain_effect_tt
		}
	}
}
