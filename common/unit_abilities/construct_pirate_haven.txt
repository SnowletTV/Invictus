construct_pirate_haven = {

	duration = 0
	enable = yes
	toggle = no
	soundeffect = "event:/SFX/UI/Unit/sfx_ui_unit_remove_pirate"

	army_only = yes
	
	allow = {
		NOR = {
			has_variable = build_pirate_haven_cooldown
		}
		unit_owner = {
			trigger_if = {
				limit = {
					tag = ROM
					is_republic = yes
				}
				has_law = lex_caecilia
			}
			trigger_else_if = {
				limit = {
					tag = CAR
					is_republic = yes
				}
				has_law = republican_mediterranean_car_3 
			}
			trigger_else_if = {
				limit = { is_republic = yes }
            	has_law = republican_mediterranean_3
			}
			trigger_else = { 
				OR = {
					political_influence >= 50
				}
			}
		}
		is_army = yes
		in_combat = no
		is_moving = no
		in_siege = no
		is_exiled = no
		unit_location = {
			has_owner = yes
			is_coastal = yes
			OR = {
				controller = root.unit_owner
				controller = {
					is_subject_of = root.unit_owner
				}
			}
			NOT = {
				has_province_modifier = pirate_haven_modifier
			}
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	
	
	start_effect = {
		unit_location = {
			add_province_modifier = {
				name = pirate_haven_modifier
				duration = -1
			}
		}
		unit_owner = {
			if = {
			add_treasury = { 10 50 }
			}
		}
		root = {
			set_variable = {
  			name = build_pirate_haven_cooldown
    		days = 3650
   		}	
		}
	}
	
	ai_will_do = {
		modifier = {
			trigger = {
				unit_location = {
					NOT = {
						has_province_modifier = pirate_haven_modifier
					}
					has_owner = yes
					controller = root.unit_owner
				}
				unit_owner = {
					OR = {
						has_law = republican_mediterranean_4
						has_law = antipiracy_edict
						has_law = lex_claudia
						has_law = republican_mediterranean_car_4
						has_country_modifier = bosporan_antipiracy_mod
					}
				}
			}
			add = {
				value = 100
			}
		}
	}
}
