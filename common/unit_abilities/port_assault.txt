port_assault = {

	enable = yes
	toggle = no
	soundeffect = "event:/SFX/UI/Unit/sfx_ui_unit_port_assualt"


	navy_only = yes


	allow = {
		is_navy = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		custom_tooltip = {
			text = port_assault_explain_tt
			any_sub_unit = {
				sub_unit_type = mega_galley
				subunit_strength_percentage >= 0.95
			}
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						province_id = 84 #Syracuse
					}
				}
			}
			custom_tooltip = {
				text = "capture_port_no_valid_port_syracuse"
				NOT = {
					unit_location = {
						any_neighbor_province = {
							province_id = 84 #Syracuse
						}
					}
				}		
			}			
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						owner = {
							exists = yes
							invention = archimedes_death_ray_inv
						}
					}
				}
			}
			custom_tooltip = {
				text = "capture_port_no_valid_port_archimedes"
				NOT = {
					unit_location = {
						any_neighbor_province = {
							is_sea = no
							has_owner = yes
							is_port = yes
							owner = {
								exists = yes
								invention = archimedes_death_ray_inv
							}
						}
					}
				}		
			}			
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						NOT = {
							province_id = 84 #Syracuse
						}
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			unit_location = {
				any_neighbor_province = {
					is_sea = no
					has_owner = yes
					is_port = yes
					NOT = {
						province_id = 84 #Syracuse
					}
					fort_level >= 1
					NOT = { controller = root.unit_owner }
					controller = { war_with = root.unit_owner }
					NOT = { has_enemy_army = root.unit_owner }
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = port_assault_no_valid_port
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						NOT = {
							province_id = 84 #Syracuse
						}
						is_port = yes
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
		}
	}
	
	
	
	start_effect = {
		random_sub_unit = {
			limit = {
				sub_unit_type = mega_galley
				subunit_strength_percentage >= 0.95
			}
			if = {
				limit = {
					has_variable = improved_naval_siege
				}
				add_subunit_strength = -0.50
			}
			else = {
				add_subunit_strength = -0.70
			}
		}
		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						province_id = 84 #Syracuse
					}
				}
			}
			custom_tooltip = capture_port_no_valid_port_syracuse			
		}
		else_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						has_owner = yes
						owner = {
							invention = archimedes_death_ray_inv
						}
					}
				}
			}
			custom_tooltip = capture_port_no_valid_port_archimedes		
		}
		else_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						NOT = {
							province_id = 84 #Syracuse
						}
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			unit_location = {
				random_neighbor_province = {
					limit = {
						is_sea = no
						has_owner = yes
						is_port = yes
						NOT = {
							province_id = 84 #Syracuse
						}
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
					remove_building_level = fortress_building
				}
			}
		}
		else = {
			custom_tooltip = port_assault_explain_effect_tt
		}
	}
	
}
