root_out_pirates = {

	duration = 0
	enable = yes
	toggle = no
	soundeffect = "event:/SFX/UI/Unit/sfx_ui_unit_remove_pirate"

	army_only = yes

	hidden = {
		unit_owner = {
			is_tribal = yes
		}
	}
	
	allow = {
		unit_owner = {
			trigger_if = {
				limit = {
					tag = ROM
					is_republic = yes
				}
				has_law = lex_claudia
			}
			trigger_else_if = {
				limit = {
					tag = CAR
					is_republic = yes
				}
				has_law = republican_mediterranean_car_4
			}
			trigger_else_if = {
				limit = { is_republic = yes }
            	has_law = republican_mediterranean_4
			}
			trigger_else = { 
				OR = {
					has_law = antipiracy_edict
					has_country_modifier = bosporan_antipiracy_mod 
				}
			}
		}
		is_army = yes
		in_combat = no
		is_moving = no
		in_siege = no
		is_exiled = no
		unit_location = {
			has_owner = yes
			OR = {
				controller = root.unit_owner
				controller = {
					is_subject_of = root.unit_owner
				}
			}
			has_province_modifier = pirate_haven_modifier
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	
	
	start_effect = {
		unit_owner = {
			if = {
				limit = {
					has_variable = bosporan_pirate_haven_bonus_var
				}
				add_treasury = { 65 100 }
				capital_scope = {
					create_state_pop = slaves
				}
			}
		}
		unit_location = {
			remove_province_modifier = pirate_haven_modifier
		}
	
	}
	
	ai_will_do = {
		modifier = {
			trigger = {
				unit_location = {
					has_province_modifier = pirate_haven_modifier
					has_owner = yes
					controller = root.unit_owner
				}
				unit_owner = {
					OR = {
						has_law = republican_mediterranean_4
						has_law = antipiracy_edict
						has_law = lex_claudia
						has_law = republican_mediterranean_car_4
					}
				}
			}
			add = {
				value = 100
			}
		}
	}
	
	idle_entity_state = raiding
}
