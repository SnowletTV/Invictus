#
# WAR GOAL TYPES:
#	- take_province

diadochi_wargoal = {
	type = superiority
	uses_civil_war_conquest = yes
	allow = { #Will this show up in the declare war window.
		is_subject = no
		OR = { 
			tag = EGY
			tag = MAC
			tag = TRE
			tag = SEL
			tag = PRY
			has_variable = pleistarchus_satrapy_flag
		}
		NOT = { has_variable = epigoni } #No longer usable if you had the Epigoni event
		has_variable = war_of_the_diadochs #Turns on after the intro event which explains what it does.
		scope:defender = {
			OR = {
				tag = EGY
				tag = MAC
				tag = TRE
				tag = SEL
				tag = PRY
				AND = {
					tag = CAP
					primary_culture = macedonian
				}
				has_variable = pleistarchus_satrapy_flag
				has_variable = asia_satrapy
				has_variable = galatia_satrapy
			}
		}
	}
	
	attacker = {
		conquer_cost = 0.6
	}
	defender = {
		conquer_cost = 0.6
	}
	ticking_war_score = 0.5

}

imperial_conquest_wargoal = {
	type = superiority
	uses_civil_war_conquest = yes
	allow = { #Will this show up in the declare war window.
		is_subject = no
		OR = {
			invention = land_won_by_the_spear
			has_variable = horde_invasion_origin #Parthian Unification
		}
		scope:defender = {
			rank >= major_power
		}
	}
	
	attacker = {
		conquer_cost = 0.6
	}
	defender = {
		conquer_cost = 0.6
	}
	ticking_war_score = 0.5

}

conquer_wargoal = {
	type = take_province
	allow = {
		OR = {
			AND = {
				is_subject = yes
				is_subject_type = vassal_tribe
				is_subject_type = tributary
				is_subject_type = satrapy
				is_subject_type = fiefdom
			}
			is_subject = no	
		}

		trigger_if = {
			limit = {
				is_subject = yes
			}
			scope:defender = {
				OR = {
					AND = {
						is_subject = yes
						is_subject_type = vassal_tribe
						is_subject_type = tributary
						is_subject_type = satrapy
						is_subject_type = fiefdom
					}
					is_subject = no	
				}
	           scope:defender.overlord = {
	                NOT = {
	                    THIS = root.overlord
	                }
	            }
			}
		}
	}

	attacker = {
		conquer_cost = -0.25
	}
	defender = {
	}
	ticking_war_score = 1
}
