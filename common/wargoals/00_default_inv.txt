maurya_decline_wargoal = {
	type = superiority
	allow = {	#Will this show up in the declare war window.
		NOT = { has_global_variable = mauryan_collapse_prevented }
		current_date < 600.1.1
		any_neighbour_country = {
			tag = MRY
		}		
		is_subject = no
		OR = {
			country_culture_group = aryan
			country_culture_group = indian
			country_culture_group = pracyan
			country_culture_group = tibetan
			country_culture_group = hellenic
			country_culture_group = persia
		}
    	global_var:acting_as_ashoka = {
    		is_alive = no
    	}
		scope:defender = {
			tag = MRY
		}
	}
	
	attacker = {
		conquer_cost = -0.1	#Really Low
	}
	defender = {
		conquer_cost = 1.5 		#Really High
	}
	ticking_war_score = 1	
}
maurya_revival_wargoal = {
	type = superiority
	allow = {	#Will this show up in the declare war window.
		has_variable = revival_war_goal_enabled
		tag = MRY
		is_subject = no
		scope:defender = {
			any_owned_province = {
				is_target_in_global_variable_list = {
					name = MRY_land
					target = this
				}
			}
		}
	}
	
	attacker = {
		conquer_cost = 0.2
	}
	defender = {
		conquer_cost = 0.6
	}
	ticking_war_score = 1	
}

conquer_wargoal = {
	type = take_province
	allow = {
		OR = {
			AND = {
				is_subject = yes
				OR = {
					is_subject_type = vassal_tribe
					is_subject_type = tributary
					is_subject_type = temple_state
					is_subject_type = satrapy
					is_subject_type = fiefdom
				}
			}
			is_subject = no	
		}

		trigger_if = {
			limit = {
				scope:defender = { is_subject = yes }
			}
			scope:defender = {
	           scope:defender.overlord = {
	                NOT = {
	                    THIS = root.overlord
	                }
	            }
			}
		}
	}

	attacker = {
		conquer_cost = -0.25
	}
	defender = {
	}
	ticking_war_score = 1
}
