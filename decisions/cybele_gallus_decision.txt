country_decisions = {
	make_gallus = {
		potential = {
			num_of_cities >= 1
			religion = anatolian_religion
		}
		
		allow = {
			NOT = {
				has_variable = gallus_cooldown
			}
			political_influence >= 25
			any_character = {
				is_female = no
				trigger_if = {
				   limit = {
					  root = { is_monarchy = yes }
				   }
				   has_office = office_high_priest_monarchy
				   NOT = { has_trait = gallus }
				}
				trigger_if = {
				   limit = {
					  root = { is_republic = yes }
				   }
				   has_office = office_pontifex
				   NOT = { has_trait = gallus }
				}
				trigger_if = {
				   limit = {
					  root = { is_tribal = yes }
				   }
				   has_office = office_high_priest
				   NOT = { has_trait = gallus }
				}
			}
			custom_tooltip = {
				text = "gallus_tt"
			}
		}
		effect = {
			add_political_influence = -25
			random_character = {
				limit = {
					OR = {
						has_office = office_high_priest_monarchy
						has_office = office_pontifex
						has_office = office_high_priest
					}
				}
				force_add_trait = gallus
			}
			set_variable = {
				name = gallus_cooldown
				days = 9000
			}
		}
	}
	become_gallus = {
		potential = {
			num_of_cities >= 1
			religion = anatolian_religion
			current_ruler = {
				zeal >= 8
				is_female = no
			}
			is_ai = no
		}
		allow = {
			political_influence >= 30
			current_ruler = {
				zeal >= 8
				NOT = {
					has_variable = gallus
				}
			}
		}
		effect = {
			add_political_influence = -30
			current_ruler = {
				force_add_trait = gallus
				set_variable = gallus
			}
		}
	}
}
