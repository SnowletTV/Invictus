			OR = {
				country_culture_group = germanic
			}
			NOT = {
				tag = GEE
			}
			NOT = {
				is_endgame_tag_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {
				OR = {
					is_in_region = germania_region
					is_in_region = bohemia_area
					is_in_region = scandinavia_region
					#is_in_region = germania_superior_region
					is_in_area = rugia_area

					is_in_area = vangionia_area
					is_in_area = treveria_area
					is_in_area = menapia_orientalis
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_germania_exists
				NOT = {
					any_country = {
						tag = GEE
					}
				}
			}
			can_form_nation_trigger = yes

				is_monarchy = yes

			}
			owns_or_subject_owns_region = germania_region

		effect = {
			custom_tooltip = "tier_3_formable_tooltip"
			change_country_name = "GERMANIA_NAME"
			hidden_effect = {
				change_country_adjective = "GERMANIA_ADJECTIVE"
				if = { #If Scandinavian, use special Scandinavian gfx
					limit = { primary_culture = scandian }
					change_country_color =  "germania_sca_color"
					change_country_flag = GERMANIA_SCA_FLAG
				}
				else_if = { #Special gfx for Anglo-Saxon Germania
					limit = { tag = ENG } #Those who remain in Germania'll get normal flag
					change_country_color =  "germania_color"
					change_country_flag = ANGLO_GERMANIA_FLAG
				}
				else_if = { #Special gfx for Frankish Germania
					limit = { tag = FRA } #Those who remain in Germania'll get normal flag
					#Map color change
					if = { #If player selected red map color earlier, they'll get a red map color
						limit = { has_variable = francia_red_map_color }
						change_country_color =  franco_germania_map_red
					}
					else_if = { #If player selected putrid green map color earlier, they'll get a putrid green map color

						}

							every_region_province = { add_claim = root }
						}
					}

						change_country_color =  franco_germania_map_blue
					}
					#Flag change
					if = { #If player selected red flag earlier, they'll get a red flag
						limit = { has_variable = francia_red_flag }
						change_country_flag = FRANCO_GERMANIA_FLAG_RED
					}
					else_if = { #If player selected putrid green flag earlier, they'll get a putrid green flag 
						limit = { has_variable = francia_green_flag }
						change_country_flag = FRANCO_GERMANIA_FLAG_GREEN
					}
					else = { #Fallback'll be blue
						change_country_flag = FRANCO_GERMANIA_FLAG_BLUE
					}
				}
				else_if = { #Special gfx for an Indo-Germanic-cultured Germania
					#Realistically, this should never happen, but if a player can find a way, they'll be rewarded
					limit = { primary_culture = indo_germanic }
					change_country_color =  "germania_brown"
					change_country_flag = INDO_GERMANIC_GERMANIA_FLAG
					trigger_event = { #Flavor event giving the player an option to breed elephants in the capital
						id = flavor_germanics.900
					}
				}
				else = { #Fallback'll use standard Germania flag
					change_country_color =  "germania_color"
					change_country_flag = GERMANIA_FLAG
				}

			base = 1
		}
