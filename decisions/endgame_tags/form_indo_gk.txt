country_decisions = {
	
	# Form Indo-Greek Kingdom
	form_indo_gk = {
		
		potential = {
			NOT = { has_variable = formed_indo_greek_kingdom }
			num_of_cities >= 1
			country_culture_group = hellenic
			is_monarchy = yes
			OR = {
				tag = PAR	
				tag = BAC
				tag = SEL
				tag = ARS
				has_variable = is_indo_greek_invader
			}
			NOT = {
				tag = IDC
			}
			NOT = {
				is_endgame_tag_trigger = yes
			}
			OR = {
				is_ai = no
				num_of_cities >= 20
			}
		}
		
		highlight = {
			scope:province = {
				OR = {
					is_in_area = gandhara_area
					is_in_area = taxila_area
					is_in_area = panchanada_area
					is_in_area = kasmira_area
					is_in_area = madra_area
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_indus_exists
				NOT = {
					any_country = {
						tag = IDC
					}
				}
			}
			is_monarchy = yes
			can_form_nation_trigger = yes
			owns_or_subject_owns_area = gandhara_area
			owns_or_subject_owns_area = taxila_area
			owns_or_subject_owns_area = panchanada_area
			owns_or_subject_owns_area = kasmira_area
			owns_or_subject_owns_area = madra_area
			trigger_if = {
				limit = { 
					tag = SEL
					is_ai = yes
				}
				NOT = {
					owns_or_subject_owns_area = asoristan_area
					owns_or_subject_owns_area = babylonia_area
				}
			}
		}
		
		effect = {
			custom_tooltip = "tier_3_formable_tooltip"
			hidden_effect = {
				if = {
					limit = { tag = BAC }
					set_variable = was_bactria
				}
			}
			show_as_tooltip = {
				add_country_modifier = {
					name = indo_greek_kingdom_modifier
					duration = -1
				}
			}
            trigger_event = indo_greek.1
			change_country_name = "INDUS_NAME"
			hidden_effect = {
				change_country_adjective = "INDUS_ADJECTIVE"
				change_country_color = "indus_color"
				change_country_tag = IDC
				change_country_flag = INDIA_FLAG
			}
			set_variable = formed_indo_greek_kingdom
		}
		ai_will_do = {
			base = 1
		}
	}
}
