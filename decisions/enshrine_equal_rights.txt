country_decisions = {
	
	enshrine_social_equity = {
		potential = {
			num_of_cities >= 1
			gender_equality = no
			OR = {
				gender_rules = yes
				has_global_variable = game_rule_gender_rules
			}
		}
		allow = {
			political_influence >= influence_large
			trigger_if = {
				limit = {
					has_variable = social_equity_var
				}
				custom_tooltip = {
					text = social_equity_tooltip
					NOT = {
						has_variable = social_equity_var
					}
				}
			}
		}
		effect = {
			add_political_influence = subtract_influence_large
			if = {
				limit = { 
					is_republic = yes
					has_party_type = traditionalist_party
				}
				add_party_approval = {
					party_type = traditionalist_party
					value = -10
				}
			}
			set_variable = {
				name = social_equity_var
				days = 7300
			}
			set_gender_equality = yes
			hidden_effect = {
				set_global_variable = enshrine_social_equity_used
			}
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
	abolish_social_equity = {
		potential = {
			num_of_cities >= 1
			gender_equality = yes
			OR = {
				gender_rules = yes
				has_global_variable = game_rule_gender_rules
			}
		}
		allow = {
			political_influence >= influence_large
			NOT = {
				current_ruler = {
					is_female = yes
				}
			}
			trigger_if = {
				limit = {
					has_variable = social_equity_var
				}
				custom_tooltip = {
					text = social_equity_tooltip
					NOT = {
						has_variable = social_equity_var
					}
				}
			}
		}
		effect = {
			add_political_influence = subtract_influence_large
			if = {
				limit = { 
					is_republic = yes
					has_party_type = traditionalist_party
				}
				add_party_approval = {
					party_type = traditionalist_party
					value = -10
				}
			}
			set_variable = {
				name = social_equity_var
				days = 7300
			}
			set_gender_equality = no
			every_character = {
				limit = {
					is_female = yes
					OR = {
						has_any_office = yes
						has_tech_office = yes
						AND = {
							is_general = yes
							loyalty >= 33
						}
						AND = {
							is_admiral = yes
							loyalty >= 33
						}
						is_governor = yes
					}
				}
				remove_all_positions = yes
			}
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
}
