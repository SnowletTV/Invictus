				heritage = sidon_heritage
				heritage = arados_heritage

				has_global_variable = expanded_kition_kothon
			}
		}

			scope:province = {
				province_id = 332
			}
		}

			owns_or_subject_owns = 332 # Kition
			trigger_if = {
				limit = {
					is_ai = yes
				}

					has_no_ongoing_construction = yes
					great_work_locator_is_free = secondary_great_work
				}
			}

				}

		}

				limit = { is_ai = yes }

			}

			custom_tooltip = "expand_kition_kothon_deadline"

				}

		            }

						great_work = "kition_kothon"
						locator = primary_great_work
						is_ancient = yes						
					}
				}
			}
		}
		ai_will_do = {
			factor = 1

				factor = 0

			}

	}
