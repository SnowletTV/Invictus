			OR = {
				government = despotic_monarchy

				government = theocratic_monarchy
				government = aristocratic_monarchy
				government = plutocratic_monarchy
				government = imperium
				government = dictatorship

			}

			OR = {
				has_law = work_for_the_best
				has_law = standing_army_law
			}

		}
		allow = {
			stability >= 40
		}
		effect = {
			trigger_event = {
				id = regency_events.1
			}
		}

		}
	}

				id = regency_events.2
			}
		}

			base = 0
		}
	}
