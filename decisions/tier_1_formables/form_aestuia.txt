country_decisions = {
	
	# Form Aestuia
	form_aestuia = {
		
		potential = {
			num_of_cities >= 1
			primary_culture = aestian
			is_tribal = yes
			NOT = {
				tag = AUA
			}
			OR = {
				is_ai = no
				num_of_cities >= 10
			}
			NOR = {
				is_tier_1_formable_trigger = yes
				is_tier_2_formable_trigger = yes
				is_endgame_tag_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {
				OR = {
					AND = {
						has_owner = yes
						owner = {
							primary_culture = aestian
						}
					}
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_aestuia_exists
				NOT = {
					any_country = {
						num_of_cities >= 1
						tag = AUA
					}
				}
			}
			is_tribal = yes
			can_form_nation_trigger = yes
			custom_tooltip = {
				text = any_aestian_country_tt
				OR = {
					any_country = {
						count = all
						filter = {
							primary_culture = aestian
							num_of_cities >= 1
							NOT = {
								this = root
							}
						}
						is_subject_of = root
					}
					AND = {
						num_of_cities >= 1
						any_country = {
							count = 1
							primary_culture = aestian
						}
					}
				}
			}
		}
		
		effect = {
			custom_tooltip = "tier_1_formable_tooltip"
			change_country_name = "AESTUIA_NAME"
			hidden_effect = {
				change_country_adjective = "AESTUIA_ADJECTIVE"
				change_country_color =  "aestuia_color"
				change_country_tag = AUA
			}
			capital_scope = {
				capital_formable_tribal_effect = yes
				formable_capital_modifier_tribal_effect = yes
			}
			add_country_modifier = {
				name = aestuia_modifier
				duration = 7300
			}
			if = {
				limit = {
					any_subject = {
						primary_culture = aestian
					}
				}
				every_subject = {
					limit = {
						primary_culture = aestian
					}
					every_owned_province = {
						limit = {
							OR = {
								is_in_region = germania_region
								is_in_region = vistulia_region
								is_in_region = bohemia_area
							}
						}
						set_owned_by = root
					}
				}
			}
			add_3_free_province_investments = yes
		}
		
		ai_will_do = {
			base = 1
		}
	}

}
