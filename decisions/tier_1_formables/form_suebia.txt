country_decisions = {
	
	# Form Suebia
	form_suebia = {
		
		potential = {
			num_of_cities >= 1
			OR = {
				primary_culture = irminonic
				primary_culture = suebian
			}
			is_tribal = yes
			NOT = {
				tag = SBN
			}
			OR = {
				is_ai = no
				num_of_cities >= 10
			}
			NOR = {
				is_tier_1_formable_trigger = yes
				is_tier_2_formable_trigger = yes
				is_endgame_tag_trigger = yes
			}
		}
		
		highlight = {
			scope:province = {
				OR = {
					AND = {
						has_owner = yes
						owner = {
							OR = {
								primary_culture = irminonic
								primary_culture = suebian
							}
						}
					}
					is_in_area = semnonia_area
					is_in_area = pharodenia_area
					is_in_area = langobardia_area
					is_in_area = suradonia_area
					is_in_area = caluconia_area
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = formable_not_suebia_exists
				NOT = {
					any_country = {
						num_of_cities >= 1
						tag = SBN
					}
				}
			}
			is_tribal = yes
			can_form_nation_trigger = yes
			OR = {
				custom_tooltip = {
					text = any_suebian_country_tt
					OR = {
						any_country = {
							count = all
							filter = {
								OR = {
									primary_culture = irminonic
									primary_culture = suebian
								}
								num_of_cities >= 1
								NOT = {
									this = root
								}
							}
							is_subject_of = root
						}
						AND = {
							num_of_cities >= 1
							any_country = {
								count = 1
								OR = {
									primary_culture = irminonic
									primary_culture = suebian
								}
							}
						}
					}
				}
				AND = {
					owns_or_subject_owns_area = semnonia_area
					owns_or_subject_owns_area = pharodenia_area
					owns_or_subject_owns_area = langobardia_area
					owns_or_subject_owns_area = suradonia_area
					owns_or_subject_owns_area = caluconia_area
				}
			}
		}
		
		effect = {
			custom_tooltip = "tier_1_formable_tooltip"
			change_country_name = "SUEBIA_NAME"
			hidden_effect = {
				change_country_adjective = "SUEBIA_ADJECTIVE"
				change_country_color =  "suebia_color"
				change_country_tag = SBN
			}
			capital_scope = {
				capital_formable_tribal_effect = yes
				formable_capital_modifier_tribal_effect = yes
			}
			add_country_modifier = {
				name = suebia_modifier
				duration = 7300
			}
			if = {
				limit = {
					any_subject = {
						OR = {
							primary_culture = irminonic
							primary_culture = suebian
						}
					}
				}
				every_subject = {
					limit = {
						OR = {
							primary_culture = irminonic
							primary_culture = suebian
						}
					}
					every_owned_province = {
						limit = {
							OR = {
								is_in_region = germania_region
								is_in_region = vistulia_region
								is_in_region = bohemia_area
							}
						}
						set_owned_by = root
					}
				}
			}
			add_3_free_province_investments = yes
		}
		
		ai_will_do = {
			base = 1
		}
	}

}
