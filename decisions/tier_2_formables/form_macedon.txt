			}

				tag = MAC
			}
			NOT = {
				tag = MCD
			}
			OR = {
				is_ai = no
				num_of_cities >= 20

			NOT = {
				is_endgame_tag_trigger = yes

			}
			NAND = {
				tag = PRY

				NOT = { has_completed_mission = antigonid_02_aspiration }

		}
		
		highlight = {
			scope:province = {

						tag = MCD
					}
				}

			owns = 379 #Pella
			owns = 428 #Aegae
			owns = 365 #Amphipolis
			owns = 394 #Pharsalos
			owns = 3175 #Eratyra

						is_culture = macedonian
					}
				}
				any_country_culture = {
					is_integrated = yes
					is_culture = macedonian
				}
			}
		}
