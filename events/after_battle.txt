after_battle.1 = {
	type = character_event
	hidden = yes
	picture = elephant_battle
	
	trigger = {
		NOR = {
			has_trait = victorious

			has_trait = comradely
		}
		is_adult = yes
	}
	
	weight_multiplier = {
		modifier = {
			add = {
				value = root.martial
				divide = 4
			}
		}
	}
	
	immediate = {
		save_scope_as = after_battle_1_target
		if = {
			limit = {
				exists = employer 
			}
			employer = {
				trigger_event = {
					id = after_battle.11
				}
			}
		}
	}
}

after_battle.14 = {
	type = country_event
	title = "after_battle.4.t"
	desc = "after_battle.4.d"
	picture = revolt
	
	left_portrait = current_ruler
	right_portrait = scope:after_battle_4_target
	
	trigger = {
		scope:after_battle_4_target = {
			is_alive = yes
			NOT = { has_loyalty = loyalty_fled_battle }
			is_adult = yes
		}
	}
	
	option = {
		name = "after_battle.4.a"
		scope:after_battle_4_target = {
			random_list = {
				0 = { 
					if = {
						limit = {
							has_trait = brave
						}
						add_popularity = 15
					}
					else = {
						if = {
							limit = {
								has_trait = coward
							}
							remove_trait = coward
						}
						add_trait = brave
					}
				}
				0 = { add_popularity = 10 }
				2 = {
					if = {
						limit = { has_legion_trigger = yes }
						hidden_effect = {
							legion = {
								add_legion_history = { #
									key = commander_death_history
									commander = scope:after_battle_4_target
									province = scope:after_battle_4_target.location
								}
							}
						}
					}
					death = { death_reason = death_battle }
				}
			}
		}
	}
	option = {
		name = "after_battle.4.b"
		scope:after_battle_4_target = {
			if = {
				limit = {
					has_trait = coward
				}
				add_loyalty = loyalty_fled_battle
				add_popularity = -10
			}
			else = {
				if = {
					limit = {
						has_trait = brave
					}
					remove_trait = brave
				}
				add_trait = coward
			}
		}
	}
}

after_battle.20 = {
	type = country_event
	title = "after_battle.10.t"
	desc = "after_battle.10.d"
	picture = looting
	
	left_portrait = current_ruler
	right_portrait = scope:after_battle_20_target
	
	trigger = {
		scope:after_battle_20_target = {
			is_alive = yes
			is_adult = yes
		}
	}
	
	option = {
		name = "after_battle.10.a"
		#Increased amount of injuries also increases chance to die.
		scope:after_battle_20_target = {
			random_list = {
				30 = {  }
				20 = { 
					trigger = {
						NOT = { has_trait = wounded }
					}
					add_trait = wounded
				}
				20 = { 
					trigger = {
						NOT = { has_trait = maimed }
					}
					add_trait = maimed
				}
				20 = { 
					trigger = {
						NOT = { has_trait = limp }
					}
					add_trait = limp
				}
				10 = {
					if = {
						limit = { has_legion_trigger = yes }
						hidden_effect = {
							legion = {
								add_legion_history = { #
									key = commander_death_history
									commander = scope:after_battle_20_target
									province = scope:after_battle_20_target.location
								}
							}
						}
					}
					death = {
						death_reason = death_succumbed_to_injuries
					}
				}
			}
		}
	}
