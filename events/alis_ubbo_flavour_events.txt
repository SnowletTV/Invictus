namespace = alis_ubbo_events

alis_ubbo_events.1 = { # Tribe seeks asylum
	type = country_event
	title = "alis_ubbo_events.1.t"
	desc = "alis_ubbo_events.1.desc"
	picture = alis_ubbo_tribe

	trigger = {
		is_ai = no
		NOT = {
			has_variable = alis_ubbo_events1_variable
		}
	}
	
	immediate = {
		set_variable = {
			name = alis_ubbo_events1_variable
			days = 365
		}
	}
	
	option = {
		name = "alis_ubbo_events.1.a"
		current_ruler = {
			add_popularity = popularity_small
		}
	}
	
	option = {
		name = "alis_ubbo_events.1.b"
		capital_scope = {
        		define_pop = {
        			type = tribesmen
        			religion = druidism
        			culture = turdulian
        		}
        		define_pop = {
        			type = tribesmen
        			religion = druidism
        			culture = turdulian
        		}
        		define_pop = {
        			type = tribesmen
        			religion = druidism
        			culture = turdulian
        		}
		}
		every_country_culture = {
			limit = {
				OR = {
					is_culture = ROOT.culture
					is_integrated = yes
				}
			}
			add_country_culture_modifier = {
				name = "alis_ubbo_tribe_asylum_modifier"
				duration = 1825
			}
		}
	}
}

alis_ubbo_events.4 = { # Bribe
	type = country_event
	title = "alis_ubbo_events.4.t"
	desc = "alis_ubbo_events.4.desc"
	picture = alis_ubbo_greek_city_man

	left_portrait = current_ruler
	right_portrait = scope:alis_ubbo_briber

	trigger = {
		is_ai = no
		NOT = {
			has_variable = alis_ubbo_events4_variable
		}
	}
	
	immediate = {
		set_variable = {
			name = alis_ubbo_events4_variable
			days = 365
		}
		random_character = {
			limit = {
				age > 25
				age < 90
				is_female = no
				country = employer
			}
			save_scope_as = alis_ubbo_briber
		}
		hidden_effect = {
    		c:ROM = {
				add_country_modifier = {
					name = alis_ubbo_temp_modifier
					duration = -1
				}
	    	}
    		c:CAR = {
				add_country_modifier = {
					name = alis_ubbo_temp_modifier
					duration = -1
				}
	    	}
		}
	}
	
	option = {
		name = "alis_ubbo_events.4.b"
		current_ruler = {
			add_corruption = 2
		}
		add_treasury = 30.0
	}
	
	option = {
		name = "alis_ubbo_events.4.a"
		current_ruler = {
			add_popularity = popularity_small
		}
	}
}
