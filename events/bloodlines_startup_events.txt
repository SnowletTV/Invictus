bloodlines_startup.14 = { #Senonia

	type = country_event
	hidden = yes

	trigger = {
		tag = SNI
	}

	immediate = {

		create_character = {
			first_name="Venextos"
			family_name = "Brennus"
			birth_date="398.6.6"
			culture="sennonian"
			religion="druidism"
			no_stats=yes
			add_martial=12
			add_charisma=9
			add_finesse=5
			add_zeal=6
			add_trait="brennus"
			add_gold=600
			add_popularity=80
			save_scope_as = brennus
		}

		random_character = {
			limit = {
				is_head_of_family = yes
			}
			save_scope_as = unlucky_lad
		}

		every_character = {
			limit = {
				has_same_family = scope:unlucky_lad
			}
			set_as_minor_character = this
		}

		new_family_generation_effect = { FAMILY_CANDIDATE = brennus }

		set_as_ruler = scope:brennus

		every_character = {
			limit = {
				has_same_family = scope:brennus
			}
			force_add_trait = brennus
		}
		hidden_effect = {
			trigger_event = { id = bloodlines_setup_completion.1 }
		}
	}
}

		}

			first_name="Mago"
			family_name = "Magonid"
			culture="carthaginian"
			religion="carthaginian_pantheon"
			add_trait="magonid"
			save_scope_as = magonid
		}

		random_character = {
			limit = {
				is_head_of_family = yes

					has_trait = barca

			limit = {

			}

		}

			limit = {
