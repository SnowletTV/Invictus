		random_character = {
			limit = {
				OR = {
					has_office = office_high_priest
					has_office = office_high_priest_monarchy
					has_office = office_pontifex
				}
			}
			save_scope_as = murder_priest
		}
	}
	option = { #choice 1
		name = child_sacrifice_inv.1.a
		scope:sacrifice_1 = {
			death = {

		scope:sacrifice_2 = {
			add_character_modifier = {
