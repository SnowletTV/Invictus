country_rename.4 = {
	type = country_event
	title = "country_rename.4.t"
	desc = "country_rename.4.desc"

	option = {
		name = "country_rename.4.a"
		trigger = {
			exists = c:ETR
			c:ETR = {
				has_land = yes
			}
		}
		show_as_tooltip = {
			change_government = tyrannic_monarchy
			custom_tooltip = "change_ruler_for_tarquinius"
			c:ETR = {
				make_subject = {
					target = ROOT
					type = feudatory
				}
				every_owned_province = {
					limit = {
						is_in_area = roma_area
					}
					set_owned_by = c:ROM
				}
			}
		}
		c:ETR = {
			trigger_event = {
				id = country_rename.6
			}
		}
		ai_chance = {
			modifier = {
				factor = 60
			}
		}
	}
	
	option = {
		name = "country_rename.4.b"
		custom_tooltip = "no_way_tooltip"
		show_as_tooltip = {
			c:ROM = {
				change_country_name = "LOST_ROME"
			}
		}
		c:ETR = {
			trigger_event = {
				id = country_rename.5
			}
		}
		ai_chance = {
			modifier = {
				factor = 40
			}
			modifier = {
				factor = 0
				c:ROM = {
					num_of_cities >= 50
				}
			}
		}
	}
	
}

		c:ROM = {
			change_government = despotic_monarchy
			custom_tooltip = "change_ruler_for_tarquinius"
			hidden_effect = {
				create_character = {
					first_name = "Tarquinius"
					age = 32
					culture = etruscan
					religion = roman_pantheon
					save_scope_as = tarquinius
				}
				set_as_ruler = scope:tarquinius
			}
			add_truce = {
				target = c:ETR
				duration = 1825 # 5 years
			}
			c:ETR = {
				make_subject = {
					target = c:ROM
					type = feudatory
				}
				every_owned_province = {
					limit = {
						is_in_area = roma_area
						NOT = {
							this = root.capital_scope
						}
					}
					set_owned_by = c:ROM
				}
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					is_ai = no
					NOR = { 
						tag = ETR 
						tag = ROM
					}
				}
				trigger_event = {
					id = country_rename.7
				} 
			}
		}
	}
