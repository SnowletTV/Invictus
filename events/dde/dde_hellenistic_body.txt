dhe_body.9 = {
	type = country_event
	title = "dhe_body.9.t"
	desc = "dhe_body.9.desc"
	
	picture = mausoleum
	
	left_portrait = current_ruler
	right_portrait = scope:relative_of_alexander
	
	trigger = {
		has_dlc = "Hellenistic World Flavor Pack"
		NOT = {
			has_variable = dhe_body_9_toggle
		}
		is_monarchy = yes
		any_owned_province = {
			has_variable = mausoleum_constructed
		}
		any_character = {
			has_culture_group = hellenic
			trigger_if = {
				limit = {
					employer.current_ruler = {
						is_minor_character = no
					}
				}
				from_ruler_family = no
			}
			family = {
				prestige > 2000
			}
			NOT = {
				has_trait = argeads
			}
		}
	}
	
	immediate = {
		set_variable = {
			name = dhe_body_9_toggle
		}
		random_character = {
			limit = {
				exists = family
				family = {
					prestige >= 2000
				}
				has_culture_group = hellenic
				from_ruler_family = no
				NOT = {
					has_trait = argeads
				}
			}
			weight = {
				modifier = {
					factor = 50
					martial > 10
				}
				modifier = {
					factor = 50
					finesse > 10
				}
				modifier = {
					factor = 50
					charisma > 10
				}
				modifier = {
					factor = 50
					zeal > 10
				}
			}
			save_scope_as = relative_of_alexander
		}
	}
	
	option = {
		name = "dhe_body.9.a"
		add_legitimacy = -25
		scope:relative_of_alexander = {
			family = {
				every_family_member = {
					limit = {
						is_male = yes
						employer = root
						NOT = {
							has_trait = argeads
						}
						NOT = {
							this = scope:relative_of_alexander
						}
					}
					hidden_effect = {
						add_trait = argeads
						add_loyalty = loyalty_argead_blood
					}
				}
			}
		}
		scope:relative_of_alexander = {
			add_trait = argeads
		}
		custom_tooltip = argead_and_loyalty_tooltip
	}
	
	option = {
		name = "dhe_body.9.b"
		add_tyranny = 10

			family = {
				every_family_member = {
					limit = {
						employer = root
					}

					}
				}
			}
		}
