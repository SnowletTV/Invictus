namespace = dhe_dde_pyrrhus

dhe_dde_pyrrhus.1 = { # Molossian Ultimatum
	type = country_event
	title = "dhe_dde_pyrrhus.1.t"
	desc = {
		first_valid = {
			triggered_desc = { # Neoptolemus is available
				trigger = {
					char:118 = { # Is Neoptolemus available?
						is_alive = yes
						is_ruler = no
						employer = root
					}
				}
				desc = "dhe_dde_pyrrhus.1.desc_1"
			}
			triggered_desc = { # Neoptolemus is unavailable
				trigger = {
					always = yes
				}
				desc = "dhe_dde_pyrrhus.1.desc_2"
			}
		}
	}
	picture = greek_court
	left_portrait = scope:pyrrhus
	right_portrait = scope:neoptolemus
	fire_only_once = yes

	trigger = { # Is Epirus, Pyrrhus is alive, Pyrrhus is ruler, Neoptolemus or Other char available
		has_dlc = "Epirus Flavor Pack"
		tag = EPI
		has_land = yes
		char:70 = { # Pyrrhus
			is_alive = yes
			is_ruler = yes
			employer = root
		}
		OR = {
			char:118 = { # Neoptolemus
				is_alive = yes
				is_ruler = no
				employer = root
			}
			any_character = { # Fallback Aeacid
				is_ruler = no
				trigger_if = {
					limit = {
						employer.current_ruler = {
							is_minor_character = no
						}
					}
					from_ruler_family = yes
				}
				age >= 20
				is_female = no
				prisoner = no
			}
			any_character = { # Fallback Epirote
				is_ruler = no
				age >= 20
				is_female = no
				prisoner = no
			}
		}
	}

	immediate = { # Save scopes
		set_variable = epi_molossian_ultimatum_flag
		char:70 = { save_scope_as = pyrrhus }
		if = { # Neoptolemus
			limit = {
				NOT = { exists = scope:neoptolemus }
				me_epirus_4_3_neoptolemus_eligible_trigger = yes
			}
			char:118 = { save_scope_as = neoptolemus }
		}
		else_if = { # Fallback Aeacid
			limit = {
				NOT = { exists = scope:neoptolemus }
				any_character = { me_epirus_4_3_aeacid_pretender_trigger = yes }
			}
			ordered_character = {
				limit = { me_epirus_4_3_aeacid_pretender_trigger = yes }
				order_by = num_of_supporters
				save_scope_as = neoptolemus
			}
		}
		else_if = { # Fallback Epirote
			limit = {
				NOT = { exists = scope:neoptolemus }
			}
			ordered_character = {
				limit = { me_epirus_4_3_fallback_pretender_trigger = yes }
				order_by = num_of_supporters
				save_scope_as = neoptolemus
			}
		}
	}

	option = { # Depose Pyrrhus
		name = "dhe_dde_pyrrhus.1.a"
		set_as_ruler = scope:neoptolemus # Set new ruler
		if = {
			limit = {
				c:TLN.current_ruler = {
					this = char:105
					is_friend = char:70
				}
			}
			reverse_add_opinion = { modifier = dhe_dde_pyrrhus_deposed_pyrrhus_op target = c:TLN } # Add -50 opinion of Epirus to Taulantia
			char:105 = { add_rival = scope:neoptolemus }
			if = { # Break alliance with TLN if revelant
				limit = { alliance_with = c:TLN }
				break_alliance = c:TLN
				add_truce = {
					target = c:TLN
					duration = 1825 # 5 years
				}
			}
		}
		c:MAC = {
			if = {
				limit = {
					has_opinion = {
						modifier = pyrrhos_crowned_opmod
						target = c:EPI
					}
				}
				remove_opinion = {
					modifier = pyrrhos_crowned_opmod
					target = c:EPI
				}
			}
		}
		dhe_dde_pyrrhus_1_move_effect = yes # Trigger event .2 for picked country
		custom_tooltip = dhe_dde_pyrrhus.1.at

		set_variable = pyrrhos_left_epirus
		char:70 = {
			set_variable = {
				name = pyrrhus_travelling
				days = 2000
			}
			trigger_event = {
				id = dhe_dde_pyrrhus.9
				days = { 1750 1850 }
			}
        }
		if = {
			limit = {
				is_ai = yes
			}
			add_country_modifier = { # Global unrest +4.0, Monthly legitimacy -0.30
				name = pyrrhos_ai_buff_money
				duration = 9125 # 10 years
			}
		}
		ai_chance = { factor = 95 }
	}

	option = { # Support Pyrrhus
		name = "dhe_dde_pyrrhus.1.b"
		scope:pyrrhus = { add_popularity = subtract_popularity_medium } # Pyrrhus loses popularity
		if = {
			limit = {
				NOT = {
					scope:pyrrhus = { is_rival = scope:neoptolemus }
				}
				scope:pyrrhus = { can_get_rivals = yes }
				scope:neoptolemus = { can_get_rivals = yes }
			}
			scope:neoptolemus = { add_rival = scope:pyrrhus }
		}
		add_country_modifier = { # Global unrest +4.0, Monthly legitimacy -0.30
			name = dhe_dde_pyrrhus_molossians_angered_cm
			duration = 3650 # 10 years
		}
		add_legitimacy = subtract_legitimacy_huge # Epirus loses legitimacy
		remove_stability_effect = yes # Epirus loses stability
		reverse_add_opinion = { modifier = dhe_dde_pyrrhus_angered_macedon_op target = c:MAC } # Add -75 opinion of Epirus to Macedon
		trigger_event = {
			id = dhe_dde_pyrrhus.39
			days = { 100 200 }
		}
		set_variable = { name = settle_disputes_flag }
		if = {
			limit = {
				is_ai = yes
			}
			add_country_modifier = { # Global unrest +4.0, Monthly legitimacy -0.30
				name = pyrrhos_ai_buff_money
				duration = 9125 # 10 years
			}
		}
		ai_chance = { factor = 5 }
	}
}

			limit = {

					char:118 = { is_rival = scope:pyrrhus }
				}
			}
			char:118 = { add_rival = scope:pyrrhus }
		}

			limit = {

				}
			}
		}

	}

	option = { # Pyrrhus returns to court as a failure

		name = dhe_dde_pyrrhus.5.d

		pyrrhus_family_mover_effect = yes

	title = "dhe_dde_pyrrhus.33.t"
	desc = "dhe_dde_pyrrhus.33.desc"
	picture = pyrrhus
	left_portrait = scope:actor.commander
	right_portrait = scope:target.commander
	fire_only_once = yes

	trigger = {
		has_dlc = "Epirus Flavor Pack"
		current_ruler = char:70
		has_land = yes
	}

	option = {
		name = "dhe_dde_pyrrhus.33.a"
		scope:actor.commander = {
			if = {
				limit = { popularity <= 99 }
				add_popularity = popularity_medium
			}
			add_nickname = "NICKNAME_AETOS"
			if = {
				limit = {
					NOT = { has_trait = wounded }
				}
				add_trait = wounded
			}
		}
		scope:target.commander = {
			if = {
				limit = { popularity >= 1 }
				add_popularity = subtract_popularity_large
			}
			if = {
				limit = {
					NOT = { has_trait = wounded }
				}
				add_trait = wounded
			}
		}
		scope:actor = {
			if = {
				limit = {
					any_sub_unit = {
						OR = {
							has_personal_loyalty = no
							AND = {
								has_personal_loyalty = yes
								NOT = { personal_loyalty = scope:actor.commander }
							}
						}
					}
				}
				while = {
					count = 3
					limit = {
						any_sub_unit = {
							OR = {
								has_personal_loyalty = no
								AND = {
									has_personal_loyalty = yes
									NOT = { personal_loyalty = scope:actor.commander }
								}
							}
						}
					}
					random_sub_unit = {
						limit = {
							OR = {
								has_personal_loyalty = no
								AND = {
									has_personal_loyalty = yes
									NOT = { personal_loyalty = scope:actor.commander }
								}
							}
						}
						set_personal_loyalty = scope:actor.commander
					}
				}
			}
		}
		scope:target.unit_owner = {
			trigger_event = { id = dhe_dde_pyrrhus.34 }
		}
	}
}

dhe_dde_pyrrhus.34 = { # Duel ping
	type = country_event
	title = "dhe_dde_pyrrhus.33.t"
	desc = "dhe_dde_pyrrhus.34.desc"
	picture = pyrrhus
	left_portrait = scope:target.commander
	right_portrait = scope:actor.commander

	trigger = { has_dlc = "Epirus Flavor Pack" }

	immediate = {
		show_as_tooltip = {
			scope:target.commander = {
				add_popularity = subtract_popularity_large
				add_trait = wounded
			}
			scope:actor.commander = {
				add_popularity = popularity_medium
				add_nickname = "NICKNAME_AETOS"
				add_trait = wounded
			}
		}
		scope:target = {
			if = {
				limit = {
					any_sub_unit = {
						has_personal_loyalty = yes
						personal_loyalty = scope:target.commander
					}
				}
				while = {
					count = 3
					limit = {
						any_sub_unit = {
							has_personal_loyalty = yes
							personal_loyalty = scope:target.commander
						}
					}
					random_sub_unit = {
						limit = {
							has_personal_loyalty = yes
							personal_loyalty = scope:target.commander
						}
						remove_subunit_loyalty = this
					}
				}
			}
		}
	}

	option = { name = "dhe_dde_pyrrhus.34.a" }
}

dhe_dde_pyrrhus.36 = { # Opalcus
	type = country_event
	title = "dhe_dde_pyrrhus.36.t"
	desc = {
		desc = "dhe_dde_pyrrhus.36.desc"
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:actor.commander = { is_alive = no }
				}
				desc = "dhe_dde_pyrrhus.36.desc_1"
			}
			triggered_desc = {
				trigger = {
					scope:actor.commander = { is_alive = yes }
				}
				desc = "dhe_dde_pyrrhus.36.desc_2"
			}
		}
	}
	picture = pyrrhus
	left_portrait = char:70
	left_portrait = scope:leonnatus_scope
	right_portrait = scope:target.commander
	right_portrait = scope:opalcus_scope

	trigger = {
		has_land = yes
		has_dlc = "Epirus Flavor Pack"
		current_ruler = char:70
	}

	immediate = {
		hidden_effect = {
			create_character = {
				first_name = "Leonnatos"
				culture = epirote
				religion = roman_pantheon
				age = 29
				female = no
				no_traits = yes
				add_trait = brave
				add_trait = cautious
				no_stats = yes
				add_martial = 7
				add_finesse = 6
				add_charisma = 7
				add_zeal = 5
				save_scope_as = leonnatus_scope
			}
			scope:target.unit_owner = {
				create_character = {
					first_name = "Opalcus"
					culture = ligurian
					religion = roman_pantheon
					age = 36
					female = no
					no_traits = yes
					add_trait = brave
					add_trait = reckless
					save_scope_as = opalcus_scope
				}
			}
			scope:opalcus_scope = {
				death = {
					death_reason = death_stabbing
					killer = scope:leonnatus_scope
				}
			}
		}
		hidden_effect = {
			random_list = {
				60 = {}
				30 = {
					scope:actor.commander = { set_variable = pyrrhos_wound_flag }
					modifier = {
						factor = 2
						scope:actor.commander = { has_trait = reckless }
					}
				}
				10 = {
					trigger = {
						root = { is_ai = yes }
					}
					scope:actor.commander = {
						death = {
							death_reason = death_stabbing
							killer = scope:opalcus_scope
						}
					}
					modifier = {
						factor = 2
						scope:actor.commander = { has_trait = reckless }
					}
				}
			}
		}
		scope:actor.commander = {
			if = {
				limit = { has_trait = reckless }
				remove_trait = reckless
			}
			else = {
				add_trait = cautious
			}
		}
	}

	option = {
		name = "dhe_dde_pyrrhus.36.a"
		trigger = {
			scope:actor.commander = { is_alive = no }
		}
	}

	option = {
		name = "dhe_dde_pyrrhus.36.b"
		trigger = {
			scope:actor.commander = {
				is_alive = yes
				has_variable = pyrrhos_wound_flag
			}
		}
		scope:actor.commander = {
			if = {
				limit = {
					NOR = {
						has_trait = maimed
						has_trait = wounded
					}
				}
				add_trait = wounded
			}
			else_if = {
				limit = { has_trait = wounded }
				add_trait = maimed
			}
			else_if = {
				limit = { has_trait = maimed }
				add_health = -10
			}
		}
		scope:leonnatus_scope = {
			add_popularity = popularity_small
			add_martial = 1
		}
		if = {
			limit = {
				char:70 = { can_get_friends = yes }
				scope:leonnatus_scope = { can_get_friends = yes }
			}
			char:70 = { add_friend = scope:leonnatus_scope }
		}
	}

	option = {
		name = "dhe_dde_pyrrhus.36.c"
		trigger = {
			scope:actor.commander = {
				is_alive = yes
				NOT = { has_variable = pyrrhos_wound_flag }
			}
		}
		scope:leonnatus_scope = {
			add_popularity = popularity_small
			add_martial = 1
		}
		if = {
			limit = {
				char:70 = { can_get_friends = yes }
				scope:leonnatus_scope = { can_get_friends = yes }
			}
			char:70 = { add_friend = scope:leonnatus_scope }
		}
	}

	after = {
		if = {
			limit = {
				scope:actor.commander = { has_variable = pyrrhos_wound_flag }
			}
			scope:actor.commander = { remove_variable = pyrrhos_wound_flag }
		}
	}
}

		has_dlc = "Epirus Flavor Pack"
	}

	immediate = {
		char:70 = { save_scope_as = pyrrhus }

			if = {
				limit = {

					prisoner = no
					NOT = { is_friend = char:70 }
				}
				save_scope_as = neoptolemus
			}
			else = {
				root = {
					random_character = {
						limit = {
							family = char:70.family
							is_alive = yes
							prisoner = no
							employer = root
							NOT = { is_friend = char:70 }
						}
						save_scope_as = neoptolemus
					}
				}
			}
		}
	}

	option = { # Pyrrhus
		name = "dhe_dde_pyrrhus.37.a"
		custom_tooltip = dhe_dde_pyrrhus.37.at
		set_variable = pyrrhos_supported
		ai_chance = {
			factor = 99
		}
	}

	option = { # Neoptolemus
		name = "dhe_dde_pyrrhus.37.b"
		trigger = { scope:neoptolemus = char:118 }
		custom_tooltip = dhe_dde_pyrrhus.37.bt
		ai_chance = {
			factor = 0
		}
	}

	option = { # Other
		name = "dhe_dde_pyrrhus.37.c"
		trigger = {
			NOT = { scope:neoptolemus = char:118 }
		}
		custom_tooltip = dhe_dde_pyrrhus.37.bt
		ai_chance = {
			factor = 0
		}
	}

	after = {
		trigger_event = {
			id = dhe_dde_pyrrhus.38
			days = 10
		}
	}
}

dhe_dde_pyrrhus.38 = { # Disputes 2
	type = country_event
	title = "dhe_dde_pyrrhus.38.t"
	desc = {
		desc = "dhe_dde_pyrrhus.38.desc"
		first_valid = {
			triggered_desc = { # Neoptolemus is available
				trigger = { has_variable = pyrrhos_supported }
				desc = "dhe_dde_pyrrhus.38.desc_1"
			}
			triggered_desc = { # Neoptolemus is unavailable
				trigger = {
					always = yes
				}
				desc = "dhe_dde_pyrrhus.38.desc_2"
			}
		}
		desc = "dhe_dde_pyrrhus.38.desc_3"
	}
	picture = assassination
	left_portrait = current_ruler
	right_portrait = scope:depose_target

	trigger = {
		has_land = yes
		has_dlc = "Epirus Flavor Pack"
		char:70 = {
			is_alive = yes
			is_ruler = yes
			employer = root
		}
		scope:neoptolemus = {
			is_alive = yes
			employer = root
		}
	}

	immediate = {
		if = {
			limit = { has_variable = pyrrhos_supported }
			scope:neoptolemus = { save_scope_as = depose_target }
		}
		else = {
			char:70 = { save_scope_as = depose_target }
			set_as_ruler = scope:neoptolemus
		}
		scope:depose_target = {
			death = {
				death_reason = death_execution
				killer = root.current_ruler
			}
		}
		set_variable = { name = settle_disputes_flag }
		custom_tooltip = dhe_dde_pyrrhus.38.tt
	}

	option = {
		name = "dhe_dde_pyrrhus.38.a"
		trigger = { has_variable = pyrrhos_supported }
		remove_variable = pyrrhos_supported
		add_stability = 10
		add_legitimacy = -5
	}

	option = {
		name = "dhe_dde_pyrrhus.38.b"
		trigger = {
			NOT = { has_variable = pyrrhos_supported }
		}
		add_stability = -5
		add_legitimacy = -10
	}
}
