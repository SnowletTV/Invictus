			}

				id = bloodlines_setup_completion.1
			}

		}
	}

	option = {
		name = "dhe_parthians.1.a"
		set_as_ruler = scope:arsaces_character
		hidden_effect = {
			scope:arsaces_character = {
				add_gold = 900
			}
		}
		add_country_modifier = {
			name = rise_of_arsaces_cmod
			duration = 9125
		}
	}
