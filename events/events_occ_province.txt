namespace = province_siege

province_siege.1 = {
	type = province_event
	hidden = yes
	
	trigger = {
		NOT = {
			has_global_variable = diadochi_old
		}
		exists = owner
		exists = controller
	}

	immediate = {

		# add AE and WE for the territory
		if = {
			limit = {
				owner = {
					NOT = {
						is_target_in_variable_list = {
							name = my_land
							target = ROOT
						}
					}
					any_current_war = {
						any_war_attacker = {
							has_variable = war_of_the_diadochs
							root = {
								is_previous_owner = prev
							}
						}
					}
				}
			}

			show_animated_text = ae_we
			root.owner = {
				add_aggressive_expansion = 0.32
				add_war_exhaustion = 0.1
			}
		}


		if = {
			limit = {
				owner = {
					has_variable = war_of_the_diadochs
					any_countries_at_war_with = {
						has_variable = war_of_the_diadochs
						root = {
							is_previous_owner = prev
						}
					}
				}
				#do not flip if there are hostile forts
				area = {
					NOT = {
						any_area_province = {
							owner = {
								root = {
									is_previous_owner = prev
								}
							}
							num_of_fortress_building > 0
						}
					}
					#do not flip if there are hostile units in either our land or theirs
					NOT = {
						any_area_province = {
							owner = {
								OR = {
									this = root.owner
									root = {
										is_previous_owner = prev
									}
								}
							}
							any_unit_in_province = {
								unit_owner = {
									war_with = root.owner
								}
							}
						}
					}
				}
			}

			# flip
			area = {
				every_area_province = {
					limit = {
						owner = {
							root = {
								is_previous_owner = prev
							}
						}
						controller = owner
					}
					set_conquered_by = root.controller

					#add AE and WE for all flipped territores
					if = {
						# add only to the attacker
						# limit = {
						# 	root.owner = {
						# 		any_current_war = {
						# 			any_war_defender = {
						# 				has_variable = war_of_the_diadochs
						# 				NOT = {
						# 					this = root.owner
						# 				}
						# 				root = {
						# 					is_previous_owner = prev
						# 				}
						# 			}
						# 		}
						# 	}
						# }

						limit = {
							owner = {
								NOT = {
									is_target_in_variable_list = {
										name = my_land
										target = prev
									}
								}
							}
						}
						show_animated_text = ae_we
						root.owner = {
							add_aggressive_expansion = 0.32
							add_war_exhaustion = 0.1
						}
					}
				}
			}
		}
	}
}

province_siege.2 = {
	type = country_event
	hidden = yes

	trigger = {
		OR = {
			tag = PRY
			tag = SEL
			tag = TRE
			tag = MAC
			tag = EGY
		}
	}

	immediate = {
		every_owned_province = {
			ROOT = {
				add_to_variable_list = {
					name = my_land
					target = prev
				}
			}
		}
	}
}
