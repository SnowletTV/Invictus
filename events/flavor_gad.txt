namespace = flavor_gad

# Gadir Negotiating Truce with a Turdetanian - Hidden Start
flavor_gad.1 = {
	type = country_event
	hidden = yes
	
	fire_only_once = yes
	
	trigger = {
		current_date < 500.1.1
		primary_culture = turdetanian
		any_current_war = {
			is_war_leader = ROOT
			is_war_leader = c:GAD
		}
		OR = {
			p:1344 = {
				owner = c:GAD
				controller = ROOT
			}
			p:1348 = {
				owner = c:GAD
				controller = ROOT
			}
		}
	}
	
	immediate = {
		root = {
			save_scope_as = gadir_turdetanian_opponent
		}
		c:GAD = { trigger_event = flavor_gad.2 }
	}
}

# Gadir Negotiating Truce with a Turdetanian - Start
flavor_gad.2 = {
	type = country_event
	title = "flavor_gad.2.t"
	desc = "flavor_gad.2.desc"
	picture = romans_marching
	left_portrait = current_ruler
	right_portrait = scope:gadir_turdetanian_opponent.current_ruler
	
	option = {
		name = flavor_gad.2.a
		add_treasury = -150
		scope:gadir_turdetanian_opponent = { trigger_event = { id = flavor_gad.3 days = 10 } }
		ai_chance = { factor = 90 }
	}
	
	option = {
		name = flavor_gad.2.b
		ai_chance = { factor = 10 }
	}
}

# A Proposal from Gadir
flavor_gad.3 = {
	type = country_event
	picture = romans_marching
    title = flavor_gad.3.t
    desc = flavor_gad.3.desc
	left_portrait = current_ruler
	right_portrait = c:GAD
	
    option = {
        name = flavor_gad.3.a
		custom_tooltip = flavor_gad.3.tooltip
		add_treasury = 150
		hidden_effect = {
			if = {
				limit = {
					war_with = c:GAD
					any_current_war = {
						is_war_leader = ROOT
						is_war_leader = c:GAD
					}
				}
				every_current_war = {
					limit = {
						is_war_leader = ROOT
						is_war_leader = c:GAD
					}
					force_white_peace = THIS
				}
			}
			c:GAD = {
				add_truce = {
					target = scope:gadir_turdetanian_opponent
					duration = 1825
				}
			}
			scope:gadir_turdetanian_opponent = {
				add_truce = {
					target = c:GAD
					duration = 1825
				}
			}
		}
		c:GAD = { trigger_event = flavor_gad.4 }
		ai_chance = { factor = 90 }
    }
	
	option = {
        name = flavor_gad.3.b
		c:GAD = { trigger_event = flavor_gad.5 }
		ai_chance = { factor = 10 }
    }
}
