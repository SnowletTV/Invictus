flavor_mat.20 = { 							# The Apotheosis of Aletes - Prestige Events
	type = country_event
	hidden = yes

	immediate = {
		if = {
			limit = { # Check if Aletes is alive
				has_variable = aletes_apotheosis_progress
				var:aletes_apotheosis_progress < 20
				char:553 = {
					is_alive = yes
					employer = root
				}
			}
			random_list = { # Prestige Events
				2 = { # Prestige Opportunity - Sponsoring Games
					trigger_event = {
						id = flavor_mat.24
					}
				}
				2 = { # Prestige Opportunity - Waves of Charity
					trigger = {
						char:553 = {
							wealth >= 100
						}
					}
					trigger_event = {
						id = flavor_mat.27
					}
				}
				2 = { # Prestige Opportunity - Recording a Life (Tudhaliya Easter Egg)
					trigger = {
						char:553 = {
							wealth >= 100
						}
						NOT = {
							any_character = {
								has_variable = tudhaliya_flag
							}
						}
					}
					trigger_event = {
						id = flavor_mat.28
					}
				}
				2 = { # Prestige Opportunity - Asking for an Office
					trigger = {
						char:553 = {
							has_any_office = no
						}
						NOT = { has_variable = scheme_8_cooldown }
					}
					trigger_event = {
						id = flavor_mat.30
					}
				}
				2 = { # Prestige Opportunity - A Political Marriage
					trigger = {
						char:553 = {
							is_married = no
						}
					}
					trigger_event = {
						id = flavor_mat.27
					}
				}
				2 = { # Prestige Opportunity - Buying Holdings
					trigger = {
						
					}
					trigger_event = {
						id = flavor_mat.27
					}
				}
				2 = { # Prestige Opportunity - Outshining the Ruler
					trigger_event = {
						id = flavor_mat.27
					}
				}
				2 = { # Prestige Opportunity - Visiting Another City
					trigger = {
						
					}
					trigger_event = {
						id = flavor_mat.27
					}
				}
				2 = { # Prestige Opportunity - Founding a Temple for a Local Deity
					trigger_event = {
						id = flavor_mat.27
					}
				}
				2 = { # Prestige Opportunity - Befriending the High Priest
					trigger = {
						
					}
					trigger_event = {
						id = flavor_mat.27
					}
				}
				2 = { # Prestige Opportunity - Patronize Head of Great Family Campaing (the one of his wife)
					trigger = {
						
					}
					trigger_event = {
						id = flavor_mat.27
					}
				}
			}
			change_variable = {
				name = aletes_apotheosis_progress
				add = 1
			}
			trigger_event = { # Trigger next event
				id = flavor_mat.20
				days = { 730 1095 } # 2 - 3 years
			}
		}
		else_if = { # Aletes is too full of himself... Be prepared! (Not Mastia)
			limit = {
				NOT = { tag = MAT }
				has_variable = aletes_apotheosis_progress
				var:aletes_apotheosis_progress >= 20
				char:553 = {
					is_alive = yes
					is_ruler = no
					employer = root
				}
			}
			trigger_event = {
				id = flavor_mat.50
			}
		}
		else_if = { # Aletes is too full of himself... Be prepared! (Mastia)
			limit = {
				tag = MAT
				has_variable = aletes_apotheosis_progress
				var:aletes_apotheosis_progress >= 20
				char:553 = {
					is_alive = yes
					is_ruler = no
					employer = root
				}
			}
			trigger_event = {
				id = flavor_mat.51
			}
		}
		else = {
			remove_variable = aletes_apotheosis_progress
		}
	}
}
