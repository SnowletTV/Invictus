flavor_maurya.1 = {		#Chandragupta kills himself and Bindusara becomes ruler
    type = country_event
    title = flavor_maurya_1_title
    desc = flavor_maurya_1_desc
    picture = interesting_histories_india_city
    fire_only_once = yes

    left_portrait = char:22
    right_portrait = char:22

    trigger = {
    	tag = MRY
    	current_date > 458.3.5
    	char:22 = {
    		is_alive = yes
    	}
    }

    immediate = {	
    	hidden_effect = {
	    	#Ashoka or Bindusara becomes ruler if Bindusara isn't already the primary heir
            if = {
                limit = { 
                    char:65 = {
                        is_primary_heir = no
                        is_alive = yes
                    }
                }
                char:22 = {
                    death = {
                        death_reason = death_ritual_sallekhana
                    }
                }
                char:65 = {
                    c:MRY= {
                        set_as_ruler=char:65
                    }
                }                    		
            }
            else_if = {
                limit = { 
                    global_var:acting_as_ashoka = {
                        is_alive = no
                    }
                }
                random_character = {
                    limit = {
                        is_primary_heir = yes
                    }
                    set_global_variable = {
                        name = acting_as_ashoka
                        value = this
                    }
                    every_country = {
                        limit = {
                            has_variable = ashoka_but_for_loc_local
                        }
                        set_variable = {
                            name = ashoka_but_for_loc_local
                            value = global_var:acting_as_ashoka
                        }
                    }
                }
                char:22 = {
                    death = {
                        death_reason = death_ritual_sallekhana
                    }
                }
                c:MRY= {
                    set_as_ruler = global_var:acting_as_ashoka
                }
                set_variable = bindusara_is_a_shithead  #bindusara died early somehow 
            }
	    	else = {
                limit = {
                    char:65 = {
                        is_alive = no
                    }                    
                }
                char:22 = {
                    death = {
                        death_reason = death_ritual_sallekhana
                    }
                }
                c:MRY= {
                    set_as_ruler = global_var:acting_as_ashoka
                }
                set_variable = bindusara_is_a_shithead  #bindusara died early somehow            
            }
		}    	
    }

    option = {    #May he rest in peace.  
        name = flavor_maurya_1.a
        show_as_tooltip = {
            char:22 = {
                death = {
                    death_reason = death_ritual_sallekhana
                }
            }
        }
        if = {
            limit = { has_variable = bindusara_is_a_shithead }
            custom_tooltip = flavor_maurya_5_tooltip    #Start mission tree
        }        
    }
}

flavor_maurya.3 = {		#Convert Changraputa, a few charcters, and the capital to Jain.

   	right_portrait = global_var:jain_priest_var

   	trigger = {

						count = 25

                            is_ruler = no

                        }
                    }
                }
                c:MRY= {
                    set_as_ruler = global_var:acting_as_ashoka
                }            
            }

		hidden_effect = {
            if = {
                limit = {
                    any_character = {
                        is_adult = yes
                        is_primary_heir = no
                        is_ruler = no
                        is_governor = yes
                    }
                }
                ordered_character = {
                    limit = {
                        is_adult = yes
                        is_primary_heir = no
                        is_ruler = no
                        is_governor = yes                        
                    }
                    check_range_bounds = no
                    order_by = power_base
                    save_scope_as = rebel_scum_leader
                    max = 1
                }
            }
            else = {
                ordered_character = {
                    limit = {
                        is_ruler = no
                        is_adult = yes
                        is_primary_heir = no
                        is_minor_character = yes
                    }
                    max = 1
                    check_range_bounds = no
                    order_by = power_base
                    save_scope_as = rebel_scum_leader
               }
            }
            ordered_governorships = {
                limit = {
                    NOT = {
                        any_governorship_state = {
                            is_capital_state = yes
                        }
                    }
                }
                max = 1
                order_by = {
                    every_governorship_state = {
                        every_state_province = {
                            add = total_population
                        }
                    }
                }
                save_scope_as = rebel_scum_base
            }
            if = {
                limit = {
                    scope:rebel_scum_leader = {
                        is_governor = no
                    }
                }
                scope:rebel_scum_base = {
                    random_governorship_state = {
                        random_state_province = {
                            set_as_governor = scope:rebel_scum_leader
                        }
                    }
                }

            start_civil_war = scope:rebel_scum_leader

                save_scope_as = civil_war_cap
            }
            scope:civil_war_cap.owner = {

                change_country_color = bharat_green_color
                change_country_flag = BHARAT_REVOLT_FLAG
                change_government = imperium
                save_scope_as = bharat_revolt_country
            }
            if = {
                limit = { is_ai = yes }
                #encourage AI expansion into India
                region:aparanta_region = { formable_region_province_effect = yes }
				region:vindhyaprstha_region = { formable_region_province_effect = yes }
				region:avanti_region = { formable_region_province_effect = yes }
				region:maru_region = { formable_region_province_effect = yes }
				region:gandhara_region = { formable_region_province_effect = yes }
                region:madhyadesa_region = { formable_region_province_effect = yes }
                region:pracya_region = { formable_region_province_effect = yes }
                #effect players get in the missions
                global_var:acting_as_ashoka = {
                    if = {
                        limit = { has_trait = cruel }
                        remove_trait = cruel
                        force_add_trait = just								
                    }
                    add_character_modifier = {
                        name = ashoka_the_great
                        duration = -1	
                    }
                }
            }
        }		  	
    }

    option = {    #Ashoka will bring greatness to Maurya. (Start a small civil war)

                    name = ashoka_but_for_loc_local
                    value = global_var:acting_as_ashoka
                }
            }
        }
    }
}

flavor_maurya.6 = {		#Ashoka is Dead!
    type = country_event
    title = flavor_maurya_6_title
    desc = {
        first_valid = {
            triggered_desc = {  ##Collapse Description
                trigger = { NOT = { has_global_variable = mauryan_collapse_prevented } }  
                desc = "flavor_maurya_6.desc_1"
            }
            triggered_desc = {  ##Prevented Collapse Description
                trigger = { has_global_variable = mauryan_collapse_prevented }    
                desc = "flavor_maurya_6.desc_2"
            }           
        }
    }    
    picture = interesting_histories_india_city
    fire_only_once = yes

    left_portrait = global_var:acting_as_ashoka
    right_portrait = global_var:acting_as_ashoka

    trigger = {
        NOT = { has_global_variable = mauryan_collapse_game_rule }
        NOT = { has_global_variable = maurya_collapse_happened }
        tag = MRY
        OR = {
            global_var:acting_as_ashoka = {
                is_alive = no
            }
            AND = {
                is_ai = yes
                current_date > 550.1.1 
            } 
        }
    }

    immediate = {
        set_global_variable = maurya_collapse_happened
        every_country = {
            limit = { 
                is_ai = no 
                NOT = { tag = MRY }
                NOT = { tag = BHA }
            }
            trigger_event = {
                id = flavor_maurya.16
            }
        }	
        if = {
            limit = { 
                NOT = { has_global_variable = mauryan_collapse_prevented } 
            }
            add_stability = -20
            set_country_heritage = mauryan_abandoned_heritage
            hidden_effect = {
                random_neighbour_country = {
                    limit = {
                        NOT = { tag = SEL }
                        OR = {
                            country_culture_group = aryan
                            country_culture_group = indian
                            country_culture_group = pracyan
                            country_culture_group = tibetan
                            country_culture_group = persia
                        }
                        NOT = {
                            is_subject_of = ROOT
                            has_truce_with = ROOT
                            war_with = ROOT
                        }
                        num_of_cities > 20
                        has_civil_war = no
                        war = no
                        has_war_exhaustion < 13
                        is_subject = no	
                        is_ai = yes			
                    }
                    save_scope_as = mry_attack_target_country
                    set_variable = {
                        name = is_not_allowed_to_go_broke
                        days = 9125
                    }
                    set_variable = {
                        name = is_ignoring_ae
                        days = 3650
                    }
                }
                random_list = {
                    60 = {
                        if = {
                            limit = {
                                c:MRY = {
                                    is_ai = yes
                                }
                            }
                            scope:mry_attack_target_country = {
                                declare_war_with_wargoal = {
                                    war_goal = maurya_decline_wargoal
                                    target = c:MRY
                                }
                                add_treasury = 500
                                add_manpower = 10                        
                                create_unit = {
                                    navy = no
                                    location = scope:mry_attack_target_country.capital_scope
                                    while = {
                                        count = 4
                                        add_subunit = light_infantry
                                    }
                                    while = {
                                        count = 5
                                        add_subunit = warelephant
                                    }
                                    while = {
                                        count = 5
                                        add_subunit = light_cavalry
                                    }
                                    while = {
                                        count = 2
                                        add_subunit = supply_train
                                    }                                              
                                    add_food = 200
                                    add_morale = 2
                                }            
                            }
                        }
                    }
                    40 = {}
                } 		    		
            }
            random_list = {
                60 = {
                    trigger = { is_ai = yes }
                    trigger_event = {
                        id = flavor_maurya.7
                    }
                }
                40 = {}
            }
            random_list = {
                60 = {
                    trigger = { is_ai = yes }
                    trigger_event = {
                        id = flavor_maurya.8
                        days = 735
                    }
                }
                40 = {}
            }
            #longer for ai
            if = {
                limit = { is_ai = yes }
                trigger_event = {
                    id = flavor_maurya.9
                    days = 730
                }            
            }
            else = {
                limit = { is_ai = no }
                trigger_event = {
                    id = flavor_maurya.9
                    days = 365
                }
            }
            add_country_modifier = {    #Clears after collapse missions
                name = mauryan_collapse
                duration = -1
            }
        }
        else = {    #Fire the Indo-Greek Event even if the missions are completed.
            limit = { has_global_variable = mauryan_collapse_prevented }
            trigger_event = {
                id = flavor_maurya.12
                days = 720
            }        
        }        		    	
    }

    option = {    #Trouble approches...  
        name = flavor_maurya_6.a       
    }
}
