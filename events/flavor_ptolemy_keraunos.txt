				death_reason = death_internal_assassination
				killer = scope:assassinator
			}
		}
		scope:assassinator = {
			if = {
				limit = {
					has_variable = number_of_murders
				}
				change_variable = {
					name = number_of_murders
					add = 1
				}
			}
			else = {
				set_variable = {
					name = number_of_murders
					value = 1
				}
			}
		}
	}
	
}
