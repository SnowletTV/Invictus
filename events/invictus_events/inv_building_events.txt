# Crumbling Fortress
inv_building_events.1 = {
    type = country_event
    title = inv_building_events.1.t
    desc = inv_building_events.1.desc
    picture = greek_states_path_6_event
    goto_location = scope:building_province
    left_portrait = current_ruler

    trigger = {
        NOT = { has_variable = military_building_cooldown }
    	any_owned_province = {
    		num_of_fortress_building > 1
    		province_income < 3
    	}
    }

    immediate = {
        set_variable = {
            name = military_building_cooldown
            days = very_long_cooldown
        }
        random_owned_province = {
            limit = {
                num_of_fortress_building > 1
                province_income < 3
            }
            save_scope_as = building_province
        }
    }

    option = {
        name = inv_building_events.1.a
        scope:building_province = {
            add_province_modifier = {
                name = local_unrest_generic_unrest
                duration = 730
            }
        }
    }
    option = {
        name = inv_building_events.1.b
        add_treasury = {
            subtract = root.has_monthly_income
				multiply = 3
        }
    }
}

    goto_location = scope:building_province
    left_portrait = current_ruler

        NOT = { has_variable = military_building_cooldown }

            num_of_fortress_building > 1

    }

    immediate = {
        set_variable = {
            name = military_building_cooldown

        random_owned_province = {
            limit = {
                num_of_fortress_building > 1
                governor_policy = borderlands
            }
            save_scope_as = building_province
        }
        scope:building_province = {
            if = {
                limit = {
                    exists = holding_owner
                }
                save_scope_as = fort_guy
            }
            else_if = {
                limit = {
                    exists = governor
                    NOT = { governor = root.current_ruler }
                }

            else = {
                root = {
                    create_character = {
                        female = no
                        good_finesse_trait_effect = yes

            }
        }
    }

    option = {
        name = inv_building_events.3.a
        add_stability = 3
        current_ruler = {
            add_popularity = 5
        }
    }
    option = {
        name = inv_building_events.3.b
        add_treasury = {
            subtract = root.has_monthly_income
			multiply = 6
        }
        capital_scope = {
            add_province_modifier = {
                name = fort_guy_assisting_local_fort
                duration = 3650
            }
        }
    }
}

    goto_location = scope:burning_fort_province
    left_portrait = current_ruler

    trigger = {
        NOT = { has_variable = military_building_cooldown }
        any_owned_province = {
            num_of_fortress_building > 1
            OR = {
                AND = {
                    exists = governor_or_ruler
                    governor_or_ruler = { has_trait = pyromaniac }
                }
                AND = {
                    exists = holding_owner
                    holding_owner = { has_trait = pyromaniac }
                }
            }
        }
    }

    immediate = {
        set_variable = {
            name = military_building_cooldown
            days = long_cooldown
        }
        random_owned_province = {
            limit = {
                OR = {
                    AND = {
                        exists = governor_or_ruler
                        governor_or_ruler = { has_trait = pyromaniac }
                    }
                    AND = {
                        exists = holding_owner
                        holding_owner = { has_trait = pyromaniac }
                    }
                }
            }
            save_scope_as = burning_fort_province
        }
        scope:burning_fort_province = {
            if = {
                limit = {
                    exists = holding_owner
                    holding_owner = { has_trait = pyromaniac }
                }
                holding_owner = {
                    save_scope_as = pyromaniac_administrator
                }
            }
            else_if = {
                limit = {
                    exists = governor_or_ruler
                    governor_or_ruler = { has_trait = pyromaniac }
                }
                holding_owner = {
                    save_scope_as = pyromaniac_administrator
                }
            }
        }
    }
    option = {
        name = inv_building_events.4.a
        current_ruler = {
            add_popularity = -5
        }
    }
    option = {
        name = inv_building_events.4.b

            add_gold = -100
            add_loyalty = loyalty_publicly_slandered
        }
        current_ruler = {
            add_popularity = 5
        }
    }
}
