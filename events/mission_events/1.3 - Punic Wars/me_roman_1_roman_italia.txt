me_roman_1_roman_italia.1 = {							# 1 - Adriatic Adventures - Response
    type = country_event
    title = me_roman_1_roman_italia.39.t
    desc = {
		first_valid = {
			triggered_desc = {		# Success offer
				trigger = {
					has_variable = rm_offer_alliance_var
					scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
				}
				desc = "me_roman_1_roman_italia.1.desc_1"
			}
			triggered_desc = {		# Success demand
				trigger = {
					NOT = { has_variable = rm_offer_alliance_var }
					scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
				}
				desc = "me_roman_1_roman_italia.1.desc_2"
			}
			triggered_desc = {		# Inconclusive
				trigger = {
					has_variable = rm_offer_alliance_var
					NOT = {
						scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
					}
				}
				desc = "me_roman_1_roman_italia.1.desc_3" 
			}
			triggered_desc = {		# Inconclusive
				trigger = {
					NOR = { 
						has_variable = rm_offer_alliance_var
						scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
					}
				}
				desc = "me_roman_1_roman_italia.1.desc_4" 
			}
		}
	}
    picture = council_country
    
    left_portrait = current_ruler
	right_portrait = scope:rm_offer_scope.current_ruler

	fire_only_once = yes

    trigger = {
		has_land = yes
    }

    immediate = {
		if = {
			limit = {
				has_variable = rm_offer_alliance_var
				scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
			}
			show_as_tooltip = {
				make_subject = {
					target = scope:rm_offer_scope
					type = tributary
				}
			}
		}
		else_if = {
			limit = {
				NOT = { has_variable = rm_offer_alliance_var }
				scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
			}
			show_as_tooltip = {
				make_subject = {
					target = scope:rm_offer_scope
					type = client_state
				}
			}
		}
		else = {
			hidden_effect = {
				scope:rm_offer_scope = {
					every_owned_province = {
						add_claim = c:ROM
					}
				}
			}
			custom_tooltip = me_roman_1_roman_italia.1.tt
		}
    }

    option = {      
        name = me_roman_1_roman_italia.1.a
		trigger = { 
			scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
		}
    }
	
    option = {    
        name = me_roman_1_roman_italia.1.b
		trigger = {
			NOT = {
				scope:rm_offer_scope = { has_variable = rm_offer_accepted_var }
			}
		}
	}
}
