        tag = ODR
        is_subject_of = c:TRE
        NOT = { has_variable = TRE_ODR_revolts_stopped } # Don't trigger when future revolts have been stopped
        NOT = { has_variable = TRE_ODR_peace_timer } # Don't trigger too soon after Odrysia has been forced into peace (via the event)

                has_variable = ODR_uprising_diadochi_war
            }
        }
    }

    weight_multiplier = {
        modifier = {
           factor = 5
           opinion = { target = c:TRE value < 50 }
        }
        modifier = {
           factor = 10
           opinion = { target = c:TRE value <= 0 }
        }
        modifier = {
           factor = 5
           has_subject_loyalty <= 50
        }
        modifier = {
           factor = 10
           has_subject_loyalty <= 25
        }
        modifier = {
            factor = 0.25
            c:TRE = {
                NOR = {
                    war_with = MAC
                    war_with = PRY
                    war_with = SEL
                    war_with = EGY
                }
            }
        }
    }

    immediate = {
        random_country = {
            limit = {
                OR = {
                    tag = MAC
                    tag = PRY
                    tag = SEL
                    tag = EGY

                war_with = TRE
            }
            save_scope_as = ODR_thracian_enemy
        }
        set_variable = {
            name = TRE_ODR_rebellion
            days = 1095
        }
    }

    option = { # Good idea
        name = me_thrace_2_paper_kingdom.3.a
        ai_chance = {
            factor = 80
            modifier = {
                has_variable = TRE_negotiations
                factor = 0.5
            }
            modifier = {
                has_variable = TRE_unsavory_compromise
                factor = 0.25
            }
        }
        c:TRE = {
            if = {
                limit = {
                    NOT = {
                        has_variable = odrysia_revolted
                    }
                }

                }
            }

                    add = 1
                    name = "odrysia_revolted"
                }
            }

        }
        hidden_effect = {
            every_current_war = {
                limit = {
                    any_war_participant = { tag = TRE }
                }

            }
        }

            war_goal = independence_wargoal

            target = TRE
        }
        hidden_effect = {
            every_current_war = {
                limit = {
                    OR = {
                        AND = {
                            any_war_attacker = { tag = TRE }
                            any_war_defender = {
                                OR = {
                                    tag = MAC
                                    tag = PRY
                                    tag = SEL
                                    tag = EGY
                                }
                            }
                        }
                        AND = {
                            any_war_attacker = {
                                OR = {
                                    tag = MAC
                                    tag = PRY
                                    tag = SEL
                                    tag = EGY
                                }
                            }
                            any_war_defender = { tag = TRE }
                        }
                    }
                }
                set_variable = ODR_uprising_diadochi_war
            }
            every_current_war = {
                limit = {
                    any_war_attacker = {
                        tag = ODR
                    }
                    any_war_defender = {
                        tag = TRE
                    }
                }
                set_variable = ODR_uprising_war
            }
        }
    }

    option = { # Nah, we will stay loyal
        name = me_thrace_2_paper_kingdom.3.b
        ai_chance = {
            factor = 20
            modifier = {
                c:TRE = {
                    OR = {

                            has_variable = odrysia_revolted
                            var:odrysia_revolted < 1      
                        }
                    }
                }
                factor = 0
            }
        }
    }
