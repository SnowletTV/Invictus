me_albion.2 = { #council of druids
	type = country_event
	title = me_albion_2_title
	desc = me_albion_2_desc
	picture = celtic_chiefs
	left_portrait = current_ruler

	immediate = {
		set_variable = {
			name = political_gain
			value = 0
		}
		if = {
			limit = {not = {has_law = nomadic_law_tribal  } }
			change_variable = {
				name = political_gain
				add = 25
			}
		}
		if = {
			limit = {not = {has_law = tribal_centralized_laws_1  } }
			change_variable = {
				name = political_gain
				add = 25
			 }
		}
		if = {
			limit = {not = {has_law = super_centralized_law_one  } }
			change_variable = {
				name = political_gain
				add = 25
			 }
		}
	}

	option = { #We should listen all each other
		
	name = me_albion.2.a
		add_political_influence = -50
		add_country_modifier = {
			name = albion_mod_4
			duration = 10950
		}
	}

	option = { #Expule the druids from the debate
	name = me_albion.2.b
		add_country_modifier = {
			name = albion_mod_2
			duration = 10950
		}
	}

		option = { #listen them
			
		name = me_albion.2.c
		if = {
			limit = { var:political_gain >= 1}
		add_political_influence = var:political_gain
		}
		add_centralization = -50
			add_country_modifier = {
				name = albion_mod_3
				duration = 10950
			}
		if = {
			limit = {not = {has_law = nomadic_law_tribal  } }
			change_law = nomadic_law_tribal
		}
		if = {
			limit = {not = {has_law = tribal_centralized_laws_1  } }
			change_law = tribal_centralized_laws_1
		}
		if = {
			limit = {not = {has_law = super_centralized_law_one  } }
			change_law = super_centralized_law_one
		}
	}
	after = {
		remove_variable = political_gain
	}
}
