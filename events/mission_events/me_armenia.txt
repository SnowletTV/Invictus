me_armenia.1000 = { # final event
    type = country_event
    title = me_armenia.1000.title
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = { var:persian_culture_points >= 5 }
                desc = me_armenia.1000.desc_a
            }
            triggered_desc = {
                trigger = { var:hellenic_culture_points >= 5 }
                desc = me_armenia.1000.desc_b
            }
            triggered_desc = {
                trigger = { var:armenian_culture_points >= 5 }
                desc = me_armenia.1000.desc_c
            }
            triggered_desc = {
                trigger = { always = yes }
                desc = me_armenia.1000.desc_d
            }
        }
    }
    picture = war_council
    left_portrait = current_ruler

    option = {
        trigger = { var:persian_culture_points >= 5 }
        name = "me_armenia.1000.a"
        exclusive = yes

        add_3_free_province_investments = yes
        add_country_modifier = {
            name = persia_culture_path_end_bonus
            duration = -1
        }
    }

    option = {
        trigger = { var:hellenic_culture_points >= 5 }
        name = "me_armenia.1000.b"
        exclusive = yes

        add_3_free_province_investments = yes
        add_country_modifier = {
            name = hellenic_culture_path_end_bonus
            duration = -1
        }
    }

    option = {
        trigger = { var:armenian_culture_points >= 5 }
        name = "me_armenia.1000.c"
        exclusive = yes

        add_3_free_province_investments = yes
        add_country_modifier = {
            name = armenian_culture_path_end_bonus
            duration = -1
        }
    }

    option = {
        trigger = {
            NOR = {
                var:armenian_culture_points >= 5
                var:hellenic_culture_points >= 5
                var:persian_culture_points >= 5
            }
        }
        name = "me_armenia.1000.d"

        add_3_free_province_investments = yes
        add_country_modifier = {
            name = mixed_influence_end_bonus
            duration = -1
        }
    }
}
