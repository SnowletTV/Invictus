me_arvernia.4 = {
    type = country_event

    title = me_arvernia_4_title
    desc = me_arvernia_4_desc
    picture = great_library

    option = {
        name = "me_arvernia_4.a"
        custom_tooltip = we_will_release_arvernia_volcaea_tt
        add_political_influence = -75
        add_centralization = -10
        hidden_effect = {
                p:2260 = {
                    create_country = {
                        name = {
                            name = "VOLCAEA_NAME"
                            adjective = "VOLCAEA_ADJECTIVE"
                        }
                        change_country_flag = VOLCAE_ARV_FLAG
                        change_country_color = arverni_subject_color
                        if = {
                            limit = {
                                NOT = {
                                    primary_culture = volcae
                                }
                            }
                            set_primary_culture_cleanup_effect = {
                                NEW_PRIMARY_CULTURE = volcae
                                MAKE_OLD_CULTURE_INTEGRATED = yes
                            }
                        }
                        set_country_religion = druidism
                        change_government = tribal_federation
                        add_treasury = 100
                        add_stability = 20
                        save_scope_as = arvernian_volcae_scope
                    }
                }
                every_owned_province = {
                    limit = {
                        OR = {
                            is_in_area = narbonensis_centralis_area
                            is_in_area = narbonensis_orientalis_area
                        }
                    }
                    set_owned_by = scope:arvernian_volcae_scope
                }
                make_subject = { 
                    target = scope:arvernian_volcae_scope
                    type = feudatory
                }
                reverse_add_opinion = {
                    modifier = opinion_appreciate_protection
                    target = scope:arvernian_volcae_scope
                }
                scope:arvernian_volcae_scope = {
                    add_country_modifier = {
                        name = arvernia_volcae_loyalty_mod
                        duration = 7300
                    }
                }
            }
        custom_tooltip = me_arvernia_4_tooltip
    }

    option = {
        name = "me_arvernia_4.b"
        current_ruler = {
            force_add_trait = conqueror
        }
        custom_tooltip = me_arvernia_4_tooltip
    }
}
