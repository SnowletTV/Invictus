me_bharat.2 = {
	#Civil War! ##Hopefully this works##
	type = country_event
	title = me_bharat_2_title
	desc = me_bharat_2_desc
	picture = revolt
	left_portrait = current_ruler
	right_portrait = scope:preston_garvey
	goto_location = capital_scope
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					any_character = {
						is_adult = yes
						is_primary_heir = no
						is_ruler = no
						is_governor = yes
					}
				}
				ordered_character = {
					limit = {
						is_adult = yes
						is_primary_heir = no
						is_ruler = no
						is_governor = yes
					}
					check_range_bounds = no
					order_by = power_base
					save_scope_as = preston_garvey
					max = 1
				}
			}
			else = {
				ordered_character = {
					limit = {
						is_ruler = no
						is_adult = yes
						is_primary_heir = no
						is_minor_character = yes
					}
					max = 1
					check_range_bounds = no
					order_by = power_base
					save_scope_as = preston_garvey
				}
			}
			ordered_governorships = {
				limit = {
					NOT = {
						any_governorship_state = {
							is_capital_state = yes
						}
					}
				}
				max = 1
				order_by = {
					every_governorship_state = {
						every_state_province = {
							add = total_population
						}
					}
				}
				save_scope_as = rebel_scum_capital
			}
			if = {
				limit = {
					scope:preston_garvey = {
						is_governor = no
					}
				}
				scope:rebel_scum_capital = {
					random_governorship_state = {
						random_state_province = {
							set_as_governor = scope:preston_garvey
						}
					}
				}
			}
			start_civil_war = scope:preston_garvey
			scope:preston_garvey.location = {
				save_scope_as = civil_war_capital
			}
			scope:civil_war_capital.owner = {
				change_country_name = "indian_revolter_name"
				change_country_adjective = "indian_revolter_adjective"
				change_country_color = roman_country_color
				change_country_flag = BHARAT_REVOLT_FLAG
				change_government = imperium
				save_scope_as = bharat_revolt_country
			}
			ordered_governorships = {
				limit = {
					NOT = {
						any_governorship_state = {
							is_capital_state = yes
						}
					}
					trigger_if = {
						limit = {
							exists = governor
						}
						governor = {
							OR = {
								from_ruler_family = no
								loyalty < 60
							}
						}
					}
				}
				max = {
					ROOT = {
						every_governorships = {
							add = 1
						}
					}
					divide = 3
					min = 1
				}
				order_by = {
					every_governorship_state = {
						every_state_province = {
							limit = {
								dominant_province_culture = root.culture
							}
							add = total_population
						}
					}
				}
				every_governorship_state = {
					every_state_province = {
						set_owned_by = scope:bharat_revolt_country
						add_claim = root
					}
				}
			}
			every_subject = {
				limit = {
					OR = {
						is_subject_type = feudatory
						is_subject_type = satrapy
						is_subject_type = client_state
					}
					has_subject_loyalty < 70
				}
				switch = {
					#Must remain the same type of subject
					trigger = is_subject_type
					feudatory = {
						random_list = {
							50 = {
								ROOT = {
									release_subject = THIS
								}
								save_scope_as = released_subject
								scope:bharat_revolt_country = {
									make_subject = {
										target = scope:released_subject
										type = feudatory
									}
								}
							}
							50 = {
							}
						}
					}
					client_state = {
						random_list = {
							50 = {
								ROOT = {
									release_subject = THIS
								}
								save_scope_as = released_client_subject
								scope:bharat_revolt_country = {
									make_subject = {
										target = scope:released_client_subject
										type = client_state
									}
								}
							}
							50 = {
							}
						}
					}
				}
			}
			every_owned_province = {
				add_claim = scope:bharat_revolt_country
			}
			every_character = {
				limit = {
					loyalty < 60
				}
				move_country = scope:bharat_revolt_country
			}
		}
	}
	option = {
		name = me_bharat_2.a		#We're doomed...
		show_as_tooltip = {
			start_civil_war = scope:preston_garvey
		}
		add_stability = -20
	}
}

		current_ruler = {
			add_trait = blood_of_bharata
			add_popularity = 20
		}
		show_as_tooltip = {
			if = {
				limit = {
					NOT = {
						primary_culture = kuru
					}
				}
				set_primary_culture_cleanup_effect = {
					NEW_PRIMARY_CULTURE = kuru
					MAKE_OLD_CULTURE_INTEGRATED = yes
				}
			}
		}
		hidden_effect = {
			every_character = {
				limit = {
					OR = {
						from_ruler_family = yes
						is_minor_character = no
					}
				}
				add_trait = blood_of_bharata
				set_culture = kuru
			}
			every_owned_province = {
				limit = {
					OR = {
						is_in_region = gandhara_region
						is_in_region = maru_region
						is_in_region = madhyadesa_region
					}
				}
				ordered_pops_in_province = {
					order_by = pop_hapiness
					limit = {
						OR = {
							pop_type = citizen
							pop_type = nobles
							pop_type = freemen
						}
						NOT = {
							pop_culture = kuru
						}
					}
					set_pop_culture = kuru
					max = 250
					check_range_bounds = no
				}
			}
			ordered_owned_province = {
				order_by = total_population
				max = 16
				check_range_bounds = no
				ordered_pops_in_province = {
					order_by = pop_hapiness
					limit = {
						OR = {
							pop_type = citizen
							pop_type = nobles
						}
						NOT = {
							pop_culture = kuru
						}
					}
					set_pop_culture = kuru
					max = 12
					check_range_bounds = no
				}
			}
			if = {
				limit = {
					NOT = {
						primary_culture = kuru
					}
				}
				set_primary_culture_cleanup_effect = {
					NEW_PRIMARY_CULTURE = kuru
					MAKE_OLD_CULTURE_INTEGRATED = yes
				}
			}
		}
	}
	option = {
		name = me_bharat_3.b		##We don't need to adheer to tradition!
		current_ruler = {
			add_charisma = 3
			add_martial = 2
			add_popularity = 20
		}
		add_country_modifier = {
			name = indian_dynasty
			duration = -1
		}
		add_stability = 15
	}
}
