namespace = bithynia_battle_counting

bithynia_battle_counting.1 = {
    type = country_event
    hidden = yes
    trigger = {
        tag = THE
       # NOT = { has_variable = stop_counting_idiot_2 }
        scope:actor = { #This is the unit scope
            has_legion_trigger = yes
            commander = {
                has_triggered_character_modifier = commander_of_bithynia
            }
        }
    }
    immediate = {
        if = {
            limit = {
                NOR = {
                    has_variable = error_log_appeasing_var_4
                    has_variable = legion_battle_counter
                }
            }
            set_variable = error_log_appeasing_var_4
        }
        if = {
            limit = {
                NOT = { has_variable = legion_battle_counter }
            }
            set_variable = {
                name = legion_battle_counter
                value = 1
            }
        }
        else_if = {
            limit = { var:legion_battle_counter < 26 }
            change_variable = {
                name = legion_battle_counter
                add = 1
            }
        }        
    }     
}

me_bithynia.1 = { 
    type = country_event

    title = me_bithynia_1_title
    desc = me_bithynia_1_desc
    picture = interesting_histories_greek_port

    immediate = {
        ordered_owned_province = {
            limit = {
                has_city_status = yes
                is_capital = no
            }
            order_by = total_population
            max = 1
            save_scope_as = new_capital
        }
    }

    option = { 
        name = me_bithynia_1.a
        custom_tooltip = me_bithynia_1_tooltip
        if = {
            limit = {
                capital_scope = { has_city_status = yes } 
            }
            add_treasury = 100
            add_1_free_province_investments = yes
            capital_scope = {
                while = {
                    count = 4
                    define_pop = {
                        type = citizen
                        culture = root.culture
                        religion = root.religion
                    }
                }
            }
        }
        else = {
            limit = {
                capital_scope = { has_city_status = no }
            }
            set_capital = scope:new_capital
            add_1_free_province_investments = yes
            capital_scope = {
                while = {
                    count = 3
                    define_pop = {
                        type = citizen
                        culture = root.culture
                        religion = root.religion
                    }
                }
            }
        }    
    }
}

                    limit = {
                        has_trait = achaemenid

                        age > 15
                    }
                    if = {

                        divorce_character = spouse
                    }
                    save_scope_as = marriage_partner
                }
            }
            char:306 = { marry_character = scope:marriage_partner }
        }
        else_if = {
            limit = {
                c:HCP = { 
                    has_land = yes
                    any_character = {
                        has_trait = achaemenid
                        is_female = yes
                        age <= 40
                        age > 15                        
                    }
                }
            }
            c:HCP = {
                random_character = {
                    limit = {
                        has_trait = achaemenid
                        is_female = yes
                        age <= 40
                        age > 15
                    }
                    if = {
                        limit = { is_married = yes }
                        divorce_character = spouse
                    }
                    save_scope_as = marriage_partner
                }
            }
            current_ruler = {
                ordered_child = {
                    limit = {
                        age >= 16
                        is_male = yes
                        is_alive = yes
                    }
                    order_by = age
                    max = 1
                    if = {
                        limit = { is_married = yes }
                        divorce_character = spouse
                    }
                    marry_character = scope:marriage_partner
                }
            }
        }
        else_if = {
            limit = {
                #c:HCP = { has_land = no }
            }
            add_treasury = 100
            current_ruler = {
                add_popularity = 5
            }
