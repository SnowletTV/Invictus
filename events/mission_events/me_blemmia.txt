me_blemmia.3 = { # The Emerald Uprising
    type = country_event

    title = me_blemmia_3_title
    desc = me_blemmia_3_desc
    picture = revolt
    left_portrait = current_ruler
    right_portrait = p:590.state.governor_or_ruler
    right_portrait = c:EGY.current_ruler
    goto_location = p:590

    option = {
        name = "me_blemmia_3.a"
        custom_tooltip = me_blemmia_3_tooltip
        p:590.state.governor_or_ruler = {
            move_country = root
            set_home_country = root
            remove_character_modifier = foreign_citizen
        }
        p:583 = {
            add_city_smoke_effect = yes
            While = {
                count = 3
                random_pops_in_province = {
                    limit = {
                        NOT = { pop_culture = blemmyan }
                    }
                    kill_pop = yes
                }
            }
        }
        c:EGY = { 
            area:troglodytica_area = {
                add_provincial_claim_effect = yes
            }
            add_opinion = {
                modifier = troglodytica_uprising
                target = root
            }
        }
    }
}

me_blemmia.4 = { # Red Gold Prospecting
    type = country_event

    title = me_blemmia_4_title
    desc = me_blemmia_4_desc
    picture = quarry
    left_portrait = current_ruler
    goto_location = p:8088

    option = {
        name = "me_blemmia_4.a"
        add_treasury = -40
        add_manpower = -3
        random_list = {
            30 = {
                p:8088 = {
                    set_trade_goods = precious_metals
                    while = {
                        count = 3
                        define_pop = {
                            type = slaves
                            culture = root.culture
                            religion = root.religion
                        }
                    }
                    add_building_level = slave_mine_building
                    add_permanent_province_modifier = {
                        name = blemmyan_gold_mining
                        duration = -1
                    }
                }
            }
            30 = {
                create_character = {
                    add_finesse = 10
                    add_trait = crafty
                    add_trait = wise
                    religion = egyptian_pantheon
                    culture = blemmyan
                    female = no
                }
            }
            30 = {
                add_political_influence = -10
                current_ruler = {
                    add_gold = -15
                }
            }
        }
    }

    option = {
        name = "me_blemmia_4.b"
        add_treasury = 20
    }
}
