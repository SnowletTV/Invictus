    type = country_event
    title = cr_carthage_invasion_missions.1.t
    desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					p:3360.owner = {
						this = c:CYR
					}

				desc = cr_carthage_invasion_missions.1.desc
			}
			triggered_desc = {
				trigger = {
					always = yes
				}

		}
	}
    picture = religious_rite

		custom_tooltip = cr_carthage_invasion_missions.1.a_tt
		
		ai_chance = {
			factor = 75
		}
    }
	
	option = {      
        name = cr_carthage_invasion_missions.1.b
		custom_tooltip = cr_carthage_invasion_missions.1.b_tt
		trigger = {
			tag = CYR
			current_ruler = {
				NOR = {
					has_trait = honest
					has_trait = guileless
					has_trait = content
					has_trait = just
					has_trait = righteous
				}
			}
		}
		current_ruler = {
			if = {
				limit = {
					NOT = {
						has_trait = crafty
					}
				}
				add_trait = crafty
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = deceitful
					}
				}
				add_trait = deceitful				
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = shrewd
					}
				}
				add_trait = shrewd				
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = reckless
					}
				}
				add_trait = reckless				
			}
		}
		c:CAR = {
			set_variable = cr_philaeni_cheated_CYR
		}
		ai_chance = {
			factor = 1
			modifier = {
				current_ruler = {
					OR = {
						has_trait = ambitious
						has_trait = corrupt
						has_trait = crafty
						has_trait = deceitful
						has_trait = reckless
						has_trait = shrewd
						has_trait = aggressive
					}
				}
				factor = 50
			}
		}
    }
	
	
	option = {      
        name = cr_carthage_invasion_missions.1.b
		trigger = {
			NOT = { tag = CYR }
			p:3360.governor = {
				NOR = {
					has_trait = honest
					has_trait = guileless
					has_trait = content
					has_trait = just
					has_trait = righteous
				}
			}
		}
		p:3360.governor = {
			if = {
				limit = {
					NOT = {
						has_trait = crafty
					}
				}
				add_trait = crafty
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = deceitful
					}
				}
				add_trait = deceitful				
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = shrewd
					}
				}
				add_trait = shrewd				
			}
			else_if = {
				limit = {
					NOT = {
						has_trait = reckless
					}
				}
				add_trait = reckless				
			}
		}
		c:CAR = {
			set_variable = cr_philaeni_cheated_CYR
		}
		ai_chance = {
			factor = 1
			modifier = {
				current_ruler = {
					OR = {
						has_trait = ambitious
						has_trait = corrupt
						has_trait = crafty

						has_trait = reckless
						has_trait = shrewd

					}
				}
				factor = 50
			}
		}
    }
