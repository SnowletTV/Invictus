# Maintaining the Wonder during the Build
me_greek_5_pontus.46 = {
	type = country_event
	hidden = yes

	immediate = {
		if = {	# Building progresses
			limit = {
				var:taurican_temple_progress < 3 
				owns = scope:taurican_temple_territory_scope
			}
			random_list = {
				1 = {	# Building progresses without any event
					change_variable = {
						name = taurican_temple_quality
						add = 1
					}
				}
				2 = {	# Complication event - Sunken ship with materials
					trigger_event = {
						id = me_greek_5_pontus.47
					}
				}
				2 = {	# Complication event - Uninspired workers
					trigger_event = {
						id = me_greek_5_pontus.48
					}
				}
				2 = {	# Complication event - Theft and sabotage
					trigger_event = {
						id = me_greek_5_pontus.49
					}
				}
			}
			change_variable = {
				name = taurican_temple_progress
				add = 1
			}
			trigger_event = {
				id = me_greek_5_pontus.46
				days = { 150 300 }
			}
		}
		else_if = {	# Building finished
			limit = {
				var:taurican_temple_progress >= 3
				owns = scope:taurican_temple_territory_scope
			}
			if = {	# Mediocre Temple
				limit = {
					var:taurican_temple_quality < 3
				}
				trigger_event = {
					id = me_greek_5_pontus.50
				}
			}
			else_if = {	# Great Temple
				limit = {
					var:taurican_temple_quality >= 3
				}
				trigger_event = {
					id = me_greek_5_pontus.51
				}
			}
		}
		else = {	# Fallback cleanup
			hidden_effect = {
				remove_variable = taurican_temple_quality
				remove_variable = taurican_temple_progress
			}
		}
	}
}
