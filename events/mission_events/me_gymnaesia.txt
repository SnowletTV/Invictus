me_gymnaesian.5 = { # Remove Ibasim from the war and declare on them directly without carthage
    type = country_event
    title = me_gymnaesian_5_title
    desc = me_gymnaesian_5_desc
    picture = celtic_chiefs
    left_portrait = current_ruler
    right_portrait = c:IBA.current_ruler
    trigger = { 
        always = yes
    }

    option = {
        name = me_gymnaesian_5.a

        # trigger for player
        trigger = {
            is_ai = no
        }

        random_current_war = {
            limit = {
                any_war_participant = {
                    this = c:CAR #Carthage
                }
                any_war_participant = {
                    this = c:IBA #Ibasim
                }
                is_war_leader = c:CAR
            }
            remove_from_war = c:IBA
        }
        if = {
            limit = {
                c:IBA = { is_subject_of = c:CAR }
            }
            c:CAR = { release_subject = c:IBA }
        }
        declare_war_with_wargoal = {
            war_goal = conquer_wargoal
            target = c:IBA
            province = c:IBA.capital_scope
        }
    }

    option = {
        name = me_gymnaesian_5.a

        # trigger for AI - some bonuses otherwise the AI will keep repeating this mission and declaring war forever
        trigger = {
            is_ai = yes
        }

        random_current_war = {
            limit = {
                any_war_participant = {
                    this = c:CAR #Carthage
                }
                any_war_participant = {
                    this = c:IBA #Ibasim
                }
                is_war_leader = c:CAR
            }
            remove_from_war = c:IBA
        }
        if = {
            limit = {
                c:IBA = { is_subject_of = c:CAR }
            }
            c:CAR = { release_subject = c:IBA }
        }
        declare_war_with_wargoal = {
            war_goal = conquer_wargoal
            target = c:IBA
            province = c:IBA.capital_scope
        }

        create_unit = {
            navy = yes
            location = 1466
            add_subunit = trireme
            add_subunit = trireme
            add_subunit = trireme
            add_subunit = trireme
            add_subunit = trireme
            add_subunit = trireme
            add_subunit = trireme
            add_subunit = trireme
            add_subunit = trireme
        }

        add_country_modifier = {
            name = conquer_ibasim
            duration = 3650
        }
    }
}
