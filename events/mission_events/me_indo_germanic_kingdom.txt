me_indo_germanic_kingdom.6 = {
	type = country_event
	title = "me_indo_germanic_kingdom.6.t"
	desc = "me_indo_germanic_kingdom.6.desc"
	
	picture = revolt
	
	left_portrait = current_ruler
	right_portrait = scope:purge_target
	
	trigger = {
		NOT = { has_variable = tribal_reform_2_cooldown }
		trigger_if = {
			limit = {
				has_country_modifier = igk_mission_convince_elite_modifier
			}
			exists = scope:indgerFamily1
			scope:indgerFamily1 = {
				any_family_member = {
					employer = root
					is_ruler = no
					is_clan_chief = no
					OR = {
						has_any_office = yes
						is_governor = yes
						has_tech_office = yes
						is_general = yes
						is_admiral = yes
					}
				}
			}
		}
		trigger_else = {
			always = no
		}
	}
	
	immediate = {
		set_variable = {
			name = tribal_reform_2_cooldown
			days = 45
		}
		if = {
			limit = {
				has_country_modifier = igk_mission_convince_elite_modifier
			}
			scope:indgerFamily1 = {
				random_family_member = {
					limit = {
						employer = root
						is_ruler = no
						is_clan_chief = no
						OR = {
							has_any_office = yes
							is_governor = yes
							has_tech_office = yes
							is_general = yes
							is_admiral = yes
						}
					}
					save_scope_as = purge_target
				}
			}
		}
	}
	
	option = {
		name = "me_indo_germanic_kingdom.6.a"
		scope:purge_target = {
			remove_all_positions = yes
			if = {
				limit = {
					is_minor_character = no
				}
				family = {
					add_prestige = -100
				}
			}
		}
		if = {
			limit = {
				exists = scope:purge_target.family
				exists = scope:indgerFamily1
				scope:purge_target.family = scope:indgerFamily1
			}
			scope:purge_target.family = {
				every_family_member = {
					limit = {
						employer = root
					}
					custom = all_indgerFamily1_members
					add_loyalty = loyalty_purged_family
				}
			}
		}
		else_if = {
			limit = {
				exists = scope:purge_target.family
				exists = scope:republic_family
				scope:purge_target.family = scope:republic_family
			}
			scope:purge_target.family = {
				every_family_member = {
					limit = {
						employer = root
					}
					custom = all_republic_family_members
					add_loyalty = loyalty_purged_family
				}
			}
		}
		add_stability = 2
	}
	
	option = {
		name = "me_indo_germanic_kingdom.6.b"
		add_stability = -5
	}
	
	option = {
		name = "me_indo_germanic_kingdom.6.c"
		add_tyranny = 10
		add_stability = 5
		imprison = {
			target = scope:purge_target
		}
		if = {
			limit = {
				exists = scope:purge_target.family
				exists = scope:indgerFamily1
				scope:purge_target.family = scope:indgerFamily1
			}
			scope:purge_target.family = {
				every_family_member = {
					limit = {
						employer = root
					}
					custom = all_indgerFamily1_members
					add_loyalty = loyalty_purged_family_example
				}
			}
		}
	}
	
}
