    option = { # Don't hire
        name = "me_maeotia_20.b"
        add_treasury = 50
        current_ruler = {
            add_gold = 50
            add_corruption = 5
        }
        hidden_effect = {
            scope:maeotian_swamp_man = {
                death = {
					death_reason = death_flogging
				}
            }
        }
