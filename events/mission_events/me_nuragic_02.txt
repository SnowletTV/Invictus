        custom_tooltip = me_nuragic_02_inv_150_tooltip.a
        name = "me_nuragic_02_inv_150.a"
        add_political_influence = -75

        hidden_effect = {
            p:2108 = {
                create_country = {
                    name = {
                        name = SARDOMAVIA_FEUDATORY_NAME
                        adjective = SARDOMAVIA_FEUDATORY_ADJECTIVE
                    }
                    change_country_color = coracensian_blue
                    if = {
                        limit = {
                            NOT = {
                                primary_culture = root.culture
                            }
                        }
                        set_primary_culture_cleanup_effect = {
                            NEW_PRIMARY_CULTURE = root.culture
                            MAKE_OLD_CULTURE_INTEGRATED = yes
                        }
                    }	
                    set_country_religion = root.religion
                    change_government = oligarchic_republic
                    set_country_heritage = monapia_heritage
                    save_scope_as = sardomavia_country_scope
                }
            }
            make_subject = {
                target = scope:sardomavia_country_scope
                type = feudatory
            }
            reverse_add_opinion = {
                modifier = opinion_friendly_neighbor
                target = scope:sardomavia_country_scope
            }
			p:2108 = {
				define_pop = {
					type = nobles
					culture = root.culture
					religion = root.religion
				}
			}
		}
        
        trigger_event = { # Manavia Flavour 2
            id = me_nuragic_02_inv.151
            days = { 1 2 }
        }
    }
    
    option = { # No colony.
        custom_tooltip = me_nuragic_02_inv_150_tooltip.b
        name = "me_nuragic_02_inv_150.b"
