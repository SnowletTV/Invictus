# Large Event Chain
me_paropamisadae_macedon.4 = { # Food
    type = country_event

    title = me_paropamisadae_macedon_1_title
    desc = me_paropamisadae_macedon_1_desc
    picture = great_library

    immediate = {
        trigger_event = {
            id = me_paropamisadae_macedon.400
            days = 5
        }
        hidden_effect = {
             trigger_event = {
                id = me_paropamisadae_macedon.404 # Late Bandits
                days = 360
            }
        }
    }

    option = {
        name = "me_paropamisadae_macedon_1.a"
        #custom_tooltip = me_paropamisadae_macedon_1_tooltip
        add_treasury = -200
        add_country_modifier = {
            name = well_fed_and_rested
            duration = 2555
        }
        set_variable = indian_food
        hidden_effect = {
            if = {
                limit = {
                    NOT = { has_variable = easy_passage_india }
                }
                trigger_event = {
                    id = me_paropamisadae_macedon.403 # Indian Passage Bonus
                    days = 20
                }
            }
        }
    }
    option = {
        name = "me_paropamisadae_macedon_1.b"
        add_treasury = -100
        set_variable = indian_food
    }
    option = {
        name = "me_paropamisadae_macedon_1.c"
        add_treasury = -50
    }
}
