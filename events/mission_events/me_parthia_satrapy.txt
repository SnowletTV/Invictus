me_parthia_satrapy.1301 = { # try to subvert a governor 
	type = country_event
	title = me_parthia_satrapy.1301.t
	desc = me_parthia_satrapy.1301.desc
	picture = successors

	left_portrait = current_ruler
	right_portrait = overlord.current_ruler
	right_portrait = scope:cap_beguile_gov_char_1
	right_portrait = scope:cap_beguile_gov_char_2

	trigger = {
		OR = {
			region:media_region = {
				any_region_province = {
					owner = root.overlord
				}
			}
			region:ariana_region = {
				any_region_province = {
					owner = root.overlord
				}
			}
		}
	}

	immediate = {
		overlord = {
			random_owned_province = {
				limit = {
					OR = {
						is_in_region = media_region
						is_in_region = ariana_region
					}
				}
				governor = {
					save_scope_as = cap_beguile_gov_char_1
				}
			}
			if = {
				limit = {
					any_owned_province = {
						OR = {
							is_in_region = media_region
							is_in_region = ariana_region
						}
						NOT = {
							governor = scope:cap_beguile_gov_char_1
						}
					}
				}
				random_owned_province = {
					limit = {
						OR = {
							is_in_region = media_region
							is_in_region = ariana_region
						}
						NOT = {
							governor = scope:cap_beguile_gov_char_1
						}
					}
					governor = {
						save_scope_as = cap_beguile_gov_char_2
					}
				}
			}
		}
	}

	option = { # try to sway gov 1
		name = me_parthia_satrapy.1301.a
		trigger = {
			exists = scope:cap_beguile_gov_char_1
		}
		trigger_event = {
			id = me_parthia_satrapy.1302
			days = { 7 21 }
		}
		#add_political_influence = -25
	}

	option = { # try to sway gov 2
		name = me_parthia_satrapy.1301.b
		trigger = {
			exists = scope:cap_beguile_gov_char_2
		}
		trigger_event = {
			id = me_parthia_satrapy.1303
			days = { 7 21 }
		}
		#add_political_influence = -25
	}

	option = { # ignore, we can do it alone
		name = me_parthia_satrapy.1301.c
	}
}

me_parthia_satrapy.13 = { # declare imperial challenge cb on overlord to take their place.
	type = country_event
	title = me_parthia_satrapy.13.t
	desc = me_parthia_satrapy.13.desc
	picture = assassination

	left_portrait = current_ruler
	right_portrait = overlord.current_ruler
	left_portrait = scope:rebelling_governor_char

	immediate = {
		if = {
			limit = {
				overlord = {
					any_character = {
						has_variable = cap_suppoorting_coup
					}
				}
			}
			overlord = {
				random_character = {
					limit = { has_variable = cap_suppoorting_coup }
					save_scope_as = rebelling_governor_char
				}
			}
		}
		if = {
			limit = {
				exists = scope:rebelling_governor_char
				scope:rebelling_governor_char = { #might have been removed...
					is_governor = yes
				}
			}
			custom_tooltip = gov_defected_cap_tt
			scope:rebelling_governor_char = {
				hidden_effect = {
					every_governor_state = {
						every_state_province = {
							set_owned_by = root
							save_scope_as = to_set_as_gov
						}
					}
					remove_all_positions = yes
					move_country = root
					remove_character_modifier = foreign_citizen
					set_as_minor_character = THIS
					set_home_country = root
					hidden_effect = {
						if = {
							limit = {
								exists = scope:to_set_as_gov 
							}
							scope:to_set_governor = {
								set_as_governor = scope:rebelling_governor_char
							}
						}
					}
				}
			}
		}
	}

	option = {
		name = me_parthia_satrapy.13.a
		every_current_war = {
			remove_from_war = root
		}
		declare_war_with_wargoal = {
			war_goal = imperial_conquest_wargoal
			target = overlord
		}
	}
}
