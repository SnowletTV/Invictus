me_patauion_02.2 = {
	type = country_event 
	hidden = yes

	immediate = {
		if = {
			limit = {
				primary_culture = peloponnesian
			}
			random_integrated_culture = {
				limit = {
					this.culture = peloponnesian
				}
				save_scope_as = patauion_primary_culture
			}
		}
		if = {
			limit = {
				primary_culture = enetoian
			}
			random_integrated_culture = {
				limit = {
					this.culture = enetoian
				}
				save_scope_as = patauion_primary_culture
			}
		}
		trigger_event = { id = me_patauion_02.29 }
	}
}

me_patauion_02.28 = {
	type = country_event 
	title = me_patauion_02.28.t #Expansion of the Border lands
	desc = me_patauion_02.28.desc
	picture = greeks_marching
	left_portrait = current_ruler

	immediate = {
		every_owned_province = {
			limit = {
				is_in_area = ferraria_area
				NOT = {
					AND = {
						province_id = 3594
						dominant_province_culture = enetoian #if Patauion already has most pops made into Enetoi, its Venetic minority can stay
					}
				}
			}
			add_province_modifier = {
				name = me_assimilation_of_the_veneti_pat_pmod
				duration = 3650 #10 years
			}
		}
	}

	option = {
		name = me_patauion_02.28.a #Most excellent
		custom_tooltip =  "me_patauion_02.28.a_tt" #mission task finished
	}
}

		name = me_patauion_02.30.a #Let us give the Atestines their Autonomy

			p:3595 = {
                create_country = {
                    name = {
                        name = ATESTE_COUNTRY_NAME

                    }
                    change_country_color = red_nougat
                    change_country_flag = ATESTE_FLAG
                    set_country_heritage = atestine_heritage

                    add_treasury = 50

                        NEW_PRIMARY_CULTURE = venetic
                        MAKE_OLD_CULTURE_INTEGRATED = yes

                        name = me_loyal_venetic_polis_cmod
