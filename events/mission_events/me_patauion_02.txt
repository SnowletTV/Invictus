me_patauion_02.2 = {
	type = country_event 
	hidden = yes

	immediate = {
		if = {
			limit = {
				primary_culture = peloponnesian
			}
			random_integrated_culture = {
				limit = {
					this.culture = peloponnesian
				}
				save_scope_as = patauion_primary_culture
			}

			limit = {
				primary_culture = enetoian

	}
}

me_patauion_02.7 = {
	type = country_event 
	hidden = yes

	immediate = {
		every_country = {
            limit = {
            	AND = {
            		war_with = ROOT
            		capital_scope = {
                        OR = {
                            is_in_region = central_italy_region
                            is_in_region = magna_graecia_region
                        }
                    }
                    any_owned_province = {
                        count >= 25
                        is_in_region = central_italy_region
                    }
            	}
            }
            add_to_list = italic_powers_at_war_with_pat
		}
		every_province = {
			limit = {
				AND = {
					is_in_region = central_italy_region
					owner = {
						is_in_list = italic_powers_at_war_with_pat
					}
					controller = ROOT
				}
			}
			add_to_list = italic_provinces_controlled_by_pat_to_sack
		}
		trigger_event = { id = me_patauion_02.42 }
	}
}

me_patauion_02.28 = {
	type = country_event 
	title = me_patauion_02.28.t #Expansion of the Border lands
	desc = me_patauion_02.28.desc
	picture = greeks_marching
	left_portrait = current_ruler

	immediate = {
		every_owned_province = {
			limit = {
				is_in_area = ferraria_area
				NOT = {
					AND = {
						province_id = 3594
						dominant_province_culture = enetoian #if Patauion already has most pops made into Enetoi, its Venetic minority can stay
					}
				}
			}
			add_province_modifier = {
				name = me_assimilation_of_the_veneti_pat_pmod
				duration = 3650 #10 years
			}
		}
	}

	option = {
		name = me_patauion_02.28.a #Most excellent
		custom_tooltip =  "me_patauion_02.28.a_tt" #mission task finished
	}
}

me_patauion_02.30 = {
	type = country_event 
	title = me_patauion_02.30.t #The city of Ateste
	desc = me_patauion_02.30.desc
	picture = city_province
	left_portrait = current_ruler

	option = {
		name = me_patauion_02.30.a #Let us give the Atestines their Autonomy
		custom_tooltip = "me_patauion_02.30.a_tt"
		hidden_effect = {
			p:3595 = {
                create_country = {
                    name = {
                        name = ATESTE_COUNTRY_NAME
                        adjective = ATESTE_COUNTRY_ADJECTIVE
                    }
                    change_country_color = red_nougat
                    change_country_flag = ATESTE_FLAG
                    set_country_heritage = atestine_heritage
                    change_government = aristocratic_republic
                    add_treasury = 50
                    set_primary_culture_cleanup_effect = {
                        NEW_PRIMARY_CULTURE = venetic
                        MAKE_OLD_CULTURE_INTEGRATED = yes
                    }
                    set_country_religion = italic_pantheon
                    save_scope_as = atestine_country
                    add_country_modifier = {
                        name = me_loyal_venetic_polis_cmod
                        duration = 36500 #100 years
                    }
                }
                change_province_name = PROV3595_venetic
				remove_province_modifier = me_assimilation_of_the_veneti_pat_pmod
			}
			make_subject = {
                target = scope:atestine_country
                type = feudatory
            }
		}
		trigger_event = { id = me_patauion_02.31 }
	}

	option = {
		name = me_patauion_02.30.b #Let us give them freedom, but under a Patauian upper class
		custom_tooltip = "me_patauion_02.30.b_tt"
		hidden_effect = {
			p:3595 = {
				if = {
					limit = {
						ROOT = {
				    		has_variable = PAT_forging_a_new_path_for_patauion
				    	}
					}
				    random_pops_in_province = {
                    	while = {
                    		count = 2
                    		set_pop_culture = enetoian
                    		set_pop_religion = roman_pantheon
                    	}
                    }
                }
                else = {
				    random_pops_in_province = {
                    	while = {
                    		count = 2
                    		set_pop_culture = peloponnesian
                    		set_pop_religion = roman_pantheon
                    	}
                    }
                }
				create_country = {
                    name = {
                        name = ATESTE_COUNTRY_NAME
                        adjective = ATESTE_COUNTRY_ADJECTIVE
                    }
                    change_country_color = red_nougat
                    change_country_flag = ATESTE_FLAG
                    set_country_heritage = atestine_heritage
                    change_government = aristocratic_republic
                    add_treasury = 50
                    if = {
				    	limit = {
				    		ROOT = {
				    	    	has_variable = PAT_forging_a_new_path_for_patauion
				    	    }
				    	}
				        set_primary_culture_cleanup_effect = {
                            NEW_PRIMARY_CULTURE = enetoian
                            MAKE_OLD_CULTURE_INTEGRATED = yes
                        }
                    }
                    else = {
				        set_primary_culture_cleanup_effect = {
                            NEW_PRIMARY_CULTURE = peloponnesian
                            MAKE_OLD_CULTURE_INTEGRATED = yes
                        }
                    }
                    set_country_religion = roman_pantheon
                    create_character = {
                    	save_temporary_scope_as = patauian_char_to_rule
                    }
                    set_as_ruler = scope:patauian_char_to_rule
                    save_scope_as = atestine_country
                    add_country_modifier = {
                        name = me_loyal_venetic_polis_cmod
                        duration = 36500 #100 years
                    }
                }
			}
			make_subject = {
                target = scope:atestine_country
                type = feudatory
            }
		}
		show_as_tooltip = {
			p:33595 = {
				add_province_modifier = {
		    		name = me_assimilation_of_the_veneti_pat_pmod
		    		duration = 3650 #10 years
		    	}
			}
		}
		trigger_event = { id = me_patauion_02.31 }
	}

	option = {
		name = me_patauion_02.30.c #No, Ateste is ours now
		custom_tooltip = "me_patauion_02.30.c_tt" 
		show_as_tooltip = {
			p:33595 = {
				add_province_modifier = {
		    		name = me_assimilation_of_the_veneti_pat_pmod
		    		duration = 3650 #10 years
		    	}
			}
		}
		trigger_event = { id = me_patauion_02.31 }
	}
}
