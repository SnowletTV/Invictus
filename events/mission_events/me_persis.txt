me_persis.160 = {
    type = country_event
    title = me_persis.160.t
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = {
                    c:PES = { var:offer_flag = flag:only_prov }
                }
                desc = "me_persis.160.desc"
            }
            triggered_desc = {
                trigger = {
                    c:PES = { var:offer_flag = flag:entire_area }
                }
                desc = "me_persis.160.desc2"
            }
        }
    }
    picture = aqueducts
    left_portrait = current_ruler
    right_portrait = c:PES.current_ruler

    option = {  # Accept the offer
        name = me_persis.160.a
        add_opinion = {
            modifier = shared_wealth_pos_opmod
            target = c:PES
        }
        c:PES = {
            set_variable = {
                name = response_flag
                days = 90
                value = flag:accepted
            }
            trigger_event = {
                id = me_persis.161
                days = { 20 30 }
            }
        }
        ai_chance = {
            factor = 5
            modifier = {
                factor = 10
                c:PES = { var:offer_flag = flag:only_prov }
            }
        }
    }
    option = {  # Nah, fuck em
        name = me_persis.160.b
        add_opinion = {
            modifier = overlord_neg_opmod
            target = c:PES
        }
        c:PES = {
            set_variable = {
                name = response_flag
                days = 90
                value = flag:rejected
            }
            trigger_event = {
                id = me_persis.161
                days = { 20 30 }
            }
        }
        ai_chance = {
            factor = 3
            modifier = {
                factor = 2
                c:PES = { var:offer_flag = flag:entire_area }
            }
        }
    }
}

me_persis.161 = {
    type = country_event
    title = me_persis.161.t
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = {
                    var:response_flag = flag:accepted
                    var:offer_flag = flag:entire_area
                }
                desc = "me_persis.161.desc"
            }
            triggered_desc = {
                trigger = {
                    var:response_flag = flag:accepted
                    var:offer_flag = flag:only_prov
                }
                desc = "me_persis.161.desc2"
            }
            triggered_desc = {
                trigger = {
                    var:response_flag = flag:rejected
                    var:offer_flag = flag:only_prov
                }
                desc = "me_persis.161.desc3"
            }
            triggered_desc = {
                trigger = {
                    var:response_flag = flag:rejected
                    var:offer_flag = flag:entire_area
                }
                desc = "me_persis.161.desc4"
            }
        }
    }
    picture = aqueducts
    
    left_portrait = current_ruler

    option = {      
        name = me_persis.161.a
        trigger = {
            var:response_flag = flag:accepted
        }
        if = {
            limit = {
                var:offer_flag = flag:entire_area
            }
            p:4780.area = {
                every_area_province = {
                    limit = { owner = c:SEL }
                    set_owned_by = c:PES
                }
            }
            add_political_influence = 15
        }
        else = {    
            p:4780 = {
                set_owned_by = c:PES
            }
            add_political_influence = 10
        }
    }
    option = {      
        name = me_persis.161.b
        trigger = {
            var:response_flag = flag:rejected
        }
        add_opinion = {
            modifier = overlord_neg_opmod
            target = c:SEL
        }
        add_political_influence = -5
    }
}
