me_phoenicia.1301 = { # Recursive
    type = country_event
    hidden = yes
    immediate = {
        if = {
            limit = {
                has_global_variable = payday_melqart
                p:743 = {
                    is_holy_site = yes
                }
                owns_or_subject_owns = p:743
            }
            save_scope_as = melqart_temple_holder
            every_country = {
                limit = {
                    is_in_list = melqart_enjoyers
                    NOT = {
                        owns_or_subject_owns = p:743
                    }
                }
                trigger_event = {
                    id = me_phoenicia.1302 # Service for Melqart
                }
            }
            scope:melqart_temple_holder = {
                trigger_event = {
                    id = me_phoenicia.1303 # Payday Event
                    days = 30
                }
            }
            change_global_variable = {
                name = payday_melqart
                multiply = 0
            }
            trigger_event = {
                id = me_phoenicia.1301 # Recursive Recall
                days = 1825
            }
        }
        if = {
            limit = {
                OR = {
                    p:743 = {
                        is_holy_site = no
                    }
                    NOT = {
                        any_country = {
                            OR = { 
            #                   primary_culture = phoenician
                                primary_culture = carthaginian
                                }
                            OR = {
                                religion = carthaginian_pantheon
                                has_deity_in_pantheon = deity:omen_melqart
                                has_deity_in_pantheon = deity:omen_rasap_melqart
                            }
                            num_of_cities >= 1
                        }
                    }  
                }
            }
            remove_global_variable = payday_melqart
        }
    }
}
