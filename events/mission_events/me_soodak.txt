	type = country_event
	title = me_soodak_900_title
	desc = me_soodak_900_desc

		root = {
			random_character = {
				limit = {
					from_ruler_family = no
					is_head_of_family = no
					is_minor_character = no
					is_male = yes
				}
			save_scope_as = soodak_diplomatic_sacrifice_victim
			}
			
		}
		scope:soodak_diplomatic_sacrifice_victim.family = {
			head_of_family = {
				save_scope_as = soodak_diplomatic_sacrifice_victim_head
			}
		}
	}

	option = {
		name = me_soodak_900.a	
		add_treasury = -500
		capital_scope.state = {
			add_state_food = -5000
		}
	}
	option = {
		name = me_soodak_900.b
		add_treasury = -100
		capital_scope.state = {
			add_state_food = -500
		}
	}
	option = {
		name = me_soodak_900.c
		add_treasury = -50
		capital_scope.state = {
			add_state_food = -250
		}
	}
	option = {  
		name = me_soodak_900.d
		add_treasury = -100
		add_political_influence = 35
		capital_scope.state = {
			add_state_food = -50
		}
	}
	option = {
		name = me_soodak_900.e
	}
	option = {
		name = me_soodak_900.f
		custom_tooltip = "cruel_tooltip"
		trigger = {
			current_ruler = {
				has_trait = cruel
			}
		}	
		highlight = yes 
		scope:soodak_diplomatic_sacrifice_victim = {
			death = {
				killer = root.current_ruler
				death_reason = death_sacrificed_to_cannibals
			}
		}
		scope:soodak_diplomatic_sacrifice_victim_head = {
			add_zeal = 100
			add_loyalty = soodak_sacrificed_nephew
		}
	}
}
