south_arabia.37 = { # trade answer
	type = country_event
    title = south_arabia.37.t
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = {
                    has_variable = yes_but_money_var
                }
                desc = south_arabia.37.desc_1
            }
            triggered_desc = {
                trigger = {
                    has_variable = no_deal_saba_judea_var
                }
                desc = south_arabia.37.desc_2
            }
            triggered_desc = {
                trigger = {
                    always = yes
                }
                desc = south_arabia.37.desc_3
            }
        }
    }
    picture = council_country
    left_portrait = current_ruler
    right_portrait = c:SHB.current_ruler
    left_portrait = scope:marriage_target_root
    right_portrait = scope:marriage_target_judea

    immediate = {
        hidden_effect = {
            if = {
                limit = {
                    any_character = {
                        from_ruler_family = yes
                        is_adult = yes
                        is_married = no
                        prisoner = no
                        is_mercenary = no
                    }
                }
                random_character = {
                    limit = {
                        from_ruler_family = yes
                        is_adult = yes
                        is_married = no
                        prisoner = no
                        is_mercenary = no
                    }
                    save_scope_as = marriage_target_root
                }
            }
            else = {
                create_character = {
                    age = 25
                    female = no
                    set_family = root.current_ruler.family
                    save_scope_as = marriage_target_root
                }
            }
            c:JUD = {
                if = {
                    limit = {
                        any_character = {
                            from_ruler_family = yes
                            is_adult = yes
                            is_married = no
                            prisoner = no
                            is_mercenary = no
                            NOT = { is_same_gender = scope:marriage_target_root }
                        }
                    }
                    random_character = {
                        limit = {
                            from_ruler_family = yes
                            is_adult = yes
                            is_married = no
                            prisoner = no
                            is_mercenary = no
                            NOT = { is_same_gender = scope:marriage_target_root }
                        }
                        save_scope_as = marriage_target_judea
                        if = {
                            limit = {
                                is_female = yes
                            }
                            move_country = root
                        }
                        else = {
                            scope:marriage_target_root = {
                                move_country = c:JUD
                            }
                        }
                    }
                }
                else_if = {
                    limit = {
                        scope:marriage_target_root = {
                            is_female = no
                        }
                    }
                    create_character = {
                        age = 25
                        female = no
                        set_family = c:JUD.current_ruler.family
                        save_scope_as = marriage_target_judea
                    }
                }
                else_if = {
                    limit = {
                        scope:marriage_target_root = {
                            is_female = yes
                        }
                    }
                    create_character = {
                        age = 25
                        female = no
                        set_family = c:JUD.current_ruler.family
                        save_scope_as = marriage_target_judea
                    }
                }
            }
        }
    }

    option = { # yes
    	name = south_arabia.37.a
        trigger = {
            NOR = {
                has_variable = yes_but_money_var
                has_variable = no_deal_saba_judea_var
            }
        }
        add_country_modifier = {
            name = sabaean_judean_trade
            duration = 9125
        }
        hidden_effect = {
            c:JUD = {
                add_country_modifier = {
                    name = sabaean_judean_trade
                    duration = 9125
                }
            }
        }
        scope:marriage_target_root = {
            marry_character = scope:marriage_target_judea
        }
        add_opinion = {
            modifier = opinion_royal_marriage_ci
            target = c:JUD
        }
        reverse_add_opinion = {
            modifier = opinion_royal_marriage_ci
            target = c:JUD
        }
    }

    option = { # yes, but money --> pay
    	name = south_arabia.37.b
        trigger = {
            has_variable = yes_but_money_var
        }
        add_treasury = {
            value = has_monthly_income
            multiply = -2
        }
        add_country_modifier = {
            name = sabaean_judean_trade
            duration = 9125
        }
        hidden_effect = {
            add_treasury = {
                value = c:SHB.has_monthly_income
                multiply = 2
            }
            c:JUD = {
                add_country_modifier = {
                    name = sabaean_judean_trade
                    duration = 9125
                }
            }
        }
        scope:marriage_target_root = {
            marry_character = scope:marriage_target_judea
        }
        add_opinion = {
            modifier = opinion_royal_marriage_ci
            target = c:JUD
        }
        reverse_add_opinion = {
            modifier = opinion_royal_marriage_ci
            target = c:JUD
        }
    }

    option = { # yes, but money --> let it be
    	name = south_arabia.37.c
        trigger = {
            has_variable = yes_but_money_var
        }
    }

    option = { # no
    	name = south_arabia.37.d
        trigger = {
            has_variable = no_deal_saba_judea_var
        }
    }

    after = {
        if = {
            limit = {
                has_variable = yes_but_money_var
            }
            remove_variable = yes_but_money_var
        }
        if = {
            limit = {
                has_variable = no_deal_saba_judea_var
            }
            remove_variable = no_deal_saba_judea_var
        }
    }
}

south_arabia.40 = { # isarelite communities
	type = country_event
    title = south_arabia.40.t
    desc = south_arabia.40.desc
    picture = levantine_arabian_path_3_event
    left_portrait = current_ruler

    trigger = {
        has_variable = israelite_communities_events_var # gets set in an event triggering before this one (41)
    }

    option = { # accept
    	name = south_arabia.40.a
        set_variable = accepted_israelite_communities
        if = {
            limit = {
                var:israelite_communities_events_var >= 4 # out of 5 max
            }
            add_innovation = 2
            while = {
                count = 15
                random_owned_province = {
                    random_pops_in_province = {
                        limit = {
                            NOT = { pop_religion = judaism }
                        }
                        set_pop_religion = judaism
                    }
                }
            }
        }
        else_if = {
            limit = {
                var:israelite_communities_events_var >= 2 # effective end is 3
            }
            add_innovation = 2
            while = {
                count = 10
                random_owned_province = {
                    random_pops_in_province = {
                        limit = {
                            NOT = { pop_religion = judaism }
                        }
                        set_pop_religion = judaism
                    }
                }
            }
        }
        else_if = {
            limit = {
                var:israelite_communities_events_var >= 0 # effective end is 1
            }
            add_innovation = 1
            while = {
                count = 5
                random_owned_province = {
                    random_pops_in_province = {
                        limit = {
                            NOT = { pop_religion = judaism }
                        }
                        set_pop_religion = judaism
                    }
                }
            }
        }
        else_if = {
            limit = {
                var:israelite_communities_events_var < 0 
            }
            add_political_influence = -75
        }
        else = {
            add_political_influence = -100 # fallback, should not actually ever happen
        }
    }

    option = { # tolerate
    	name = south_arabia.40.b
        if = {
            limit = { var:israelite_communities_events_var >= 0 }
            add_stability = 10
            add_country_modifier = {
                name = accepted_israelite_communities_cmod
                duration = 9125 # 25 years
            }
        }
        else_if = {
            limit = { var:israelite_communities_events_var >= -2 }
            add_stability = 5
            add_country_modifier = {
                name = accepted_israelite_communities_cmod
                duration = 3650 # 10 years
            }
        }
        else = {
            add_political_influence = -75
            add_country_modifier = {
                name = accepted_israelite_communities_cmod
                duration = 1825 # 5 years
            }
        }
    }

    option = { # expel
    	name = south_arabia.40.c
        custom_tooltip = expelled_pops_move_to_judea_tt
        hidden_effect = {
            every_owned_province = {
                every_pops_in_province = {
                    limit = {
                        pop_religion = judaism
                    }
                    move_pop = p:687 # Jerusalem
                }
            }
        }
        p:687.owner = {
            trigger_event = {
                id = south_arabia.99 # notification
            }
        }
        if = {
            limit = {
                p:687.owner = {
                    religion = judaism
                }
            }
            p:687.owner = {
                add_opinion = {
                    modifier = expelled_israelites
                    target = root
                }
            }
        }
        add_treasury = 400
        if = {
            limit = { var:israelite_communities_events_var <= -3 }
            add_military_experience = 50
            add_political_influence = 50
        }
        else_if = {
            limit = { var:israelite_communities_events_var < -1 }
            add_military_experience = 25
            add_political_influence = 25
        }
        else_if = {
            limit = { var:israelite_communities_events_var >= 0 }
            add_political_influence = -100
            add_military_experience = 15
        }
    }
}
