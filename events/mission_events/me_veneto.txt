    option = { #Very well, let us integrate further with rome!
        name = secure_venetia_mission_event.22.a 
        custom_tooltip = secure_venetia_mission_event.22.a.tt
        hidden_effect = {
            change_country_color = roman_red
            if = {
            	limit = {
            		is_tribal = yes
            	}
                change_government = aristocratic_republic
            }
            change_country_flag = VENETIA_ET_HISTRIA_FLAG
            root = {
                if = {   
                    limit = {
    	                area:histria_area = {
    	    	            any_area_province = {
	        		            count >= 5
	        		            owned_or_subject_owned = root
    	    	            }
    	                }
                    }
                    change_country_name = VENETIA_ET_HISTRIA
                }
                else = {
                    change_country_name = VENETIA
                }
            }
            if = {
                limit = {
                    NOT = {
                        is_subject_type = subject_colony
                    }
                }
                c:ROM = {
                    release_subject = ROOT
                    make_subject = {
                        target = ROOT
                        type = feudatory
                    }
                }
            }
        }
        ai_chance = {
            factor = 85
        }
    }
    option = { #This demand is a bit much, even from Rome
        name = secure_venetia_mission_event.22.b
        custom_tooltip = secure_venetia_mission_event.22.b.tt
