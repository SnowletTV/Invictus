punic_resurgence.16 = { # enemy force landed
	type = country_event
	title = punic_resurgence_16_title

	hidden = yes

	desc = punic_resurgence_16_desc

	picture = roman_trial

	trigger = {
		tag = CAR
		c:CAR = {
			any_current_war = {
				any_war_participant = {
					NOT = {
						any_province = {
							OR = {
								is_in_region = africa_region
								is_in_region = numidia_region
								is_in_region = mauretainia_region
							}
						}
					}
					any_unit = {
						unit_location = {
							OR = {
								is_in_region = africa_region
								is_in_region = numidia_region
								is_in_region = mauretainia_region
							}
						}
						NOT = {
							has_variable = flagged
						}
					}
				}
			}
			any_subject  = {
				has_variable = protectors_of_africa
			}
		}

	}

	immediate = {
		c:CAR = {
			every_current_war = {
				every_war_participant = {
					every_unit = {
						limit = {
							unit_location = {
								OR = {
									is_in_region = africa_region
									is_in_region = numidia_region
									is_in_region = mauretainia_region
								}
							}
							NOT = {
								has_variable = flagged
							}
						}
						set_variable = flagged
					}
				}
			}
		}
		if = {
			limit = {
				c:CAR = {
					NOT = { has_variable = strikes }
				}
			}
			c:CAR = {
				set_variable = {
					name = strikes
					value = 1
				}
			}
		}
		else = {
			c:CAR = {
				change_variable = {
					name = strikes
					add = 1
				}
			}
		}
		if = {
			limit = {
				c:CAR = {
					var:strikes > 15
				}
			}
			c:CAR = {
				every_subject = {
					limit = {
						has_variable = punic_league_leader
					}
					trigger_event = {
						id = punic_resurgence.17
						days = 0
					}
				}
			}
		}
	}

	#if this event should be shown, it needs an option:
	#nope just a hidden event
	option = {
		name = punic_resurgence_16.a
	}

}

punic_resurgence.18 = { # Carthage requests a fleet
	type = country_event
	title = punic_resurgence_18_title

	desc = punic_resurgence_18_desc

	picture = roman_trial

	trigger = {
		c:CAR = {
			war = yes
			NOT = { has_variable = fleet_cooldown }
			has_variable = boatsman
		}

	}
	immediate = {
		c:CAR = {
			remove_variable = happy
			remove_variable = sad
			set_variable = {
				name = fleet_cooldown
				days = 730
			}
		}
	}


	option = {
		trigger = {
			num_of_ships > 0
		}
		name = punic_resurgence_18.a
		every_navy = {

			change_unit_owner = c:CAR
			save_scope_as = boat
			set_variable = {
				name = num_boats
				value = unit_size
			}
			c:CAR = {
				if = {
					limit = {
						has_variable = boats
					}
					change_variable = {
						name = boats
						add = scope:boat.var:num_boats
					}
				}
				else = {
					set_variable = {
						name = boats
						value = scope:boat.var:num_boats
					}
				}
			}
		}
		if = {
			limit = {
				c:CAR.var:boats > 15
			}
			c:CAR = {
				set_variable = happy
				add_opinion = {
					modifier = many_boats
					target = scope:mission_country
				}
			}
		}
		else = {
			c:CAR = {
				set_variable = sad
				add_opinion = {
					modifier = few_boats
					target = scope:mission_country
				}
			}
		}
	}
	option = {

		name = punic_resurgence_18.b
		c:CAR = {
			add_opinion = {
				modifier = no_boats
				target = root
			}
		}
	}
}
