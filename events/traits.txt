	hidden = yes
	
	trigger = {
		exists = employer
		has_any_infection = yes
		NOT = {
			has_ambition = scheme_seek_treatment
		}
	}
	
	immediate = {
		random_list = {
			10 = {
				trigger = {
					NOT = {
						has_trait = pneumonia
					}
					OR = {
						has_trait = infection
						has_trait = inflammation
					}
				}
				set_variable = {
					name = second_stage_disease
					value = 1
				}
				if = {
					limit = {
						exists = employer
						is_important_person_medical = yes
					}
					save_scope_as = diseased_personage
					employer = {
						trigger_event = {
							id = traits.7
						}
					}
				}
				else = {
					if = {
						limit = {
							has_trait = infection
						}
						remove_trait = infection
					}
					else_if = {
						limit = {
							has_trait = inflammation
						}
						remove_trait = inflammation
					}
					add_trait = pneumonia
				}
			}
			10 = {
				trigger = {
					NOT = {
						has_trait = leper
					}
					OR = {
						has_trait = infection
						has_trait = inflammation
					}
				}
				set_variable = {
					name = second_stage_disease
					value = 2
				}
				if = {
					limit = {
						exists = employer
						is_important_person_medical = yes
					}
					save_scope_as = diseased_personage
					employer = {
						trigger_event = {
							id = traits.7
						}
					}
				}
				else = {
					if = {
						limit = {
							has_trait = infection
						}
						remove_trait = infection
					}
					else_if = {
						limit = {
							has_trait = inflammation
						}
						remove_trait = inflammation
					}
					add_trait = leper
				}
			}
			10 = {
				trigger = {
					is_adult = yes
					NOT = {
						has_trait = pox
					}
					OR = {
						has_trait = infection
						has_trait = inflammation
					}
				}
				set_variable = {
					name = second_stage_disease
					value = 3
				}
				if = {
					limit = {
						exists = employer
						is_important_person_medical = yes
					}
					save_scope_as = diseased_personage
					employer = {
						trigger_event = {
							id = traits.7
						}
					}
				}
				else = {
					if = {
						limit = {
							has_trait = infection
						}
						remove_trait = infection
					}
					else_if = {
						limit = {
							has_trait = inflammation
						}
						remove_trait = inflammation
					}
					add_trait = pox
				}
			}
			10 = {
				trigger = {
					NOT = {
						has_trait = gangrene
					}
					OR = {
						has_trait = wounded
						has_trait = maimed
					}
				}
				set_variable = {
					name = second_stage_disease
					value = 4
				}
				if = { 
					limit = {
						exists = employer
						is_important_person_medical = yes
					}
					save_scope_as = diseased_personage
					employer = {
						trigger_event = {
							id = traits.7
						}
					}
				}
				else = {
					add_trait = gangrene 
				}
			}
		}
	}
}

traits.8 = { #resulting permanent illness
	type = character_event
	hidden = yes
	
	trigger = {
		exists = employer
		NOT = {
			has_ambition = scheme_seek_treatment
		}
		OR = {
			has_trait = plague
			has_trait = hemorrhagic_fever
			has_trait = pneumonia
			has_trait = gangrene
			has_trait = maimed
			has_trait = pox
		}
		NOR = {
			has_trait = barren
			has_trait = castrated
		}
		NOT = {
			has_variable = seeking_treatment
		}
	}
	
	immediate = {
		if = { 
			limit = {
				exists = employer
				is_female = yes
				is_important_person_medical = yes
			}
			set_variable = {
				name = medical_result_acquired
				value = 1
			}
			save_scope_as = result_acquired_target
			employer = {
				trigger_event = {
					id = traits.9
				}
			}
		}
		else_if = { 
			limit = {
				exists = employer
				is_female = no
				is_important_person_medical = yes
			}
			set_variable = {
				name = medical_result_acquired
				value = 2
			}
			save_scope_as = result_acquired_target
			employer = {
				trigger_event = {
					id = traits.9
				}
			}
		}
		else_if = {
			limit = {
				is_female = yes
			}
			add_trait = barren
		}
		else = {
			add_trait = castrated
		}
	}
}
