namespace = tribal_pol

tribal_pol.1 = { # The tribes are decentralizing because of instability
	type = country_event 
	title = "tribal_pol.1.t"
	desc = "tribal_pol.1.desc"
	picture = celtic_chiefs
	
	left_portrait = root.current_ruler
	right_portrait = scope:warchief_official
	right_portrait = scope:magistrate_official
	
	goto_location = scope:randomized_country_province

	trigger = {
		is_tribal = yes
		NOT = { has_variable = tribal_pol_1_cooldown }
		centralization >= 0
		stability <= 50
		any_character = {
			has_office = office_warchief
		}
		any_character = {
			has_office = office_magistrate
		}
		num_of_cities >= 1
	}

	immediate = {
		set_variable = {
			name = "tribal_pol_1_cooldown"
			years = 15
		}
		random_owned_province = {
			save_scope_as = randomized_country_province
		}
		random_character = {
			limit = {
				has_office = office_warchief
			}
			save_scope_as = warchief_official
		}
		random_character = {
			limit = {
				has_office = office_magistrate
			}
			save_scope_as = magistrate_official
		}
	}

	option = {
		name = tribal_pol.1.a # Use Military Power to force people to centralize
		add_centralization = centralization_large
		add_tyranny = tyranny_medium
	}

	option = { # Use Oratory Power to force people to centralize
		name = tribal_pol.1.b
		trigger = {
			NOT = {
				root.current_ruler = {
					has_trait = silver_tongued
				}
			}
		}
		add_political_influence = subtract_influence_medium
		add_centralization = centralization_large
	}

	option = { # More effective use of Oratory Power to force people to centralize
		highlight = yes
		name = tribal_pol.1.d
		custom_tooltip = "silver_tongued_tooltip"
		trigger = {
			root.current_ruler = {
				has_trait = silver_tongued
			}
		}
		add_political_influence = subtract_influence_small
		add_centralization = centralization_huge
	}

	option = { # Let people decentralize
		name = tribal_pol.1.c
		add_centralization = subtract_centralization_large
	}
}

tribal_pol.3 = { # An Official suggests way to increase Centralization

				has_office = office_magistrate

		}

			has_country_modifier = centralization_efforts

				OR = {
					has_office = office_magistrate

					has_office = office_arbitrator
				}
				is_ruler = no
			}
			save_scope_as = randomized_official
		}
	}

	option = { # Pay for more centralization
		name = tribal_pol.3.a
		add_treasury = {
			subtract = root.has_monthly_income
			multiply = 5
		}
		add_centralization = centralization_medium
		add_country_modifier = {

	option = { # Ignore your official, and get less centralization and some disloyalty from him/her

			name = natural_centralization
			duration = 1825
		}
		scope:randomized_official = {
			add_loyalty = loyalty_advice_ignored_bad
		}
	}
}
