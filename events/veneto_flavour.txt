veneto_flavour_event.4 = { #Rome or etruria borders Veneto
    type = country_event
    hidden = yes
	
    immediate = {
        ROOT = {
            if = {
                limit = {
                    NOT = {
                        has_variable = has_bordered_great_italic_power_veneto
                    }
                }
                set_variable = {
                    name = has_bordered_great_italic_power_veneto
                }
            }
        }
        if = {
            limit = {
                OR = {
                    any_neighbour_country = {
                        count >= 1
                        OR = {
                            tag = ROM
                            tag = ETR
                        }
                        num_of_cities >= 30
                    }
                }
                NOR = {
                    has_completed_mission_task = secure_venetia_mission_task_9
                    has_completed_mission_task = secure_venetia_mission_task_11
                    current_date >= 550
                }
            }
            if = {
                limit = {
                    AND = {
                        any_neighbour_country = {
                            count >= 1
                            tag = ROM
                            num_of_cities >= 30
                        }  
                        NOR = {
                            alliance_with = c:ROM
                            alliance_with = c:ETR
                        }  
                        opinion = {
                            target = c:ROM
                            value >= -20
                        } 
                    }
                }
                trigger_event = {
                    id = veneto_flavour_event.5
                }
                remove_variable = has_bordered_great_italic_power_veneto
            }
            else_if = {
                limit = {
                    AND = {
                        any_neighbour_country = {
                            count >= 1
                            tag = ETR
                            num_of_cities >= 30
                        }  
                        NOR = {
                            alliance_with = c:ROM
                            alliance_with = c:ETR
                        }   
                        opinion = {
                            target = c:ETR
                            value >= -20
                        } 
                    }
                }
                trigger_event = {
                    id = veneto_flavour_event.22
                }
                remove_variable = has_bordered_great_italic_power_veneto
            }
            else_if = {
                limit = { 
                    AND = {
                        NOR = {
                            alliance_with = c:ROM
                            alliance_with = c:ETR
                        }   
                        c:ETR = {
                            has_land = yes
                        }
                        c:ROM = {
                            has_land = yes
                        }
                    }
                }
                trigger_event = {
                    id = veneto_flavour_event.24
                }
                remove_variable = has_bordered_great_italic_power_veneto
            }
            else = {
                trigger_event = {
                    id = veneto_flavour_event.4
                    days = 1825 #5 years
                }
                remove_variable = has_bordered_great_italic_power_veneto
            }
        }
        else_if = {
            limit = {
                current_date >= 550
            }
            remove_variable = has_bordered_great_italic_power_veneto
        }
        else = {
            trigger_event = {
                id = veneto_flavour_event.4
                days = 7200 #20 years
            }
        }
    }
}
